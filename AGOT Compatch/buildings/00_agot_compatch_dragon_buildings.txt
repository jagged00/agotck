# Vanilla-safe dragonpit chain for dragon/dragonrider gameplay.
# This intentionally avoids hard ties to AGOT-only provinces/titles.

generic_dragon_pit_01 = {
	construction_time = very_slow_construction_time

	can_construct_potential = {
		agot_can_build_or_restore_dragonpits = yes
	}

	can_construct_showing_failures_only = {
		building_requirement_tribal = no
	}

	effect_desc = building_dragonpit_effect

	on_complete = {
		county.holder = {
			trigger_event = agot_dragon_pits.0005
		}
	}

	cost_gold = expensive_building_tier_8_cost

	province_modifier = {
		monthly_income = -0.5
	}
	county_modifier = {
		county_opinion_add = -10
	}
	character_modifier = {
		dread_gain_mult = 0.05
		monthly_dynasty_prestige_mult = 0.05
		owned_legend_spread_mult = 0.05
		intimidated_vassal_tax_contribution_mult = 0.05
	}
	county_holder_character_modifier = {
		dread_gain_mult = 0.05
		monthly_dynasty_prestige_mult = 0.05
		owned_legend_spread_mult = 0.05
		intimidated_vassal_tax_contribution_mult = 0.05
	}
	type_icon = "icon_building_dragonpit.dds"

	ai_value = {
		base = 0
		modifier = {
			add = 1000
			agot_is_any_dragon_alive = yes
			scope:holder ?= {
				OR = {
					any_relation = { type = agot_dragon }
					any_child = {
						any_relation = { type = agot_dragon }
					}
				}
			}
		}
	}
}
