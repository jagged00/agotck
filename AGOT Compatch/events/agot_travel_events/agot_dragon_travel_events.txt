# Do A FLIP (0001-0004) #
# by: Steelers          #

agot_travel_events_dragon.0001 = {
	type = character_event
	title = agot_travel_events_dragon.0001.t
	desc = agot_travel_events_dragon.0001.desc
	theme = dragon
	override_background = {
		reference = agot_open_sky_video
	}
	right_portrait = {
		character = root
		camera = camera_dragonrider_ride_dragon_decision
		animation = dragon_main
		outfit_tags = { dragonrider linear_camera_zoom }
	}

	cooldown = { years = 30 }

	immediate = {
		add_character_flag = {
			flag = had_recent_dragon_travel_event
			months = { 6 9 }
		}
		agot_set_dragon_animation_flag_effect = {
			FLAG = dragon_hover
			RIDING_DRAGON = yes
		}
		var:current_dragon = {
			save_scope_as = pet_dragon
		}
	}

	option = { # Try to do a flip!
		name = agot_travel_events_dragon.0001.a

		random_list = {
			50 = { # Success - Give a decent prestige and dragonrider exp + potential modifier
				desc = agot_travel_events_dragon.0001.success.flip_past
				modifier = { # Dragon's temperament value
					add = {
						value = scope:pet_dragon.temperament
						multiply = 0.5
					}
				}
				modifier = { # Has lvl 1 rider training/bond
					add = 10
					OR = {
						has_trait_xp = {
							trait = dragonrider
							track = dragon_training
							value >= 30 # Level 1
						}
						has_trait_xp = {
							trait = dragonrider
							track = dragon_bond
							value >= 30 # Level 1
						}
					}
				}
				modifier = { # Has lvl 2 rider training/bond
					add = 10
					OR = {
						has_trait_xp = {
							trait = dragonrider
							track = dragon_training
							value >= 65 # Level 2
						}
						has_trait_xp = {
							trait = dragonrider
							track = dragon_bond
							value >= 65 # Level 2
						}
					}
				}
				modifier = { # Has lvl 3 rider training/bond
					add = 10
					OR = {
						has_trait_xp = {
							trait = dragonrider
							track = dragon_training
							value >= 100 # Level 3
						}
						has_trait_xp = {
							trait = dragonrider
							track = dragon_bond
							value >= 100 # Level 3
						}
					}
				}

				add_prestige = major_prestige_gain
				hidden_effect = {
					send_interface_toast = {
						title = agot_travel_events_dragon.0001.success
						custom_tooltip = agot_travel_events_dragon.0001.success.flip
						left_icon = root
						right_icon = scope:pet_dragon
					}
				}
				add_trait_xp = {
					trait = dragonrider
					track = dragon_training
					value = 10
				}
				add_trait_xp = {
					trait = dragonrider
					track = dragon_bond
					value = 10
				}
			}
			50 = { # Failure - Loss a decent prestige and dragonrider exp + potential injury
				desc = agot_travel_events_dragon.0001.failure.flip_past
				modifier = { # Dragon's temperament value
					add = {
						value = 0
						subtract = scope:pet_dragon.temperament
						multiply = 0.5
					}
				}
				modifier = { # Has lvl 1 rider training/bond
					add = -10
					OR = {
						has_trait_xp = {
							trait = dragonrider
							track = dragon_training
							value >= 30 # Level 1
						}
						has_trait_xp = {
							trait = dragonrider
							track = dragon_bond
							value >= 30 # Level 1
						}
					}
				}
				modifier = { # Has lvl 2 rider training/bond
					add = -10
					OR = {
						has_trait_xp = {
							trait = dragonrider
							track = dragon_training
							value >= 65 # Level 2
						}
						has_trait_xp = {
							trait = dragonrider
							track = dragon_bond
							value >= 65 # Level 2
						}
					}
				}
				modifier = { # Has lvl 3 rider training/bond
					add = -10
					OR = {
						has_trait_xp = {
							trait = dragonrider
							track = dragon_training
							value >= 100 # Level 3
						}
						has_trait_xp = {
							trait = dragonrider
							track = dragon_bond
							value >= 100 # Level 3
						}
					}
				}

				add_prestige = medium_prestige_loss
				custom_tooltip = agot_travel_events_dragon.0001.injury_warning
				hidden_effect = {
					send_interface_toast = {
						title = agot_travel_events_dragon.0001.failure
						custom_tooltip = agot_travel_events_dragon.0001.failure.flip
						left_icon = root
						right_icon = scope:pet_dragon
						random_list = { # Base 1/3 chance to get injured
							200 = { modifier = { add = prowess } } # Nothing
							1 = { add_trait = one_legged }
							1 = { add_trait = maimed }
							1 = { add_trait = disfigured }
							7 = { add_trait = incapable }
							10 = { add_trait = scarred }
							20 = { add_trait = wounded_3 }
							25 = { add_trait = wounded_2 }
							35 = { add_trait = wounded_1 }
						}
					}
				}
				add_trait_xp = {
					trait = dragonrider
					track = dragon_training
					value = -2
				}
				add_trait_xp = {
					trait = dragonrider
					track = dragon_bond
					value = -2
				}
			}
		}

		ai_chance = {
			base = 15
			modifier = { # The more irrational, the more likely to do this
				add = {
					value = 0
					subtract = ai_rationality
				}
			}
		}
	}

	option = { # No, it's not safe...
		name = agot_travel_events_dragon.0001.b

		send_interface_toast = {
			title = agot_travel_events_dragon.0001.no_attempt
			custom_tooltip = agot_travel_events_dragon.0001.no_attempt.flip
			left_icon = root
			right_icon = scope:pet_dragon
		}

		ai_chance = {
			base = 85
			modifier = { # The more rational, the less likely to do a flip
				add = ai_rationality
			}
		}
	}
}

# Determine if the a valid noble exists for the noble option
scripted_trigger agot_travel_events_dragon_0006_valid_noble_trigger = {
	root.location.province_owner ?= {
		OR = {
			AND = {  # The province owner
				is_available_adult = yes
				location = root.location
			}
			father ?= {
				is_available_adult = yes
				location = root.location
			}
			mother ?= {
				is_available_adult = yes
				location = root.location
			}
			primary_spouse ?= {
				is_available_adult = yes
				location = root.location
			}
			any_child = {
				is_available_adult = yes
				location = root.location
			}
			any_court_position_holder = {
				is_available_adult = yes
				location = root.location
			}
		}
	}
}

# 0006 - Dragon gets antsy flying over some woods
agot_travel_events_dragon.0006 = {
	type = character_event
	title = agot_travel_events_dragon.0006.t
	desc = agot_travel_events_dragon.0006.desc
	theme = dragon
	override_background = {
		reference = agot_open_sky_video
	}
	right_portrait = {
		character = root
		camera = camera_dragonrider_ride_dragon_decision
		animation = dragon_main
		outfit_tags = { dragonrider linear_camera_zoom }
	}

	trigger = {
		root.location = { # Make sure a wooded area
			OR = {
				agot_is_forest_terrain = yes
				agot_is_taiga_terrain = yes
				agot_is_jungle_terrain = yes
			}
		}
	}

	cooldown = { years = 20 }

	immediate = {
		add_character_flag = {
			flag = had_recent_dragon_travel_event
			months = { 6 9 }
		}
		agot_set_dragon_animation_flag_effect = {
			FLAG = dragon_hover
			RIDING_DRAGON = yes
		}
		var:current_dragon = {
			save_scope_as = pet_dragon
		}

		random_list = {
			20 = {
				set_variable = nothing_in_woods
			}
			80 = {
				random_list = {
					20 = { set_variable = pred }
					20 = { set_variable = prey }
					20 = { set_variable = good }
					20 = { set_variable = bad }
					20 = {
						trigger = { agot_travel_events_dragon_0006_valid_noble_trigger = yes }
						set_variable = noble
					}
				}
			}
		}
	}

	option = { # Ignore it, keep flying
		name = agot_travel_events_dragon.0006.a
		custom_tooltip = agot_travel_events_dragon.0006.a.tt

		# Dragon loses opinion/bond of you for ignoring them
		add_trait_xp = {
			trait = dragonrider
			track = dragon_bond
			value = -2
		}

		ai_chance = {
			base = 100
		}

		remove_variable = nothing_in_woods
		remove_variable = prey
		remove_variable = pred
		remove_variable = good
		remove_variable = bad
		remove_variable = noble
	}

	option = { # Check it out, land
		name = agot_travel_events_dragon.0006.b
		custom_tooltip = agot_travel_events_dragon.0006.b.tt

		if = {
			limit = {
				exists = var:nothing_in_woods
			}
			trigger_event = agot_travel_events_dragon.0007
		}
		else_if = {
			limit = {
				OR = {
					exists = var:pred
					exists = var:prey
				}
			}
			trigger_event = agot_travel_events_dragon.0008
		}
		else_if = {
			limit = {
				OR = {
					exists = var:good
					exists = var:bad
					exists = var:noble
				}
			}
			trigger_event = agot_travel_events_dragon.0009
		}

		ai_chance = {
			base = 0
		}
	}
}

# 0008 - Check it out, land - animal
agot_travel_events_dragon.0008 = {
	type = character_event
	title = agot_travel_events_dragon.0008.t
	desc = {
		desc = agot_travel_events_dragon.0008.desc
		first_valid = {
			triggered_desc = {
				trigger = { exists = var:pred }
				desc = agot_travel_events_dragon.0008.desc.pred
			}
			triggered_desc = {
				trigger = { exists = var:prey }
				desc = agot_travel_events_dragon.0008.desc.prey
			}
		}
	}
	theme = dragon
	override_background = {
		trigger = {
			exists = scope:background_terrain_scope
		}
		reference = terrain_scope # wilderness_forest
	}
	left_portrait = {
		character = root
		triggered_animation = {
			trigger = { has_trait = brave }
			animation = personality_bold
		}
		triggered_animation = {
			trigger = { has_trait = craven }
			animation = personality_coward
		}
		triggered_animation = {
			trigger = { ai_rationality <= -50 }
			animation = personality_irrational
		}
		animation = aggressive_sword
	}
	right_portrait = {
		character = scope:pet_dragon
		camera = camera_dragon_event_standing
		animation = dragon_main
	}

	immediate = {
		root.location = {
			save_scope_as = background_terrain_scope
		}
		var:current_dragon = {
			save_scope_as = pet_dragon
		}
	}

	option = { # Something - Engage - Get it!
		name = {
			trigger = { agot_can_use_valyrian_dragon_commands_trigger = yes }
			text = agot_travel_events_dragon.0008.a.valyrian
		}
		name = {
			trigger = { NOT = { agot_can_use_valyrian_dragon_commands_trigger = yes } }
			text = agot_travel_events_dragon.0008.a
		}

		var:pred ?= {
			prev = {
				add_trait_xp = {
					trait = dragonrider
					track = dragon_training
					value = 10
				}
				add_trait_xp = {
					trait = dragonrider
					track = dragon_bond
					value = 10
				}
			}
		}
		var:prey ?= {
			prev = {
				add_trait_xp = {
					trait = dragonrider
					track = dragon_training
					value = 5
				}
				add_trait_xp = {
					trait = dragonrider
					track = dragon_bond
					value = 5
				}
			}
		}

		stress_impact = {
			craven = miniscule_stress_impact_gain
			compassionate = miniscule_stress_impact_gain
		}

		ai_chance = {
			base = 50
			modifier = {
				add = 25
				has_trait = lifestyle_hunter
			}
			ai_value_modifier = {
				ai_boldness = low_chance_impact_positive_ai_value
				ai_compassion = low_chance_impact_negative_ai_value
			}
		}
	}

	option = { # Something - Disengage - RUN!
		name = agot_travel_events_dragon.0008.b
		trigger = {
			OR = {
				exists = var:pred
				AND = {
					exists = var:prey
					has_trait = craven
				}
			}
