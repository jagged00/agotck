agot_silly_mode.0002 = {
	type = character_event
	title = agot_silly_mode.0002.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					list_size:cracked_eggs = 1
				}
				desc = agot_silly_mode.0002.desc
			}
			desc = agot_silly_mode.0002.desc_mult
		}
	}

	theme = prison
	override_background = {
		reference = agot_dragonpit
	}

	left_portrait = {
		character = scope:actor
		animation = worry
	}
	right_portrait = {
		character = scope:recipient
		animation = throne_room_bow_1
	}

	lower_left_portrait = scope:eggo2
	lower_center_portrait =  scope:eggo
	lower_right_portrait = scope:eggo3

	immediate = {
		every_in_list = {
			list = cracked_eggs
			hard_imprison_character_effect = {
				TARGET = this
				IMPRISONER = scope:actor
			}
			if = {
				limit = {
					NOT = { exists = scope:eggo }
				}
				save_scope_as = eggo
			}
			else_if = {
				limit = {
					NOT = { exists = scope:eggo2 }
				}
				save_scope_as = eggo2
			}
			else_if = {
				limit = {
					NOT = { exists = scope:eggo3 }
				}
				save_scope_as = eggo3
			}
		}
	}

	option = {
		name = agot_silly_mode.0002.a
		every_in_list = {
			list = cracked_eggs
			release_from_prison = yes
		}
	}

	option = {
		name = agot_silly_mode.0002.b
	}
}

agot_silly_mode.0100 = {
	type = character_event
	theme = friendly
	title = agot_silly_mode.0100.t
	desc = agot_silly_mode.0100.desc

	left_portrait = {
		character = scope:allard
		animation = toast_goblet
		hide_info = yes
	}

	right_portrait = {
		character = root
		animation = happiness
	}

	immediate = {
		if = {
			limit = {
				character:Scaly_1 ?= { is_alive = yes }
			}
			character:Scaly_1 = {
				save_scope_as = allard
			}
		}
		else_if = {
			limit = {
				any_living_character = {
					has_character_flag = is_scaly_allard
				}
			}
			random_living_character = {
				limit = {
					has_character_flag = is_scaly_allard
				}
				save_scope_as = allard
			}
		}
		else_if = {
			limit = {
				exists = character:Scaly_1
			}
			create_character = {
				template = agot_scaly_dragonseed_template
				employer = root
				age = 30
				gender = male
				save_scope_as = allard
				after_creation = {
					copy_inheritable_appearance_from = character:Scaly_1
				}
			}
		}
		else = {
			create_character = {
				template = agot_scaly_dragonseed_template
				location = root.capital_province
				save_scope_as = spawned
			}
		}

		if = {
			limit = {
				exists = scope:spawned
			}
			scope:spawned = { save_scope_as = allard }
		}

		random_living_character = {
			limit = {
				dynasty = dynasty:dynn_Frey
			}

			# in case you need to reference the frey
			save_scope_as = frey
			death = {
				death_reason = death_vanished
			}
		}
	}

	option = {
		name = agot_silly_mode.0100.a
		add_character_modifier = {
			modifier = best_meal_of_my_life_modifier
			years = 5
		}
	}

	option = {
		name = agot_silly_mode.0100.b
		add_character_modifier = {
			modifier = suspicious_against_pies_modifier
			years = 5
		}
	}

	after = {
		hidden_effect  = {
			if = {
				limit = {
					exists = scope:spawned
				}

				scope:spawned = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
}

#Hightower Eaten by Kraken, Raid
agot_silly_mode.0300 = {
	hidden = yes
	scope = army

	trigger = {
		has_game_rule = agot_silly_mode
		scope:raider.dynasty ?= dynasty:dynn_Greyjoy
		scope:barony.title_province = {
			has_building = the_hightower_01
		}
	}

	immediate = {
		scope:barony.title_province = { remove_building = the_hightower_01 }
		scope:raider = { save_scope_as = occupant }
		scope:barony.holder = {
			save_scope_as = lowtower
			trigger_event = agot_silly_mode.0302
		}
		scope:raider = { trigger_event = agot_silly_mode.0302 }
	}
}

#Freypocalypse
agot_silly_mode.0400 = {
	hidden = yes

	immediate = {
		every_held_title = {
			set_coa = dynn_Frey
			#save_title_name_change_data = yes
			set_title_name = d_the_crossing
		}
		house ?= { set_coa = dynn_Frey }
		house ?= { set_house_name = dynn_Frey }
		if = {
			limit = { is_female = no }
			change_first_name = Walder
		}
		else = {
			change_first_name = Walda
		}
	}
}

#Candlings
agot_silly_mode.0500 = { #The Wildlings are worshipping... a giant candle in the sky?
	hidden = yes

	immediate = {
		every_player = {
			if = {
				limit = {
					has_culture = culture:lakefolk
				}
				trigger_event = {
					id = agot_silly_mode.0502
					days = 1
				}
			}
			else = {
				trigger_event = {
					id = agot_silly_mode.0501
					days = { 60 120 }
				}
			}
		}
		culture:lakefolk = {
			set_culture_name = {
				noun = candling
				prefix = candling_prefix
				collective_noun = candling_collective_noun
			}
			set_culture_pillar = ethos_mystic
			remove_culture_tradition = tradition_fp1_northern_stories
			add_culture_tradition = tradition_agot_candling
			set_variable = {
				name = recent_event_divergence
				days = 3650
			}
		}
	}
}
