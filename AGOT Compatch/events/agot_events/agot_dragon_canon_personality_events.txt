agot_dragon_canon_personality.0001 = { # dragon_friendly
	hidden = yes

	trigger = {
		NOR = {
			has_trait = dragon_aggressive
			has_trait = dragon_friendly
		}
	}

	immediate = {
		random_list = {
			70 = { # 70% chance to not get input
				add_trait = dragon_friendly
			}
			30 = { # 30% chance to get input
				modifier = {
					exists = var:cradlemate
					has_relation_agot_dragon = var:cradlemate
					var:cradlemate = {
						dynasty ?= { has_dynasty_perk = dragonrider_dynasty_legacy_1 }
					}
					add = 130
				}
				modifier = { # but only if owned
					factor = 0
					NOT = { has_character_flag = owned_dragon }
				}
				save_scope_as = growing_dragon
				court_owner = { trigger_event = agot_dragon_personality.0001 }
			}
		}
	}
}

	hidden = yes

	immediate = {
		if = {
			limit = { number_of_personality_traits >= 3 }
			remove_character_flag = agot_developing_canon_dragon_traits
		}
		else_if = {
			limit = { number_of_personality_traits = 2 }
			agot_give_random_dragon_personality_trait = yes
		}
		else_if = {
			limit = { number_of_personality_traits = 1 }
			agot_give_random_dragon_personality_trait = yes
			agot_give_random_dragon_personality_trait = yes
		}
		else_if = {
			limit = { number_of_personality_traits = 0 }
			agot_give_random_dragon_personality_trait = yes
			agot_give_random_dragon_personality_trait = yes
			agot_give_random_dragon_personality_trait = yes
		}
	}
}
