# Open Menu, Set Old Variables, Set Increment, Etc
agot_dragon_skin_color_menu = {
	scope = character
	saved_scopes = {player}
	is_shown = {
		has_trait = dragon
		OR = {
			var:current_rider ?= scope:player
			trigger_if = {
				limit = {
					exists = var:current_rider
				}
				is_ai = yes
				employer = scope:player
			}
			employer = scope:player
			debug_only = yes
			has_multiple_players = no
		}
	}
	is_valid = {
		NOT = {
			scope:player ?= {
				has_variable = editing_dragon
			}
		}
	}
	effect = {
		save_scope_as = dragon
		scope:player ?= {
			set_variable = {
				name = editing_dragon
				value = scope:dragon
			}
			set_variable = {
				name = value_to_change
				value = 0.05
			}
			trigger_event = agot_dragon_maintenance.9999
		}
		if = {
			limit = {
				any_in_global_list = {
					variable = gl_dragon_variable_storage
					var:dragon_id ?= scope:player.var:editing_dragon

						exists = var:gene_dragon_primary_color_value_old

# gene_dragon_primary_color_value +/-, reset
agot_dragon_skin_color_plus = {
	scope = character
	saved_scopes = {dragon}
	is_valid = {
		any_in_global_list = {
			variable = gl_dragon_variable_storage
			var:dragon_id ?= root.var:editing_dragon
			trigger_if = {
				limit = {
					root.var:value_to_change = 0.01
					var:gene_dragon_primary_color_value < 1
				}
				var:gene_dragon_primary_color_value <= 0.99
			}
			trigger_else_if = {
				limit = {
					root.var:value_to_change = 0.05
					var:gene_dragon_primary_color_value < 1
				}
				var:gene_dragon_primary_color_value <= 0.95
			}
			trigger_else = {
				always = no
			}
		}
	}

	effect = {
		if = {
			limit = {
				agot_has_dragon_storage_system_global_list_editor_trigger = yes
			}
			every_in_global_list = {
				variable = gl_dragon_variable_storage
				limit = {
					var:dragon_id ?= root.var:editing_dragon
				}
				save_scope_as = story
			}
			scope:story ?= {
				change_variable = {
					name = gene_dragon_primary_color_value
					add = root.var:value_to_change
				}
			}
		}
	}
}
