siege_dragon = {
	scope = character
	saved_scopes = { sieging_location }

	is_shown = {
		# do we have a friendly, allied dragon rider participating in the siege?
		scope:sieging_location ?= { # player has an army with a dragon rider at siege
			any_character_in_location = { # army commander and knights are valid riders
				agot_dragon_is_viable_siege_rider_trigger = yes
			}
		}
	}

	is_valid = {
		scope:sieging_location ?= { # player has an army with a dragon rider at siege
			custom_tooltip = { # Can't dragon siege pirate dens
				text = agot_dragon_siege.gui_tt_event_already_fired
				NOT = { has_variable = agot_dragon_event_open }
			}
			custom_tooltip = { # Can't dragon siege pirate dens
				text = agot_dragon_siege.gui_tt_cant_siege_pirate_den
				NOT = { has_holding_type = pirate_den_holding }
			}
			custom_tooltip = {
				text = agot_dragon_siege.gui_tt_cant_damage_multiple # this is only to avoid error spam
				any_character_in_location = { # army commander and knights are valid riders
					agot_dragon_is_viable_siege_rider_trigger = yes
				}
			}
			custom_tooltip = { # Block button if there is no valid rider available.
				text = agot_dragon_siege.gui_tt_no_valid_rider
				any_character_in_location = { # army commander and knights are valid riders
					agot_dragon_is_viable_siege_rider_trigger = yes
					NOR = {
						has_trait_rank = {
							trait = wounded
							rank > 1
						}
						has_trait = incapable
						has_trait = infirm
					}
					var:current_dragon ?= {
						NOT = {
							has_trait_rank = {
								trait = dragon_wounded
								rank > 1
							}
						}
					}
				}
			}
			trigger_if = { # There is only one rider at the siege.
				limit = {
					any_character_in_location = { # Army commander and knights are valid riders
						agot_dragon_is_viable_siege_rider_trigger = yes
						count = 1
						var:current_dragon ?= { save_temporary_scope_as = only_dragon }
					}
				}
				custom_tooltip = { # Can't damage more
					text = agot_dragon_siege.gui_tt_cant_damage_single
					any_character_in_location = { # army commander and knights are valid riders
						agot_dragon_is_viable_siege_rider_trigger = yes
						agot_dragon_siege_can_deal_more_damage_trigger = yes
					}
				}
				custom_tooltip = { # Only rider at siege already attacked due to being suggested/ordered
					text = agot_dragon_siege.tt.rider_already_attacked.single
					any_character_in_location = { # army commander and knights are valid riders
						agot_dragon_is_viable_siege_rider_trigger = yes
						NOT = { var:agot_ordered_dragon_use ?= scope:sieging_location }
					}
				}
			}
			trigger_else = { # There are several riders at the siege.
				custom_tooltip = { # Can't damage more
					text = agot_dragon_siege.gui_tt_cant_damage_multiple
					any_character_in_location = { # army commander and knights are valid riders
						agot_dragon_is_viable_siege_rider_trigger = yes
						agot_dragon_siege_can_deal_more_damage_trigger = yes
					}
				}
				custom_tooltip = { # No rider at siege that wasn't suggested/ordered to use dragon yet and is ai
					text = agot_dragon_siege.tt.rider_already_attacked.multiple
					any_character_in_location = { # army commander and knights are valid riders
						agot_dragon_is_viable_siege_rider_trigger = yes
						NOT = { var:agot_ordered_dragon_use ?= scope:sieging_location }
					}
				}
			}
			trigger_if = { # Was already dragon sieged by yourself.
				limit = { barony.var:used_dragon_sieged_today ?= root }
				custom_tooltip = {
					text = agot_dragon_siege.gui_tt_check
					barony = { NOT = { has_variable = used_dragon_sieged_today } }
				}
			}
			trigger_else = { # Was already dragon sieged by someone else.
				custom_tooltip = {
					text = agot_dragon_siege.gui_tt_commander_check
					barony = { NOT = { has_variable = used_dragon_sieged_today } }
				}
			}
		}
	}

	effect = {
		save_scope_as = commander

		scope:sieging_location ?= {
			every_character_in_location = { # army commander and knights are valid riders
				limit = {
					agot_dragon_is_viable_siege_rider_trigger = yes
					agot_dragon_siege_can_deal_more_damage_trigger = yes
				}
				add_to_temporary_list = t_dragon_rider_list
			}
		}

		if = { # if only one dragon, don't show selection gui # AGOT TODO if rider is war ally, add event that asks him to use his dragon --> may decide against it, cause he's craven or whatever
			limit = {
				any_in_list = {
					list = t_dragon_rider_list
					count = 1

					agot_dragon_possible_riders_trigger = { PLAYER = root }
				}
			}
			random_in_list = {
				list = t_dragon_rider_list
				limit = {
					agot_dragon_possible_riders_trigger = { PLAYER = root }
				}
				var:current_dragon ?= { save_scope_as = sieging_dragon }
				save_scope_as = sieging_rider
				if = { # if clicker is rider --> straight to normal event 0001
					limit = { this = root }
					custom_tooltip = agot_dragon_siege.gui_tt_one_dragon_free
					root = { trigger_event = agot_dragon_siege.0100 }
				}
				else = {
					if = {
						limit = { agot_dragon_rider_is_bound_trigger = yes }
						save_scope_value_as = { name = demand_dragon_use value = flag:bound }
						custom_tooltip = agot_dragon_siege.gui_tt_one_dragon_bound
					}
					else = { # rider is not bound by your orders --> only suggest to use dragon (to help AI as a player)
						custom_tooltip = agot_dragon_siege.gui_tt_one_dragon_free
						save_scope_value_as = { name = demand_dragon_use value = flag:free }
					}
					root = { trigger_event = agot_dragon_siege.0002 }
				}
			}
		}
		else_if = {
			limit = { any_in_list = { list = t_dragon_rider_list count > 1 } }
			custom_tooltip = agot_dragon_siege.gui_tt_multiple_dragons
			root = { trigger_event = agot_dragon_siege.0001 }
		}

		if = {
			limit = {
				location = {
					has_stationed_scorpions = yes
				}
			}
			custom_tooltip = agot_dragon_siege.gui_anti_dragon_tt
		}
		else_if = {
			limit = {
				location = {
					OR = {
						has_building = workshops_05
						has_building = workshops_06
						AND = {
							fort_level >= 10
							county.culture = { has_innovation = innovation_agot_anti_dragon_fighting }
						}
					}
				}
			}
			custom_tooltip = agot_dragon_siege.gui_scorpion_1_tt
		}
		else_if = {
			limit = {
				location = {
					OR = {
						has_building = workshops_07
						has_building = workshops_08
						AND = {
							fort_level >= 20
							county.culture = { has_innovation = innovation_agot_anti_dragon_fighting }
						}
					}
				}
			}
			custom_tooltip = agot_dragon_siege.gui_scorpion_2_tt
		}
	}
}

agot_show_dragon_past_rider_when_dead = {
	scope = character
	saved_scopes = {dragon}
	is_shown = {
		scope:dragon = {
			is_alive = no
		}
		AND = {
			has_global_variable_list = gl_dragon_variable_storage
			any_in_global_list = {
				variable = gl_dragon_variable_storage
				var:dragon_id ?= scope:dragon
				has_variable_list = past_riders
			}
		}
	}
	effect = {
		if = {
			limit = {
				agot_has_dragon_storage_system_global_list = yes
			}
			random_in_global_list = {
				variable = gl_dragon_variable_storage
				limit = {
					var:dragon_id = scope:dragon
				}
				save_scope_as = story
			}
		}
		scope:story = {
			every_in_list = {
				variable = past_riders
				root = {
					add_to_variable_list = {
						name = possible_characters
						target = prev
					}
				}
			}
		}
		trigger_event = agot_dragon_maintenance.0803
	}
}

				NOT = { root.liege_or_court_owner = root.location.county.holder }
			}
			NOT = { exists = root.liege_or_court_owner }
		}
		NOT = { has_character_flag = owned_dragon }
		NOT = { has_variable = current_rider }
		NOT = {
			any_relation = {
				type = agot_dragon
				is_alive = yes
			}
		}
		NOT = { has_variable = pitted_dragon_location }
	}
}
