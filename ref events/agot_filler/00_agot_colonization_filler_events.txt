agot_filler_colonization.0001 = { # [Forest] The Squirrel
	type = character_event
	title = agot_filler_colonization.0001.t
	desc = agot_filler_colonization.0001.desc
	theme = realm

	left_portrait = {
		character = root
		animation = fear
	}

	override_background = {
		reference = wilderness_forest
	}

	trigger = {
		is_available_adult = yes
		NOT = { has_character_flag = had_event_agot_filler_colonization_0001 }
		any_directly_owned_province = {
			has_holding_type = settlement_holding
			OR = {
				agot_is_forest_terrain = yes
				agot_is_taiga_terrain = yes
			}
		}
	}

	immediate = {
		add_character_flag = {
			flag = had_event_agot_filler_colonization_0001
			years = 50
		}
		random_directly_owned_province = {
			limit = {
				has_holding_type = settlement_holding
				OR = {
					agot_is_forest_terrain = yes
					agot_is_taiga_terrain = yes
				}
			}
			save_scope_as = colony_province
			county = { save_scope_as = colony_county }
		}
	}

	option = { # Attempt to fend off the squirrel
		name = agot_filler_colonization.0001.a
		skill = prowess
		duel = {
			skill = prowess
			value = 8
			20 = {
				desc = agot_filler_colonization.0001.a.success
				compare_modifier = {
					value = scope:duel_value
					multiplier = 1
				}
				add_prestige = miniscule_prestige_gain
			}

			20 = {
				desc = agot_filler_colonization.0001.a.failure
				compare_modifier = {
					value = scope:duel_value
					multiplier = -1
					min = -19
				}
				add_prestige = medium_prestige_loss
				increase_wounds_no_death_effect = { REASON = wild_animal }
				random = {
					chance = 10
					add_trait = one_eyed
				}
			}
		}

		stress_impact = {
			brave = minor_stress_loss
			craven = medium_stress_gain
		}
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_boldness = 0.5
				ai_vengefulness = 0.5
				ai_energy = 0.5
			}
			modifier = {
				factor = 0
				has_trait = craven
			}
		}
	}

	option = { # Call on your guards to help
		name = agot_filler_colonization.0001.b
		add_prestige = minor_prestige_loss
		add_dread = minor_dread_loss

		stress_impact = {
			craven = medium_stress_loss
			arrogant = medium_stress_gain
			brave = medium_stress_gain
		}
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_boldness = -0.5
			}
			modifier = {
				factor = 0
				has_trait = brave
			}
		}
	}

	option = { # Ignore it
		name = agot_filler_colonization.0001.c

		stress_impact = {
			paranoid = medium_stress_gain
		}
		ai_chance = {
