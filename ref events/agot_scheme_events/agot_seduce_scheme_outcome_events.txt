namespace = agot_seduce_outcome

agot_seduce_outcome.4901 = { # Caught waiting naked in their room
	type = character_event
	window = scheme_failed_event
	title = seduce_outcome.3901.t
	desc = {
		desc = agot_seduce_outcome.4901.opening
		first_valid = {
			triggered_desc = {
				trigger = { exists = scope:was_hard_reject }
				desc = agot_seduce_outcome.4901.reject
			}
			desc = agot_seduce_outcome.4901.normal
		}
	}
	theme = seduce_scheme
	override_background = {
		reference = bedchamber
	}
	left_portrait = {
		character = scope:owner
		animation = shame
		outfit_tags = { no_clothes }
	}
	right_portrait = {
		character = scope:target
		triggered_animation = {
			trigger = { exists = scope:was_hard_reject }
			animation = anger
		}
		animation = disgust
		outfit_tags = { nightgown }
	}

	immediate = {
		save_scope_as = target_liege
		scope:owner = { make_naked_effect = yes }
	}

	option = {
		name = agot_seduce_outcome.4901.a
		custom_tooltip = seduce_outcome.4900.failure.tt
		scope:owner = {
			seduce_outcome_publicised_attempted_crimes_or_nothing_effect = {
				TARGET = scope:target
				OWNER = scope:owner
			}
		}
	}

	after = {
		scope:owner = { remove_naked_effect = yes }
	}
}
