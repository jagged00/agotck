###SOCIAL RESCUE###
scripted_trigger agot_court_ongoing_1000_bore_trigger = {
	save_temporary_scope_as = potential_bore
	is_available_ai = yes
	is_adult = yes
	sex_opposite_of = scope:target
	NOR = {
		this = scope:target
		this = scope:owner
		is_close_or_extended_family_of = scope:target
	}
	OR = {
		reverse_opinion = {
			target = scope:target
			value < 0
		}
		AND = {
			reverse_opinion = {
				target = scope:target
				value <= 20
			}
			number_of_opposing_traits = {
				target = scope:target
				value >= 1
			}
		}
	}
}

agot_court_ongoing.1000 = {
	type = character_event
	title = agot_court_ongoing.1000.t
	desc = agot_court_ongoing.1000.desc

	theme = romance_scheme
	left_portrait = {
		character = scope:social_bore
		animation = laugh
	}
	right_portrait = {
		character = scope:target
		animation = boredom
	}
	widget = {
		gui = "event_window_widget_scheme"
		container = "custom_widgets_container"
	}

	trigger = {
		exists = scope:target.court_owner
		scope:target.court_owner = { is_landed = yes }
		scope:target = {
			OR = {
				court_owner = {
					OR = {
						agot_court_ongoing_1000_bore_trigger = yes
						any_courtier = { agot_court_ongoing_1000_bore_trigger = yes }
						any_vassal = { agot_court_ongoing_1000_bore_trigger = yes }
					}
				}
				any_relation = { type = rival agot_court_ongoing_1000_bore_trigger = yes }
			}
		}
		NOT = {
			scope:scheme = {
				OR = {
					has_scheme_modifier = courting_kindness_modifier
					has_scheme_modifier = courting_faux_pas_modifier
					has_scheme_modifier = courting_abandoned_me_modifier
				}
			}
		}
	}

	weight_multiplier = {
		base = 1
		modifier = {
			exists = scope:scheme.var:court_milestone_event
			add = { #0 at milestone 1, -0.45 at milestone 2, -0.9 at milestone 3
				value = scope:scheme.var:court_milestone_event
				multiply = -0.45
				add = 0.45
			}
		}
	}

	immediate = {
		scope:target = {
			court_owner = {
				if = {
					limit = { agot_court_ongoing_1000_bore_trigger = yes }
					add_to_list = bore_list
				}
				every_vassal = {
					limit = { agot_court_ongoing_1000_bore_trigger = yes }
					add_to_list = bore_list
				}
				every_courtier = {
					limit = { agot_court_ongoing_1000_bore_trigger = yes }
					add_to_list = bore_list
				}
			}
			every_relation = {
				type = rival
				limit = { agot_court_ongoing_1000_bore_trigger = yes }
				add_to_list = bore_list
			}

			random_in_list = {
				list = bore_list
				weight = {
					base = 1
					compare_modifier = {
						target = this
						value = diplomacy
						multiplier = -5
						offset = 12
					}
					ai_value_modifier = {
						ai_sociability = {
							value = high_negative_ai_value_inverse
							multiply = 10
						}
					}
				}
				save_scope_as = social_bore
			}

			court_owner.capital_province = { save_scope_as = location }
		}
	}

	#Intervene
	option = {
		name = agot_court_ongoing.1000.a

		duel = {
			skill = diplomacy
			target = scope:social_bore
			30 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 0.1
				}
				desc = agot_court_ongoing.1000.a.success
				send_interface_toast = {
					title = agot_court_ongoing.1000.a.success
					left_icon = scope:social_bore
					right_icon = scope:target

					reverse_add_opinion = {
						target = scope:target
						modifier = grateful_opinion
						opinion = 15
					}
					scope:scheme = {
						add_scheme_modifier = {
							type = courting_kindness_modifier
						}
					}
				}
			}
			20 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -0.1
				}
				desc = agot_court_ongoing.1000.a.failure
				send_interface_toast = {
					title = agot_court_ongoing.1000.a.failure
					left_icon = scope:social_bore
					right_icon = scope:target

					reverse_add_opinion = {
						target = scope:target
						modifier = bored_opinion
						opinion = -10
					}
					scope:scheme = {
						add_scheme_modifier = {
							type = courting_faux_pas_modifier
						}
					}
				}
			}
		}

		stress_impact = {
			shy = minor_stress_impact_gain
			humble = minor_stress_impact_gain
		}

		ai_chance = {
			base = 100

			ai_sociability_target_modifier = { VALUE = 50 }
			ai_boldness_target_modifier = { VALUE = 20 }
		}
	}

	#Don't do anything
	option = {
		name = agot_court_ongoing.1000.b

		scope:scheme = {
			add_scheme_modifier = {
				type = courting_abandoned_me_modifier
			}
		}

		stress_impact = {
			brave = minor_stress_impact_gain
			compassionate = minor_stress_impact_gain
			arrogant = minor_stress_impact_gain
		}

		ai_chance = {
			base = 50

			ai_sociability_target_modifier = { VALUE = -30 }
			ai_boldness_target_modifier = { VALUE = -20 }
		}
	}
}

	save_temporary_scope_as = combatant

		is_knight = yes

		has_focus = martial_authority_focus
