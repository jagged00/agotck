# agot_pirate_convert_whole_realm_to_feudalism_decision
agot_pirate_events.0001 = {
	type = character_event
	title = agot_pirate_events.0001.t
	desc = agot_pirate_events.0001.desc
	theme = stewardship

	left_portrait = {
		character = root
		animation = personality_rational
	}

	immediate = {
		show_as_tooltip = {
			agot_pirate_convert_whole_realm_to_feudalism_effect = yes
		}
		hidden_effect = {
			if = {
				limit = {
					OR = {
						agot_is_valid_lp_feudal_government_target = yes
						any_liege_or_above = { agot_is_valid_lp_feudal_government_target = yes }
					}
					NOT = {
						any_liege_or_above = { has_trait = nightswatch }
					}
				}
				every_liege_or_above = {
					limit = {
						NOT = { is_independent_ruler = yes }
					}
					send_interface_message = {
						type = event_stewardship_neutral_with_text
						title = msg_vassal_lp_feudalised
						desc = msg_vassal_lp_feudalised_desc
						right_icon = root
					}
				}
			}
			else = {
				every_liege_or_above = {
					limit = {
						NOT = { is_independent_ruler = yes }
					}
					send_interface_message = {
						type = event_stewardship_neutral_with_text
						title = msg_vassal_common_feudalised
						desc = msg_vassal_common_feudalised_desc
						right_icon = root
					}
				}
			}
		}
	}

	option = {
		name = agot_pirate_events.0001.a
	}
}

scripted_trigger should_pay_pirate_rates = {
	is_alive = yes
	is_landed = no
	domicile ?= {
		is_domicile_type = agot_pirate_ship
		domicile_location = {
			has_building_or_higher = pirate_den_01
			county.holder = {
				is_human = yes
			}
		}
	}
	trigger_if = {
		limit = {
			has_variable = ports_visited_variable
		}
		var:ports_visited_variable = scope:ports_visited_scope_value
	}
}

# Arrival to new Port
agot_pirate_events.0003 = {
	type = character_event
	title = agot_pirate_events.0003.t
	desc = agot_pirate_events.0003.desc
	override_icon = {
		reference = "gfx/interface/icons/event_types/type_war.dds"
	}
	override_background = {
		reference = docks
	}
	trigger = {
		is_alive = yes
		is_landed = no
		domicile ?= {
			is_domicile_type = agot_pirate_ship
			domicile_location = {
				has_building_or_higher = pirate_den_01
				county.holder = {
					is_human = yes
				}
			}
		}
	}
	theme = travel
	left_portrait = {
		character = root
		animation = thinking
	}
	right_portrait = {
		character = scope:port_holder
		animation = acknowledging
	}
	immediate = {
		save_scope_as = pirate
		domicile = {
			save_scope_as = pirate_domicile
			domicile_location = {
				save_scope_as = port_name
				county.holder = {
					save_scope_as = port_holder
				}
			}
		}
		if = {
			limit = {
				has_variable = one_time_rd_pirate_cheap_fee
			}
			save_scope_value_as = { # RD Characters are poor, give them a 1 time discount
				name = port_docking_fee
				value = 5
			}
			domicile.domicile_location.county.holder = {
				switch = {
					trigger = has_realm_law
					pirate_tax_rate_01 = {
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_01_value
						}
					}
					pirate_tax_rate_02 = {
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_02_value
						}
					}
					pirate_tax_rate_03 = {
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_03_value
						}
					}
					pirate_tax_rate_04 = {
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_04_value
						}
					}
				}
			}
			remove_variable = one_time_rd_pirate_cheap_fee
		}
		else = {
			domicile.domicile_location.county.holder = {
				switch = {
					trigger = has_realm_law
					pirate_tax_rate_01 = {
						save_scope_value_as = {
							name = port_docking_fee
							value = agot_pirate_docking_port_fee_01_value
						}
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_01_value
						}
					}
					pirate_tax_rate_02 = {
						save_scope_value_as = {
							name = port_docking_fee
							value = agot_pirate_docking_port_fee_02_value
						}
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_02_value
						}
					}
					pirate_tax_rate_03 = {
						save_scope_value_as = {
							name = port_docking_fee
							value = agot_pirate_docking_port_fee_03_value
						}
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_03_value
						}
					}
					pirate_tax_rate_04 = {
						save_scope_value_as = {
							name = port_docking_fee
							value = agot_pirate_docking_port_fee_04_value
						}
						save_scope_value_as = {
							name = yearly_port_fee
							value = agot_pirate_yearly_port_fee_04_value
						}
					}
				}
			}
		}
		if = {
			limit = {
				has_variable = ports_visited_variable
			}
			change_variable = {
				name = ports_visited_variable
				add = 1
			}
			save_scope_value_as = {
				name = ports_visited_scope_value
				value = var:ports_visited_variable
			}
		}
		else = {
			set_variable = {
				name = ports_visited_variable
				value = 1
			}
			save_scope_value_as = {
				name = ports_visited_scope_value
				value = var:ports_visited_variable
			}
		}
	}
	option = {
		name = agot_pirate_events.0003.a
		if = {
			limit = {
				NOT = {
					exists = scope:port_docking_fee
				}
			}
			save_scope_value_as = {
				name = port_docking_fee
				value = 25
			}
		}
		show_as_tooltip = {
			pay_short_term_gold = {
				target = scope:port_holder
				gold = scope:port_docking_fee
			}
		}
		hidden_effect = {
			scope:port_holder = {
				send_interface_message = {
					type = msg_pirate_ship_arrived_in_port
					title = agot_pirate_domicile_arrived_in_port.t
					desc = agot_pirate_domicile_arrived_in_port.desc
					scope:pirate = {
						pay_short_term_gold = {
							target = scope:port_holder
							gold = scope:port_docking_fee
						}
					}
					right_icon = scope:pirate
				}
			}
		}
		trigger_event = {
			id = agot_pirate_events.0004
			years = 1
		}
	}
}

# Results calculating event for the Raid Pirate Ship scheme. Scope: scheme_owner
agot_pirate_events.8031 = {
	hidden = yes

	immediate = {
		scope:scheme = {
			scheme_target_character = {
				save_scope_as = target
				domicile ?= { save_scope_as = target_domicile }
			}
			scheme_owner = {
				save_scope_as = owner
				#DISCOVERY ROLL
				random = {
					chance = scope:scheme.scheme_secrecy
					save_scope_value_as = {
						name = scheme_discovered
						value = yes
					}
				}
				#SUCCESS ROLL
				random = {
					chance = scope:scheme.scheme_success_chance
					save_scope_value_as = {
						name = scheme_successful
						value = yes
					}
				}
				if = {
					limit = { exists = scope:scheme_successful }
					save_scope_value_as = {
						name = gold
						value = {
							value = scope:target.minor_gold_value
							multiply = scope:target_domicile.num_domicile_buildings
							add = scope:target.minor_gold_value
						}
					}
					scope:target_domicile = { agot_destroy_random_pirate_ship_building_variable_effect = yes }
					scope:target = {
						if = {
							limit = {
								any_courtier = {
									count > 0
								}
							}
							every_courtier = {
								limit = {
									age >= 5
								}
								add_to_list = targets_for_death
							}
							ordered_in_list = {
								list = targets_for_death
								order_by = {
									value = 1
									if = {
										limit = { is_adult = yes }
										add = 10
									}
									if = {
										limit = { this = scope:target.player_heir }
										add = -5
									}
									if = {
										limit = {
											can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = scope:target }
										}
										add = 5
									}
								}
								check_range_bounds = no
								max = 3
								add_to_list = target_victims
							}
						}
					}
					random = {
						chance = {
							value = scope:scheme.scheme_success_chance
							divide = 2
						}
						save_scope_value_as = {
							name = scheme_critical_success
							value = yes
						}
					}
					# Bonus Rewards
					random_list = {
						10 = { # Steal an Artifact
							trigger = {
								scope:target = {
									any_character_artifact = { is_equipped = no }
								}
							}
							modifier = {
								exists = scope:scheme_critical_success
								factor = 2
							}
							scope:target = {
								random_character_artifact = {
									limit = { is_equipped = no }
									weight = {
										base = 1
										# Prefer equippable artifacts
										modifier = {
											scope:owner = { can_equip_artifact = prev }
											factor = 2
										}
										# Don't like banners
										modifier = {
											exists = var:banner_dynasty
											factor = 0.1
										}
										modifier = {
											exists = var:banner_house
											factor = 0.1
										}
									}
									save_scope_as = stolen_artifact
								}
							}
						}
						10 = { # Steal an Artifact from Courtier
							trigger = {
								scope:target = {
									any_courtier = {
										any_character_artifact = { is_equipped = no }
									}
								}
							}
							modifier = {
								exists = scope:scheme_critical_success
								factor = 2
							}
							scope:target = {
								random_courtier = {
									limit = {
										any_character_artifact = { is_equipped = no }
									}
									random_character_artifact = {
										limit = { is_equipped = no }
										weight = {
											base = 1
											# Prefer equippable artifacts
											modifier = {
												scope:owner = { can_equip_artifact = prev }
												factor = 2
											}
											# Don't like banners
											modifier = {
												exists = var:banner_dynasty
												factor = 0.1
											}
											modifier = {
												exists = var:banner_house
												factor = 0.1
											}
										}
										save_scope_as = stolen_artifact
									}
								}
							}
						}
						10 = {
							save_scope_value_as = {
								name = bonus_gold
								value = {
									value = scope:target.tiny_gold_value
									add = { 25 100 }
								}
							}
						}
					}
					trigger_event = {
						id = agot_pirate_events.8032
						days = 1
					}
				}
				else = {
					scope:scheme = {
						if = {
							limit = { exists = scope:scheme_discovered }
							random_scheme_agent_character = {
								limit = { is_alive = yes }
								weight = {
									base = 1
									modifier = {
										add = {
											value = 20
											subtract = intrigue
											subtract = prowess
										}
									}
								}
								save_scope_as = captured
							}
						}
					}
					trigger_event = {
						id = agot_pirate_events.8033
						days = 1
					}
				}
			}
		}
	}
}

scripted_effect raid_pirate_ship_opinion_effect = {
	$TARGET$ = {
		add_opinion = {
			target = $OWNER$
			modifier = agot_raided_pirate_ship_crime
		}
		progress_towards_rival_effect = {
			REASON = rival_raid_pirate_ship
			CHARACTER = $OWNER$
			OPINION = 0
		}
	}
}

scripted_effect raid_pirate_ship_success_public_effect = {
	if = {
		limit = { exists = scope:stolen_artifact }
		scope:stolen_artifact = {
			set_owner = {
				target = scope:owner
				history = {
					location = scope:target_domicile.domicile_location
					actor = scope:owner
					recipient = scope:owner
					type = stolen
				}
			}
		}
	}
	else_if = {
		limit = {
			exists = scope:bonus_gold
		}
		scope:owner = { add_short_term_gold = scope:bonus_gold }
	}
	scope:owner = { add_short_term_gold = scope:gold }
	every_in_list = {
		list = target_victims
		death = {
			death_reason = death_raid_estate
			killer = scope:owner
		}
	}
	scope:target_domicile = { agot_destroy_random_pirate_ship_building_effect = yes }
	raid_pirate_ship_opinion_effect = {
		OWNER = scope:owner
		TARGET = scope:target
	}
	scope:owner = {
		add_dread = medium_dread_gain
	}
	scope:target = {
		add_dread = medium_dread_loss
	}
}

scripted_effect raid_pirate_ship_failure_effect = {
	if = {
		limit = { exists = scope:scheme_discovered }
		if = {
			limit = { exists = scope:captured }
			scope:captured = {
				add_character_flag = {
					flag = was_abducted_block_notification_event
					days = 2
				}
			}
			rightfully_imprison_character_less_verbose_effect = {
				TARGET = scope:captured
				IMPRISONER = scope:target
			}
		}
		scope:target = {
			add_opinion = {
				target = scope:owner
				modifier = attempted_raid_pirate_ship_crime
			}
		}
	}
}

					change = scope:independence_change
				}
				resolve_title_and_vassal_change = scope:independence_change

			if = {
				limit = {
					agot_dlc_check_for_landless_pirates = yes
				}
				scope:potential_pirate = { change_government = pirate_government }
			}
			else = {
				scope:potential_pirate = { change_government = pirate_no_dlc_government }
			}
		}
		else = {
			every_held_title = {
				limit = {
					tier = tier_county
					title_province = {
						geographical_region = historical_pirate_region

							has_holding_type = pirate_den_holding

								has_holding_type = settlement_holding

					}

			}
		}
	}
