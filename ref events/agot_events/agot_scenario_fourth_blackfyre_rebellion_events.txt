# Aegor Rivers (We have won the war! I'll ask for a reward for my loyal service)
agot_fourth_blackfyre_rebellion.0002 = {
	type = character_event
	title = agot_fourth_blackfyre_rebellion.0002.t
	desc = agot_fourth_blackfyre_rebellion.0002.desc

	theme = vassal

	override_background = {
		reference = agot_iron_throne
	}
	right_portrait = {
		character = scope:daemon
		animation = chancellor
	}
	left_portrait = {
		character = root
		animation = throne_room_kneel_1
	}
	trigger = {
		root = character:Targaryen_95
		exists = title:h_the_iron_throne.holder
		title:h_the_iron_throne.holder = character:Blackfyre_11
	}

	immediate = {
		agot_fourth_blackfyre_rebellion_scopes_effect = yes
	}

	option = { # The Hand of the King would be a suitable position
		name = agot_fourth_blackfyre_rebellion.0002.a
		trigger = {
			# AI Aegor can do this option, Player only if Daemon is AI
			OR = {
				is_ai = yes
				has_multiple_players = no
			}
		}
		show_as_tooltip = {
			custom_tooltip = agot_fourth_blackfyre_rebellion.0002.gc_tt
			# Switch Player Aegor to Daemon if singleplayer
			if = {
				limit = {
					is_ai = no
					has_multiple_players = no
				}
				set_player_character = scope:daemon
			}
			destroy_laamp_effect = { ADVENTURER = scope:aegor }

			scope:daemon = {
				add_courtier ?= scope:aegor
				assign_councillor_type = {
					type = councillor_castellan
					target = scope:aegor
					remove_existing_councillor = yes
				}
				block_firing_councillor_effect = { COURT_OWNER = scope:aegor }
			}

		}
		scope:daemon = {
			add_character_flag = { flag = bittersteel_requests_hotk months = 6 }
		}
		ai_chance = { base = 100 }
	}
	option = { # I want a place on your Kingsguard... as the Lord Commander
		name = agot_fourth_blackfyre_rebellion.0002.b
		ai_chance = { base = 100 }
		trigger = {
			scope:daemon = {
				ruler_primary_title_has_kingsguard_trigger = yes
			}
			# AI Aegor can always do this option, Player only if Daemon is AI
			OR = {
				is_ai = yes
				has_multiple_players = no
			}
		}
		scope:daemon = {
			add_character_flag = { flag = bittersteel_requests_lc months = 6 }
		}
		show_as_tooltip = {
			# Switch Player Aegor to Daemon if singleplayer
			if = {
				limit = {
					is_ai = no
					has_multiple_players = no
				}
				set_player_character = scope:daemon
			}
			destroy_laamp_effect = { ADVENTURER = scope:aegor }
			hidden_effect = {
				agot_join_kingsguard_effect = {
					KINGSGUARD = scope:aegor
					KING = scope:daemon
				}
			}
			# Assign Aegor as LC
			scope:daemon = {
				if = {
					limit = { has_council_position = kingsguard_lord_commander }
					cp:kingsguard_lord_commander = { add_character_flag = can_fire_kingsguard }
					fire_councillor = cp:kingsguard_lord_commander
				}
				custom_tooltip = agot_fourth_blackfyre_rebellion.0002.gc_tt
				if = {
					limit = { NOT = { has_council_position = kingsguard_lord_commander }  }
					agot_assign_lord_commander_effect = {
						COMMANDER = scope:aegor
						KING = scope:daemon
					}
				}
			}
		}
	}
	option = { # I want Summerhall
		name = agot_fourth_blackfyre_rebellion.0002.c
		ai_chance = { base = 100 }
		scope:daemon = { add_character_flag = { flag = bittersteel_requests_summerhall months = 6 } }
		trigger = {
			OR = {
				title:c_summerfield.holder = { is_vassal_or_below_of = title:h_the_iron_throne.holder } # Summerhall is in the Iron Throne
				scope:daemon = { has_title = title:c_summerfield }
			}
			NOT = { title:c_summerfield.holder = { is_at_war_with = title:h_the_iron_throne.holder } } # And not at war with the throne
		}
		show_as_tooltip = {
			custom_tooltip = agot_fourth_blackfyre_rebellion.0002.gc_tt
			title:c_summerfield = {
				save_scope_as = target
				add_to_list = granted_titles
				holder = { save_scope_as = old_summerhall_holder }
			}
			scope:aegor = {
				# Resolve title, liege, government changes
				# Determine hierarchy
				ep3_become_landed_save_liege_effect = {
					TITLE_GIVER = scope:daemon
					ALWAYS_INDEPENDENT = no
					TITLE_LIST = granted_titles
				}
				ep3_become_landed_transfer_effect = {
					TITLE_RECEIVER = scope:aegor
					TITLE_LIST = granted_titles
					TYPE = granted
					REASON = flag:negotiated
					ENNOBLED_ADVENTURER = flag:no
				}
				# Messages for nearby players
				create_landed_ruler_message_effect = {
					LANDED = scope:aegor
					FLAVOR_CHAR = scope:aegor
					FLAVOR_TITLE = scope:target
					REASON = flag:negotiated
				}
				every_courtier = {
					custom = every_follower_custom
					limit = { is_human = yes }
					set_variable = {
						name = former_camp_leader
						value = root
					}
					add_trait = adventurer_follower
				}
			}
			# If relative lost title, come home
			scope:old_summerhall_holder = {
				if = {
					limit = {
						is_ruler = no
						is_close_or_extended_family_of = root
						NOT = { employer = root }
					}
					set_employer = root
				}
			}
		}
	}
	option = { # I would appreciate some gold...
		name = agot_fourth_blackfyre_rebellion.0002.d
		ai_chance = { base = 100 }
		scope:daemon = { add_character_flag = { flag = bittersteel_requests_gold months = 6 } }
		show_as_tooltip = {
			scope:daemon = {
				save_scope_value_as = {
					name = gold_amount
					value = 1000
				}
				remove_short_term_gold = scope:gold_amount
				scope:aegor = {
					add_gold = scope:gold_amount
					add_prestige = major_prestige_value
				}
			}
		}
	}
	option = { # Nothing, I have simply served my liege
		# AI Bittersteel should always ask for something
		ai_chance = { base = 0 }
		name = agot_fourth_blackfyre_rebellion.0002.e
		add_character_flag = requests_nothing
		stress_impact = {
			base = major_stress_impact_loss
		}
	}
	after = {
		if = {
			limit = { has_character_flag = requests_nothing }
			remove_character_flag = requests_nothing
		}
		else = { scope:daemon = { trigger_event = agot_fourth_blackfyre_rebellion.0015  } }
	}
}

# Daemon III (The Invasion)
agot_fourth_blackfyre_rebellion.0011 = {
	type = character_event
	title = agot_fourth_blackfyre_rebellion.0011.t
	desc = agot_fourth_blackfyre_rebellion.0011.desc

	theme = war

	override_background = {
		reference = AGOT_beach
	}
	left_portrait = {
		character = root
		animation = war_attacker
	}
	right_portrait = {
		character = scope:aegonv
		animation = personality_compassionate
	}
	lower_right_portrait = {
		character = scope:duncansmall
		animation = personality_content
	}
	lower_left_portrait = {
		character = scope:aegor
		animation = personality_content
	}
	trigger = {
		NOT = { has_title = title:h_the_iron_throne }
	}

	immediate = {
		agot_fourth_blackfyre_rebellion_scopes_effect = yes

		random_character_artifact = {
			limit = { has_artifact_modifier = vs_blackfyre_modifier }
			save_scope_as = blackfyre
		}
	}

	option = { # To King's Landing!
		name = agot_fourth_blackfyre_rebellion.0011.a
		flavor = agot_fourth_blackfyre_rebellion.0011.a.tt
		add_character_flag = { flag = mw_init_event_blocker days = 2 }
		start_war = {
			casus_belli = agot_blackfyre_claim
			claimant = root
			target = title:h_the_iron_throne.holder
			target_title = title:h_the_iron_throne
			target_title = title:e_the_crownlands
			target_title = title:d_kings_landing
			target_title = title:c_kings_landing
			target_title = title:k_dragonstone
			target_title = title:d_dragonstone
			target_title = title:c_dragonstone
		}
		hidden_effect = { agot_fourth_blackfyre_loyalists_alliance_effect = yes }
		random_character_war = {
			limit = { using_cb = agot_blackfyre_claim }
			save_scope_as = fourth_bf_war
		}
		# Store the War
		scope:fourth_blackfyre_rebellion_story ?= {
			set_variable = {
				name = fourth_bf_war
				value = scope:fourth_bf_war
			}
		}
		# Notification Event for Players
		every_player = {
			limit = {
				capital_province = {
					geographical_region = world_westeros_seven_kingdoms
				}
				NOR = {
					this ?= character:Blackfyre_11
					this ?= title:d_laamp_merc_golden_company.holder
				}
			}
			trigger_event = agot_fourth_blackfyre_rebellion.0012
		}
		if = {
			limit = { has_dlc_feature = roads_to_power }
			# Call in Golden Company
			if = {
				limit = {
					title:d_laamp_merc_golden_company.holder ?= {
						has_really_bad_opinion_of_root_trigger = no
						NOR = {
							is_allied_to =  title:h_the_iron_throne.holder
							has_strong_hook =  title:h_the_iron_throne.holder
							has_relation_friend =  title:h_the_iron_throne.holder
							has_truce =  title:h_the_iron_throne.holder
						}
					}
				}
				# Aegor Event
				title:d_laamp_merc_golden_company.holder ?= {
					trigger_event = agot_fourth_blackfyre_rebellion.0001
				}
				every_character_war = {
					limit = {
						primary_defender = title:h_the_iron_throne.holder
					}
					add_attacker = title:d_laamp_merc_golden_company.holder
				}
			}
		}
		else_if = {
			limit = { NOT = { has_dlc_feature = roads_to_power } }
			# Spawn Special Troops
			if = {
				limit = {
					is_landed = yes
				}
				generate_golden_company_special_troops = { LOCATION = scope:daemon.capital_province ORIGIN = scope:daemon.capital_province ARMY_NAME = blackfyre_dynamic_revolt_army }
			}
			else = {
				generate_golden_company_special_troops = { LOCATION = scope:daemon.location ORIGIN = scope:daemon.location ARMY_NAME = blackfyre_dynamic_revolt_army }
			}
		}
		scope:aegonv = {
			# Call in Duncan Small to war
			if = {
				limit = {
					character:Targaryen_126 = { is_alive = yes }
					title:k_dragonstone.holder = character:Targaryen_126
				}
				every_character_war = {
					limit = {
						primary_defender = title:h_the_iron_throne.holder
					}
					add_defender = title:k_dragonstone.holder
				}
				# If Lyonel is alive, and Duncan is betrothed with Cassandra, then add Lyonel to the war
				if = {
					limit = {
						character:Baratheon_71 = {
							is_alive = yes
							is_allied_to = scope:duncansmall
						}
						title:e_the_stormlands.holder = character:Baratheon_71
						scope:duncansmall = {
							is_betrothed = yes
						}
						scope:cassandra = {
							is_alive = yes
							is_betrothed = yes
						}
					}
					every_character_war = {
						limit = {
							primary_defender = title:h_the_iron_throne.holder
						}
						add_defender = title:e_the_stormlands.holder
					}
				}
			}
			# If Duncan abdicated, call in Jaehaerys
			else_if = {
				limit = {
					character:Targaryen_126 = { is_alive = yes }
					title:k_dragonstone.holder = character:Targaryen_127
				}
				every_character_war = {
					limit = {
						primary_defender = title:h_the_iron_throne.holder
					}
					add_defender = title:k_dragonstone.holder
				}
			}
		}
	}
}
