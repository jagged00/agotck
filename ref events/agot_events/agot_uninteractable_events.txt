namespace = agot_uninteractable_events

# Preventing raiding by creating a truce (taken from Vanilla)
# Called from army_on_actions.txt (on_raid_action_start)
# root is the raid army
# scope:raider is the person owning the raid army
# scope:barony is the barony title that got raided
# scope:county is the county title for the barony
agot_uninteractable_events.0001 = {
	hidden = yes
	scope = army

	trigger = {
		exists = scope:barony
		scope:barony.county.holder = {
			government_has_flag = government_is_uninteractable
		}
	}
	immediate = {
		scope:raider = {
			send_interface_toast = {
				title = agot_uninteractable_events.0001.t
				left_icon = scope:raider
				custom_tooltip = agot_uninteractable_events.0001.desc
			}
		}
	}
}
