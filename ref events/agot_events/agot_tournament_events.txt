namespace = agot_tournament_events

agot_tournament_events.9000 = { #Send guests home
	hidden = yes

	immediate = {
		every_courtier_or_guest = {
			limit = {
				is_foreign_court_guest_of = scope:host
				NOR = {
					any_relation = {
						type = agot_squire
						employer ?= scope:host
					}
					any_relation = {
						type = agot_knight
						employer ?= scope:host
					}
					any_relation = {
						type = ward
						employer ?= scope:host
					}
					any_relation = {
						type = guardian
						employer ?= scope:host
					}
				}
			}
			return_to_court = yes
			return_home = yes
		}
	}
}
