agot_crowned_stag.1000 = {
	type = character_event
	title = agot_crowned_stag.1000.t
	desc = agot_crowned_stag.1000.desc
	theme = family
	override_background = agot_winterfell_throneroom

	left_portrait = {
		character = root
		animation = thinking
	}
	right_portrait = {
		character = character:Stark_4
		animation = personality_honorable
	}

	trigger = {
		is_ruler = yes
		NOT = { primary_heir = character:Stark_4 }
		character:Stark_4 = {
			is_alive = yes
			is_ai = yes
			is_ruler = no
			is_healthy = yes
			agot_valid_potential_nw_member = yes
		}
	}

	#Resend
	on_trigger_fail = {
		if = {
			limit = {
				is_ruler = yes
				character:Stark_4 = {
					is_alive = yes
					is_ai = yes
					is_ruler = no
				}
			}
			trigger_event = {
				id = agot_crowned_stag.1000
				days = 365
			}
		}
	}

	option = { #if you must
		name = agot_crowned_stag.1000.a
		save_scope_as = actor
		agot_send_to_nightswatch_effect = {
			ACTOR = scope:actor
			NIGHTSWATCH_CANDIDATE = character:Stark_4
		}
		ai_chance = {
			base = 0
			modifier = {
				add = agot_historic_outcome_2
			}
		}
	}
	option = { #stay please
		name = agot_crowned_stag.1000.b
		character:Stark_4 = {
			add_opinion = {
				modifier = disappointed_opinion
				opinion = -10
				target = root
			}
		}
		hidden_effect = {
			character:Stark_4 = {
				random_list = {
					50 = { }
					50 = { add_trait = depressed_1 }
				}
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = agot_nonhistoric_outcome_2
			}
		}
	}
}
