agot_proclaim.0001 = {
	type = character_event
	title = agot_proclaim.0001.t
	desc = agot_proclaim.0001.desc

	override_background = {
		reference = agot_iron_throne
	}

	theme = crown

	immediate = {
		title:h_the_iron_throne.holder = {
			save_scope_as = agot_proclaimed_kingqueen
		}
	}

	right_portrait = {
		character = scope:agot_proclaimed_kingqueen
		animation = personality_honorable
	}

	option = {
		name = {
			trigger = {
				NOT = {
					OR = {
						this = scope:agot_proclaimed_kingqueen
						scope:agot_proclaimed_kingqueen = {
							any_vassal_or_below = {
								this = root
							}
						}
					}
				}
			}
			text = agot_proclaim.a
		}
		name = {
			trigger = {
				AND = {
					NOT = { this = scope:agot_proclaimed_kingqueen }
					scope:agot_proclaimed_kingqueen = {
						any_vassal_or_below = {
							this = root
						}
					}
				}
			}
			text = agot_proclaim.b
		}
		name = {
			trigger = {
				this = scope:agot_proclaimed_kingqueen
			}
			text = agot_proclaim.c
		}
	}
}
