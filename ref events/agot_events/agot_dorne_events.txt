			remove_trait = refusing_marriage
		}
	}

	immediate = {
		character:Uller_6 = { save_scope_as = Ellaria }
	}

	option = {
		name = agot_dorne.0001.a
		set_relation_lover = character:Uller_6

		}
	}

		name = agot_dorne.0001.b

			remove_trait = refusing_marriage
		}

		ai_chance = {
			base = 0
		}
	}
