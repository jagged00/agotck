## Called from yearly events to check if AI should colonize
## For human player version check scripted_guis/agot_wilderness.txt -> gui_colonize
agot_colonization_events.0006 = {
	hidden = yes
	scope = character

	trigger = {
		### Basic checks
		is_ai = yes
		is_landed = yes

		NOT = { any_owned_story = { story_type = story_conqueror } } #Conquerors have their own logic

		trigger_if = {
			limit = { is_independent_ruler = no }
			has_character_flag = vassal_directive_settle_wilderness
		}

		### Culture/Character ID triggers
		agot_is_dummy_character = no

		### Blocking Governments
		NOT = { government_has_flag = government_is_nw }
		NOT = { government_has_flag = government_is_uninteractable }

		### Character Income Trigger
		trigger_if = { # If count
			limit = { highest_held_title_tier = tier_county }
			gold >= 50
			monthly_character_income >= 2
		}
		trigger_else_if = { # If duke
			limit = { highest_held_title_tier = tier_duchy }
			gold >= 75
			monthly_character_income >= 1.75
		}
		trigger_else_if = { # If king
			limit = { highest_held_title_tier = tier_kingdom }
			gold >= 100
			monthly_character_income >= 1.75
		}
		trigger_else_if = { # If emporer
			limit = { highest_held_title_tier = tier_empire }
			gold >= 125
			monthly_character_income >= 1.5
		}
		trigger_else = { gold >= 150 } # Fallback

		### Check whether the ruler is above their settlement limit
		above_settlement_limit = no

		### Not on cooldown
		NOT = { has_variable = colonization_cooldown }

		### Only surrounding lands
		any_sub_realm_county = {
			any_title_to_title_neighboring_and_across_water_county = {
				holder = { government_has_flag = government_is_wilderness }
				NOT = { has_county_modifier = block_settlement_ability }
				#BTW Blocks
				trigger_if = {
					limit = {
						title_province ?= {
							geographical_region = world_westeros_beyond_the_wall
						}
					}
					root = {
						OR = {
							culture = { agot_is_wildling_culture = yes }
							capital_county.title_province ?= {
								geographical_region = world_westeros_beyond_the_wall
							}
						}
					}
				}
			}
		}
	}
	immediate = {
		debug_log = "agot_colonization_events.0006: AI considering colonization..."

		every_sub_realm_county = {
			every_title_to_title_neighboring_and_across_water_county = {
				limit = {
					holder = { government_has_flag = government_is_wilderness }
					NOR = {
						has_county_modifier = block_settlement_ability
						holder = root
					}
					#BTW Blocks
					trigger_if = {
						limit = {
							title_province ?= {
								geographical_region = world_westeros_beyond_the_wall
							}
						}
						root = {
							OR = {
								culture = { agot_is_wildling_culture = yes }
								capital_county.title_province ?= {
									geographical_region = world_westeros_beyond_the_wall
								}
							}
						}
					}
					#Pirate AI Blocks
					trigger_if = {
						limit = {
							title_province ?= {
								geographical_region = historical_pirate_region
							}
						}
						root = {
							OR = {
								government_has_flag = government_is_pirate
								has_trait = ambitious
								AND = {
									ai_boldness >= very_high_positive_ai_value
									ai_rationality >= high_positive_ai_value
								}
							}
						}
					}
				}
				add_to_list = adjacent_wilderness
			}
		}

		ordered_in_list = {
			list = adjacent_wilderness
			order_by = {
				value = 0
				every_neighboring_county = {
					limit = { holder = root }
					add = 2
				}
				every_neighboring_county = {
					limit = {
						holder = root
						duchy = prev.duchy
					}
					add = 16
				}
				every_neighboring_county = {
					limit = {
						holder = root
						kingdom = prev.kingdom
					}
					add = 8
				}
				every_neighboring_county = {
					limit = {
						holder = root
						empire = prev.empire
					}
					add = 4
				}
			}
			save_scope_as = wilderness
		}

		### MAKE AI COLONIZE COUNTY ###
		scope:wilderness ?= { debug_log_scopes = yes } # Only needed for testing
		if = {
			limit = {
				exists = scope:wilderness
			}
			ai_colonization_effect = { WILDERNESS = scope:wilderness ROOT_SCOPE = root }
		}
	}
}

## Event to trigger event from 0008 - 0015 to add a variable + upkeep modifier
agot_colonization_events.0007 = {
	hidden = yes

	immediate = {
		remove_upkeep_modifier = yes
		if = {
			limit = { highest_held_title_tier = tier_county }
			trigger_event = agot_colonization_events.0008
		}
		else_if = {
			limit = { highest_held_title_tier = tier_duchy }
			trigger_event = agot_colonization_events.0009
		}
		else_if = {
			limit = { highest_held_title_tier = tier_kingdom }
			trigger_event = agot_colonization_events.0010
		}
		else_if = {
			limit = { highest_held_title_tier = tier_empire }
			trigger_event = agot_colonization_events.0011
		}
		else = {
			trigger_event = agot_colonization_events.0012
		}
	}
}

## Adding upkeep to Counts
## Triggered in agot_colonization_events.0007
agot_colonization_events.0008 = {
	hidden = yes

	immediate = {
		if = {
			limit = {
				exists = var:num_settled_wilderness
				var:num_settled_wilderness = 1
			}
			add_character_modifier = holding_one_settlements
		}
		else_if = {
			limit = {
				exists = var:num_settled_wilderness
				var:num_settled_wilderness = 2
			}
			add_character_modifier = holding_two_settlements
		}
		else_if = {
			limit = {
				exists = var:num_settled_wilderness
				var:num_settled_wilderness = 3
			}
			add_character_modifier = holding_three_settlements
		}
		else_if = {
			limit = {
				exists = var:num_settled_wilderness
				var:num_settled_wilderness = 4
			}
			add_character_modifier = holding_four_settlements
		}
		else_if = {
			limit = {
				exists = var:num_settled_wilderness
				var:num_settled_wilderness = 5
			}
			add_character_modifier = holding_five_settlements
		}
		else_if = {
			limit = {
				exists = var:num_settled_wilderness
				var:num_settled_wilderness > 5
			}
			add_character_modifier = holding_more_than_five_settlements
		}
	}
}

agot_colonization_events.0020 = {
	hidden = yes

	trigger = {
		OR = {
			monthly_character_income < 0.3
			above_settlement_limit = yes
		}
		is_ai = yes
	}

	immediate = {
		every_held_title = {
			limit = {
				tier = tier_county
				title_province ?= { has_holding_type = settlement_holding }
			}
			add_to_list = targeted_titles
		}

		every_in_list = {
			list = targeted_titles

			if = {
				limit = {
					holder = { above_settlement_limit = yes }
					title_province ?= { has_building = settlement_01 }
				}
				make_settlement_county_wilderness = { COUNTY = this }
			}
			else_if = {
				limit = {
					holder = { above_settlement_limit = yes }
					title_province ?= { has_building = settlement_02 }
				}
				make_settlement_county_wilderness = { COUNTY = this }
			}
			else_if = {
				limit = {
					holder = { above_settlement_limit = yes }
					title_province ?= { has_building = settlement_03 }
				}
				make_settlement_county_wilderness = { COUNTY = this }
			}
		}
	}
}

agot_colonization_events.0022 = {
	hidden = yes
	scope = character

	trigger = {
		has_character_flag = vassal_directive_settle_wilderness
		any_held_county = { title_province = { has_holding_type = settlement_holding } }
	}

	immediate = {
		random_held_county = {
			limit = { title_province = { has_holding_type = settlement_holding has_building = settlement_03 } }
			save_scope_as = upgrade_county
		}

		if = {
			limit = { NOT = { exists = scope:upgrade_county } }
			random = {
				chance = 66
				random_held_county = {
					limit = { title_province = { has_holding_type = settlement_holding has_building = settlement_02 } }
					save_scope_as = upgrade_county
				}
			}
		}
		if = {
			limit = { NOT = { exists = scope:upgrade_county } }
			random = {
				chance = 33
				random_held_county = {
					limit = { title_province = { has_holding_type = settlement_holding has_building = settlement_01 } }
					save_scope_as = upgrade_county
				}
			}
		}
		agot_upgrade_settlement_to_full_holding_effect = { COUNTY = scope:upgrade_county }
		correct_wilderness_tracker = yes
	}
}

agot_colonization_events.1001 = { # Clearing out the Wolves
	type = character_event
	title = agot_colonization_events.1001.t
	desc = agot_colonization_events.1001.desc
	theme = hunting

	left_portrait = {
		character = root
		animation = personality_bold
	}

	right_portrait = {
		character = scope:knight
		animation = war_defender
	}

	lower_center_portrait = {
		character = scope:hooked_character
	}

	trigger = {
		is_available_adult = yes
	}

	immediate = {
		if = {
			limit = {
				any_knight = {
					is_available_ai_adult = yes
					count > 0
				}
			}
			ordered_knight = {
				limit = { is_available_ai_adult = yes }
				order_by = prowess
				save_scope_as = knight
			}
		}
		if = {
			limit = {
				any_hooked_character = {
					is_available_ai_adult = yes
					count > 0
				}
			}
			ordered_hooked_character = {
				limit = { is_available_ai_adult = yes }
				order_by = prowess
				save_scope_as = hooked_character
			}
		}
	}

	option = { # Take them on myself
		name = agot_colonization_events.1001.a
		skill = prowess
		duel = {
			skill = prowess
			value = 10
			30 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 2
					min = -10
				}
				desc = agot_colonization_events.1001.a.duel_success
				send_interface_toast = {
					left_icon = root
					add_prestige = medium_prestige_gain
					title = agot_colonization_events.1001.success.toast.t
					scope:obstacle_province = { remove_building = wolf_den_cleared }
				}
			}
			30 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -2
					min = -25
				}
				desc = agot_colonization_events.1001.a.duel_failure
				send_interface_toast = {
					left_icon = root
					title = agot_colonization_events.1001.failure.toast.t
					add_prestige = medium_prestige_loss
					increase_wounds_effect = { REASON = hunting_accident }
					hidden_effect = {
						scope:obstacle_province = { remove_building = wolf_den_cleared }
					}
					scope:obstacle_province = { add_building = wolf_den }
				}
			}
		}

		stress_impact = {
			brave = minor_stress_loss
			craven = medium_stress_gain
		}
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_boldness = 0.5
				ai_vengefulness = 0.5
				ai_energy = 0.5
			}
			modifier = {
				factor = 0
				has_trait = craven
			}
		}
	}

		name = agot_colonization_events.1001.b
		trigger = {
			exists = scope:knight
		}
		scope:knight = {
			duel = {
				skill = prowess
				value = 10
				30 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = 2
						min = -10
					}
					desc = agot_colonization_events.1001.b.duel_success
					root = {
						send_interface_toast = {
							left_icon = scope:knight

							title = agot_colonization_events.1001.success.toast.t
							scope:obstacle_province = { remove_building = wolf_den_cleared }
						}
					}
				}

					compare_modifier = {

						multiplier = -2
						min = -25

							title = agot_colonization_events.1001.failure.toast.t
