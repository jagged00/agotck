# Daemon can try to get Blackfyre or leave
agot_scenario_ats.2000 = {
	type = character_event
	title = agot_scenario_ats.2000.t
	desc = agot_scenario_ats.2000.desc
	theme = family

	left_portrait = {
		character = root
		animation = debating
	}

	right_portrait = {
		character = character:Targaryen_95
		animation = anger
	}

	immediate = {
		character:Targaryen_95 = {
			save_scope_as = bittercringe
		}
		root = {
			save_scope_as = daemon_blackfyre_2
		}
	}

	trigger = {
		character:Blackfyre_5 = {
			var:agot_ats_daemon_status = flag:no_blackfyre
		}
	}

	option = {
		name = agot_scenario_ats.2000.a
		custom_tooltip = agot_scenario_ats.2000.a.tt
		# Check if Daemon can do single combat
		trigger = {
			character:Blackfyre_5 = {
				can_start_single_combat_trigger = yes
			}
		}

		#Run single combat
		configure_start_single_combat_effect = {
			SC_INITIATOR = scope:daemon_blackfyre_2
			SC_ATTACKER = scope:daemon_blackfyre_2
			SC_DEFENDER = scope:bittercringe
			FATALITY = no
			FIXED = no
			LOCALE = terrain_scope
			OUTPUT_EVENT = agot_scenario_ats.2001
			INVALIDATION_EVENT = fp1_tbc.0021
		}

		ai_chance = {
			base = 0
			modifier = {
				add = agot_nonhistoric_chain_outcome
			}
		}

	}

	option = {
		name = agot_scenario_ats.2000.b
		custom_tooltip = agot_scenario_ats.2000.b.tt
		hidden_effect = {
			random_list = {
				# Daemon convinces Bittersteel
				0 = {
					# Set Blackfyre scope
					character:Targaryen_95 = {
						random_character_artifact = {
							limit = { has_artifact_modifier = vs_blackfyre_modifier }
							save_scope_as = blackfyre
						}
					}
					# Make Daemon the owner of Blackfyre
					scope:blackfyre = {
						set_owner = character:Blackfyre_5
					}
					# Equip it
					hidden_effect = {
						scope:blackfyre = {
							equip_artifact_to_owner_replace = yes
						}
					}
					# Set variable for having Blackfyre
					character:Blackfyre_5 = {
						set_variable = {
							name = agot_ats_daemon_status
							value = flag:got_blackfyre
						}
						trigger_event = agot_scenario_ats.2002
					}

					# Give Daemon his father's crown
					character:Targaryen_95 = {
						random_character_artifact = {
							limit = { has_artifact_modifier = daemonblackfyre_crown_modifier }
							save_scope_as = blackfyre_crown
						}
					}

					scope:blackfyre_crown = {
						set_owner = character:Blackfyre_5
					}

					hidden_effect = {
						scope:blackfyre_crown = {
							equip_artifact_to_owner_replace = yes
						}
					}

					modifier = {
						add = agot_nonhistoric_chain_outcome
					}
				}
				# Daemon fails to convince Bittersteel (canon outcome)
				0 = {
					character:Blackfyre_5 = {
						add_prestige = major_prestige_loss
						add_character_modifier = agot_ats_no_sword_daemon

						# Set variable for giving up on Blackfyre
						set_variable = {
							name = agot_ats_daemon_status
							value = flag:giving_up_on_blackfyre
						}
						trigger_event = agot_scenario_ats.2002
					}

					modifier = {
						add = agot_historic_chain_outcome
					}
				}
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = agot_historic_chain_outcome
			}
		}
	}

	option = {
		name = agot_scenario_ats.2000.c
		custom_tooltip = agot_scenario_ats.2000.c.tt
		# Set variable for Daemon giving up on the sword and bittersteel support, send to John the Fiddler disguise or free roam
		character:Blackfyre_5 = {
			set_variable = {
				name = agot_ats_daemon_status
				value = flag:giving_up_on_blackfyre
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = agot_nonhistoric_chain_outcome
			}
		}
	}
}

# Dunk can join the war on Eustace's side

agot_scenario_ats.9032 = {
	type = character_event
	title = agot_scenario_ats.9032.t
	desc = agot_scenario_ats.9032.desc
	theme = war

	left_portrait = {
		character = character:Tall_1
		animation = thinking
	}

	right_portrait = {
		character = character:Osgrey_55
		animation = rage
	}

	immediate = {
		if = {
			limit = {
				character:Webber_13 = {
					any_character_war = {
						is_attacker = character:Webber_13
						is_defender = character:Osgrey_55
					}
				}
			}
			character:Webber_13 = {
				every_character_war = {
					limit = {
						is_attacker = character:Webber_13
						is_defender = character:Osgrey_55
					}
					save_scope_as = ats_war
				}
			}
		}
		else_if = {
			limit = {
				character:Osgrey_55 = {
					any_character_war = {
						is_attacker = character:Osgrey_55
						is_defender = character:Webber_13
					}
				}
			}
			character:Osgrey_55 = {
				every_character_war = {
					limit = {
						is_attacker = character:Osgrey_55
						is_defender = character:Webber_13
					}
					save_scope_as = ats_war
				}
			}
		}
		character:Webber_13 = {
			save_scope_as = rohanne
		}
		character:Osgrey_55 = {
			save_scope_as = eustace
		}
	}
	# Dunk joins the war
	option = {
		name = agot_scenario_ats.9032.a
		custom_tooltip = agot_scenario_ats.9032.a.tt

		if = {
			limit = {
				scope:ats_war = {
					is_attacker = character:Osgrey_55
					is_defender = character:Webber_13
				}
			}
			scope:ats_war = {
				add_attacker = character:Tall_1
			}
		}
		if = {
			limit = {
				scope:ats_war = {
					is_attacker = character:Webber_13
					is_defender = character:Osgrey_55
				}
			}
			scope:ats_war = {
				add_defender = character:Tall_1
			}
		}

		create_maa_regiment = {
			type = armored_horsemen
			check_can_recruit = no
			size = 5
		}
	}
	# Dunk refuses to join
	option = {
		name = agot_scenario_ats.9032.b
		custom_tooltip = agot_scenario_ats.9032.b.tt

		character:Tall_1 = {
			add_character_flag = left_story
		}
	}
}
