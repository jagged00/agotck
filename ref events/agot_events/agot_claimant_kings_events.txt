namespace = agot_claimant_king_events

agot_claimant_king_events.0001 = {
	hidden = yes

	immediate = {
		if = {
			limit = {
				NAND = {
					agot_is_independent_ruler = no
					any_character_war = {
						is_war_leader = scope:attacker
						OR = {
							#Claimant
							AND = {
								OR = {
									using_cb = claim_cb
									using_cb = agot_succession_war
									using_cb = agot_claimant_faction_war
									using_cb = agot_bastard_claimant_war
								}
								root = claimant
								claimant = scope:attacker
								casus_belli = {
									any_target_title = {
										tier >= tier_kingdom
										holder = {
											highest_held_title_tier = PREV.tier
											agot_is_independent_ruler = yes
										}
									}
								}
							}
							#Independent
							AND = {
								scope:attacker.highest_held_title_tier >= tier_kingdom
								OR = {
									using_cb = independence_war
									using_cb = agot_independence_war
								}
								scope:defender = { agot_is_independent_ruler = yes }
							}
						}
					}
				}
			}
			remove_character_flag = agot_claimant_king
		}
	}
}
