namespace = agot_marriage_on_action_events

agot_marriage_on_action_events.0002 = { # Renly CoA event
	hidden = yes

	trigger = {
		this = character:Baratheon_4
		is_alive = yes
		is_landed = yes
		exists = scope:spouse
		scope:spouse = {
			dynasty ?= dynasty:dynn_Tyrell
		}
	}
	immediate = {
		agot_create_personal_coa_for_historical_ai = { ACTOR = root COA_KEY = house_BaratheonSE }
		if = {
			limit = {
				NOT = { house = dynasty:dynn_Baratheon.dynasty_founder.house }
			}

			house = { set_coa = var:my_personal_coa.house }
			every_held_title = {
				set_coa = root.house
			}
		}
	}
}
