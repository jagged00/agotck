# ToS - Succession Crisis: ToS Demanded
agot_trial_of_seven.0100 = {
	type = character_event
	title = agot_trial_of_seven.0100.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					NOT = { exists = scope:true_heir }
					NOT = { this = scope:accuser_leader }
				}
				desc = agot_trial_of_seven.0100.desc
			}
			triggered_desc = {
				trigger = {
					exists = scope:true_heir
					NOT = { this = scope:accuser_leader }
				}
				desc = agot_trial_of_seven.0100.desc.helper
			}
			triggered_desc = {
				trigger = { this = scope:accuser_leader }
				desc = agot_trial_of_seven.0100.desc.accuser
			}
		}
	}

	theme = war

	override_background = {
		reference = courtyard_location
	}

	left_portrait = {
		character = scope:defender_leader
		animation = personality_bold
	}

	right_portrait = {
		character = scope:accuser_leader
		animation = personality_bold
	}

	lower_left_portrait = {
		trigger = {
			exists = scope:true_heir
			NOT = { scope:accuser_leader = scope:true_heir }
		}
		character = scope:true_heir
	}

	immediate = {
		save_scope_as = trial_of_seven_usurpation

		location = { save_scope_as = background_terrain_scope }
		scope:defender_leader = { save_scope_as = bg_override_char }

		#Add them to their teams.
		if = {
			limit = { this = scope:defender_leader }
			scope:defender_leader = {
				add_to_variable_list = {
					name = multi_duel_tos_temp_list
					target = scope:defender_leader
				}
			}
			scope:accuser_leader = {
				add_to_variable_list = {
					name = multi_duel_tos_temp_list
					target = scope:accuser_leader
				}
			}

			#Helper Scenario Setup
			if = {
				limit = {
					scope:accuser_leader = { has_character_flag = true_heir_helper }
					scope:defender_leader.house = {
						any_house_member = {
							is_alive = yes
							agot_eligible_heir_for_succession_crisis_legit_heir = { USURPER = scope:defender_leader LEGIT_HEIR = scope:defender_leader FINAL_USURPER = scope:defender_leader }
							NOT = { this = scope:accuser_leader }
							NOT = { this = scope:defender_leader }
							NOT = { has_relation_rival = scope:accuser_leader }
							NOT = { has_relation_friend = scope:defender_leader }
							exists = var:usurped_my_birthright
							any_claim = {
								tier > tier_barony
								holder = scope:defender_leader
								count > 0
							}
							is_imprisoned = no
							save_temporary_scope_as = t_legit_heir
							scope:defender_leader.primary_title = {
								scope:t_legit_heir = {
									has_claim_on = prev
								}
							}
						}
					}
				}
				scope:defender_leader.house = {
					random_house_member = {
						limit = {
							is_alive = yes
							agot_eligible_heir_for_succession_crisis_legit_heir = { USURPER = scope:defender_leader LEGIT_HEIR = scope:defender_leader FINAL_USURPER = scope:defender_leader }
							NOT = { this = scope:accuser_leader }
							NOT = { this = scope:defender_leader }
							NOT = { has_relation_rival = scope:accuser_leader }
							NOT = { has_relation_friend = scope:defender_leader }
							exists = var:usurped_my_birthright
							any_claim = {
								tier > tier_barony
								holder = scope:defender_leader
								count > 0
							}
							is_imprisoned = no
							save_temporary_scope_as = t_legit_heir
							scope:defender_leader.primary_title = {
								scope:t_legit_heir = {
									has_claim_on = prev
								}
							}
						}
						save_scope_as = true_heir
					}
				}
			}
			scope:accuser_leader = {
				if = {
					limit = { has_character_flag = true_heir_helper }
					remove_character_flag = true_heir_helper
				}
			}

			if = {
				limit = {
					this = scope:accuser_leader
					this = { has_character_flag = agot_tos_royal_side }
				}
				scope:accuser_leader = { save_scope_as = the_royal_leader }
				scope:defender_leader = { save_scope_as = the_other_leader }
			}
			else = {
				scope:accuser_leader = { save_scope_as = the_other_leader }
				scope:defender_leader = { save_scope_as = the_royal_leader }
			}

			#Memories
			scope:defender_leader = {
				create_character_memory = {
					type = agot_trial_of_seven_accused_usurper
					participants = {
						accuser = scope:accuser_leader
					}
				}
			}
			scope:accuser_leader = {
				create_character_memory = {
					type = agot_trial_of_seven_accuser_usurper
					participants = {
						accused = scope:defender_leader
					}
				}
			}
		}
	}

	option = {
		name = {
			text = {
				first_valid = {
					triggered_desc = {
						trigger = { this = scope:defender_leader }
						desc = agot_trial_of_seven.0100.a
					}
					triggered_desc = {
						trigger = { this = scope:accuser_leader }
						desc = agot_trial_of_seven.0100.a.accuser
					}
				}
			}
		}

		if = {
			limit = { this = scope:defender_leader }

			#SCENARIO
			save_scope_as = tos_succession_crisis_scenario

			scope:accuser_leader = { trigger_event = agot_trial_of_seven.0100 }
			trigger_event = agot_trial_of_seven.1000
		}
	}
}

agot_trial_of_seven.0103 = {
	type = character_event
	window = agot_character_event_huge
	title = agot_trial_of_seven.0103.t
	desc = {
		first_valid = {
			#Legit Heir is somehow already the king - Usurper POV
			triggered_desc = {
				trigger = {
					exists = scope:final_usurper
					exists = scope:final_heir
					scope:final_usurper = scope:final_heir
					this = scope:usurper
				}
				desc = agot_trial_of_seven.0103.desc.legit_heir_won_after_all.usurper
			}
			#Legit Heir is somehow already the king - Legit POV
			triggered_desc = {
				trigger = {
					exists = scope:final_usurper
					exists = scope:final_heir
					scope:final_usurper = scope:final_heir
					this = scope:legit_heir
				}
				desc = agot_trial_of_seven.0103.desc.legit_heir_won_after_all.legit_heir
			}
			#Legit Heir is somehow already the king - ? POV
			triggered_desc = {
				trigger = {
					exists = scope:final_usurper
					exists = scope:final_heir
					scope:final_usurper = scope:final_heir
					NOT = { this = scope:usurper }
					NOT = { this = scope:legit_heir }
				}
				desc = agot_trial_of_seven.0103.desc.legit_heir_won_after_all
			}
			#Legit Heir won - Usurper POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_won
					scope:usurper = { is_alive = yes }
				}
				desc = agot_trial_of_seven.0103.desc.legit_won
			}
			#Legit Heir won - Usurper's Heir POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_won
					scope:usurper = { NOT = { is_alive = yes } }
				}
				desc = agot_trial_of_seven.0103.desc.legit_won.usurper_died
			}
			#Legit Heir Helper won - Usurper POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_helper_won
					scope:usurper = { is_alive = yes }
				}
				desc = agot_trial_of_seven.0103.desc.legit_helper_won
			}
			#Legit Heir Helper won - Usurper's Heir POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_helper_won
					scope:usurper = { NOT = { is_alive = yes } }
				}
				desc = agot_trial_of_seven.0103.desc.legit_helper_won.usurper_died
			}
			#Legit Heir Helper won but Legit Heir died - Usurper POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_helper_won_need_new_heir
					exists = scope:heir_of_legit_heir
					scope:usurper = { is_alive = yes }
				}
				desc = agot_trial_of_seven.0103.desc.legit_helper_won.legit_heir_died
			}
			#Legit Heir Helper won but Legit Heir died - Usurper's Heir POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_helper_won_need_new_heir
					exists = scope:heir_of_legit_heir
					scope:usurper = { NOT = { is_alive = yes } }
				}
				desc = agot_trial_of_seven.0103.desc.legit_helper_won.legit_heir_died.usurper_died
			}
			#Usurper won
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_usurper_won
					scope:usurper = { is_alive = yes }
				}
				desc = agot_trial_of_seven.0103.desc.usurper_won
			}
			#Legit Heir Helper won but no heir - Usurper POV
			triggered_desc = {
				trigger = {
					exists = scope:tos_story_usurpation_legit_heir_helper_won_need_new_heir
					NOT = { exists = scope:heir_of_legit_heir }
				}
				desc = agot_trial_of_seven.0103.desc.legit_helper_won.no_heir
			}
		}
	}

	theme = war

	override_background = {
		reference = courtyard_location
	}

	right_portrait = {
		trigger = { NOT = { scope:final_heir = scope:final_usurper } }
		character = scope:final_usurper
		triggered_animation = {
			trigger = {
				OR = {
					exists = scope:tos_story_usurpation_usurper_won
					trigger_if = {
						limit = { NOT = { exists = scope:heir_of_legit_heir } }
						exists = scope:tos_story_usurpation_legit_heir_helper_won_need_new_heir
					}
				}
			}
			animation = personality_bold
		}
		animation = worry
	}

	lower_right_portrait = {
		trigger = { NOT = { scope:usurper = scope:final_usurper } }
		character = scope:usurper
	}

	left_portrait = {
		character = scope:final_heir
		triggered_animation = {
			trigger = {
				OR = {
					exists = scope:tos_story_usurpation_legit_heir_won
					exists = scope:tos_story_usurpation_legit_heir_helper_won
					trigger_if = {
						limit = { exists = scope:heir_of_legit_heir }
						exists = scope:tos_story_usurpation_legit_heir_helper_won_need_new_heir
					}
					scope:final_usurper = scope:legit_heir
				}
			}
			animation = personality_bold
		}
		animation = worry
	}

	lower_left_portrait = {
		trigger = { NOT = { scope:legit_heir = scope:final_heir } }
		character = scope:legit_heir
	}

	immediate = {
		save_scope_as = bg_override_char
		scope:claim_keeper.var:succession_crisis_usurper = { save_scope_as = usurper }
		scope:claim_keeper.var:succession_crisis_legit_heir = { save_scope_as = the_og_legit_heir }
		scope:claim_keeper.var:final_usurpation_primary_title = { save_scope_as = main_title }
		save_scope_as = final_usurper
		if = {
			limit = { exists = scope:claim_keeper.var:tos_true_heir }
			scope:claim_keeper.var:tos_true_heir = { save_scope_as = legit_heir }
			scope:claim_keeper.var:succession_crisis_legit_heir = { save_scope_as = legit_heir_helper }
		}
		else = {
			scope:claim_keeper.var:succession_crisis_legit_heir = { save_scope_as = legit_heir }
		}
		scope:claim_keeper.var:succession_crisis_legit_heir = { save_scope_as = legit_combatant } #for loc only

		# Find heir for the legitimate heir
		if = {
			limit = {
				OR = {
					exists = scope:tos_story_usurpation_legit_heir_helper_won_need_new_heir
					trigger_if = {
						limit = { exists = scope:claim_keeper.var:tos_true_heir }
						scope:legit_heir = { NOT = { is_alive = yes } }
					}
				}
			}
			save_scope_as = tos_looking_for_a_new_heir
			scope:legit_heir = {
				if = {
					limit = {
						any_child = {
							agot_eligible_heir_for_succession_crisis_legit_heir = { USURPER = scope:usurper LEGIT_HEIR = scope:legit_heir FINAL_USURPER = scope:final_usurper }
							count > 0
						}
					}
					ordered_child = {
						order_by = age
						limit = {
							agot_eligible_heir_for_succession_crisis_legit_heir = { USURPER = scope:usurper LEGIT_HEIR = scope:legit_heir FINAL_USURPER = scope:final_usurper }
						}
						save_scope_as = heir_of_legit_heir
					}
				}
				else_if = {
					limit = {
						any_sibling = {
							agot_eligible_heir_for_succession_crisis_legit_heir = { USURPER = scope:usurper LEGIT_HEIR = scope:legit_heir FINAL_USURPER = scope:final_usurper }
							count > 0
						}
					}
					ordered_sibling = {
						order_by = age
						limit = {
							agot_eligible_heir_for_succession_crisis_legit_heir = { USURPER = scope:usurper LEGIT_HEIR = scope:legit_heir FINAL_USURPER = scope:final_usurper }
						}
						save_scope_as = heir_of_legit_heir
					}
				}
			}
		}

		# Final Claimant
		if = {
			limit = {
				scope:legit_heir = {
					is_alive = yes
					agot_can_be_ruler = yes
				}
			}
			scope:legit_heir = { save_scope_as = final_claimant }
		}
		else_if = {
			limit = {
				exists = scope:heir_of_legit_heir
				scope:heir_of_legit_heir = {
					is_alive = yes
					agot_can_be_ruler = yes
				}
			}
			scope:heir_of_legit_heir = { save_scope_as = final_claimant }
		}

		#Portrait
		if = {
			limit = { exists = scope:heir_of_legit_heir }
			scope:heir_of_legit_heir = { save_scope_as = final_heir }
		}
		else = { scope:legit_heir = { save_scope_as = final_heir } }

		agot_multiduel_betrayer_kingsguard_post_tos = { IMPRISONER = scope:final_usurper }

		#Memories
		if = {
			limit = { this = scope:final_usurper }
			if = {
				limit = {
					scope:legit_combatant = { is_alive = yes }
					exists = scope:tos_story_usurpation_usurper_won
				}
				scope:legit_combatant = {
					create_character_memory = {
						type = agot_trial_of_seven_loser_leader
						participants = {
							opponent = scope:usurper
						}
					}
				}
			}
			else_if = {
				limit = {
					scope:legit_combatant = { is_alive = yes }
					NOT = { exists = scope:tos_story_usurpation_usurper_won }
				}
				scope:legit_combatant = {
					create_character_memory = {
						type = agot_trial_of_seven_winner_leader
						participants = {
							opponent = scope:usurper
						}
					}
				}
			}
			if = {
				limit = {
					scope:usurper = { is_alive = yes }
					exists = scope:tos_story_usurpation_usurper_won
				}
				scope:usurper = {
					create_character_memory = {
						type = agot_trial_of_seven_winner_leader
						participants = {
							opponent = scope:legit_combatant
						}
					}
				}
			}
			else_if = {
				limit = {
					scope:usurper = { is_alive = yes }
					NOT = { exists = scope:tos_story_usurpation_usurper_won }
				}
				scope:usurper = {
					create_character_memory = {
						type = agot_trial_of_seven_loser_leader
						participants = {
							opponent = scope:legit_combatant
						}
					}
				}
			}
		}

		#Death List - My Team
		if = {
			limit = {
				has_variable_list = potential_multi_duel_bonding_list
				any_in_list = {
					variable = potential_multi_duel_bonding_list
					is_alive = no
					count > 0
				}
			}
			every_in_list = {
				variable = potential_multi_duel_bonding_list
				limit = { is_alive = no }
				show_as_tooltip = {
					death = {
						death_reason = death_duel_trial_by_seven
					}
				}
			}
		}
		#Death List - Opponent Team
		if = {
			limit = {
				has_variable_list = potential_multi_duel_bonding_opponent_list
				any_in_list = {
					variable = potential_multi_duel_bonding_opponent_list
					is_alive = no
					count > 0
				}
			}
			every_in_list = {
				variable = potential_multi_duel_bonding_opponent_list
				limit = { is_alive = no }
				show_as_tooltip = {
					death = {
						death_reason = death_duel_trial_by_seven
					}
				}
			}
		}

		#Bonding
		if = {
			limit = { scope:usurper = { is_alive = yes } }
			scope:usurper = {
				if = {
					limit = { exists = scope:tos_story_usurpation_usurper_won }
					add_character_flag = { flag = we_won_the_tos days = 3 }
				}
			}
			scope:usurper = { agot_tos_bonding_effect = yes }
		}

		if = {
			limit = { scope:legit_combatant = { NOT = { is_ai = yes } } }
			#Bonding
			if = {
				limit = { scope:legit_combatant = { is_alive = yes } }
				scope:legit_combatant = {
					if = {
						limit = { NOT = { exists = scope:tos_story_usurpation_usurper_won } }
						add_character_flag = { flag = we_won_the_tos days = 3 }
					}
				}
				scope:legit_combatant = {
					add_character_flag = {
						flag = tos_bonding_effect_received
						days = 40
					}
					agot_tos_bonding_effect = yes
				}
			}
		}
	}

	#Ok I switch to play as the new ruler.
	option = {
		name = agot_trial_of_seven.0103.a
		trigger = {
			has_multiple_players = no
			is_ai = no
			exists = scope:final_claimant
			OR = {
				exists = scope:tos_story_usurpation_legit_heir_won
				exists = scope:tos_story_usurpation_legit_heir_helper_won

		#Final Fallback: If no other options royal side can buy new characters
		if = {
			limit = {
				OR = {
					has_character_flag = tos_honorable_knight
					has_character_flag = tos_greedy_knight
					has_character_flag = tos_dishonorable_knight
				}
				OR = {
					NOT = { has_variable_list = possible_characters_temp }
					any_in_list = {
						variable = possible_characters_temp
						save_temporary_scope_as = temp_char
						scope:duel_leader = { NOT = { is_target_in_variable_list = { name = multi_duel_tos_temp_list target = scope:temp_char } } }

					}
				}
			}
			if = {
				limit = {
					OR = {
						has_character_flag = tos_greedy_knight
						has_character_flag = tos_greedy_knight_allowed
					}
					NOT = { has_character_flag = tos_non_royal_sellsword_blocker }
				}
				#A greedy knight steps forward..
				create_character = {
					age = { 16 40 }
					dynasty = none
					culture = scope:duel_leader.culture
					faith = scope:duel_leader.faith
					location = scope:duel_leader.location
					gender = male
					random_traits = yes
					trait = greedy
					prowess = { 10 20 }
					save_scope_as = new_char
					after_creation = {
						scope:duel_leader = {
							add_courtier = scope:new_char
							add_to_variable_list = {
								name = possible_characters_temp
								target = scope:new_char
							}
						}
						scope:new_char = {
							agot_multi_duel_set_weapon_for_new_char = yes
							add_trait = squire
							add_trait_xp = {
								trait = squire
								track = knight
								value = 100
							}
							if = {
								limit = {
									NOR = {
										has_trait = education_martial_prowess_1
										has_trait = education_martial_prowess_2
										has_trait = education_martial_prowess_3
										has_trait = education_martial_prowess_4
										has_trait = education_martial_prowess_5
									}
								}
								random_list = {
									3000 = { add_trait = education_martial_prowess_1 }
									3000 = { add_trait = education_martial_prowess_2 }
									2500 = { add_trait = education_martial_prowess_3 }
									1500 = { add_trait = education_martial_prowess_4 }
									1 = { add_trait = education_martial_prowess_5 }

					limit = { NOT = { has_character_flag = tos_non_royal_sellsword_gained } }

		if = {

			if = {
				limit = { is_ai = no }
				ordered_in_list = {
					variable = possible_characters_temp

						add_to_variable_list = {
							name = possible_characters
							target = prev

				}
				ordered_in_list = {
					variable = possible_characters

			else = {
				ordered_in_list = {
					variable = possible_characters_temp

		if = {
			limit = {
				OR = {
					NOT = { has_variable_list = possible_characters }

	option = {
		name = {
			text = {

					triggered_desc = {

							has_character_flag = agot_tos_lost_before_started

						desc = agot_trial_of_seven.1000.a.lost_before_started

						trigger = {
							has_character_flag = agot_tos_need_new_char
						}
						desc = agot_trial_of_seven.1000.a.need_new_char
					}
					triggered_desc = {
						trigger = {
							NOT = { has_character_flag = agot_tos_new_char_created }
							NOT = { has_character_flag = tos_squire_wants_to_be_knighted }

						trigger = {
							has_character_flag = agot_tos_new_char_created
						}

					}

						trigger = {
							has_character_flag = tos_squire_wants_to_be_knighted

		#Add new character to the team
		if = {
			limit = {
				exists = var:si_selected_character

				scope:new_champion = { NOT = { has_character_flag = tos_duelist_already_picked } }
				is_ai = no
			}
			var:si_selected_character = {
				save_scope_as = new_champion
				remove_variable = si_selecting_character
			}
			if = {
				limit = { NOT = { has_character_flag = agot_tos_lost_before_started } }
				custom_tooltip = agot_trial_of_seven.1000.a.tt
			}

			add_to_variable_list = {
				name = multi_duel_tos_temp_list
				target = scope:new_champion
			}
			scope:new_champion = {
				add_character_flag = {
					flag = tos_duelist_already_picked
					days = 100

				exists = scope:new_champion

			add_to_variable_list = {
				name = multi_duel_tos_temp_list
				target = scope:new_champion
			}
			scope:new_champion = {
				add_character_flag = {
					flag = tos_duelist_already_picked
					days = 100

		if = {
			limit = {
				has_character_flag = tos_squire_wants_to_be_knighted
				NOT = { has_character_flag = tos_already_knighted_once }

				KNIGHT_MAKER = scope:duel_leader

				flag = tos_already_knighted_once

			limit = { has_character_flag = agot_tos_new_char_created }

				add_to_variable_list = {
					name = tos_money_owed
					target = scope:new_champion

					target = scope:new_champion

			limit = {
				exists = scope:helper_knight

							}

								domicile.domicile_location = { geographical_region = world_westeros_seven_kingdoms }

									limit = { exists = scope:the_royal_leader.top_liege }

								}
							}

								is_landed = yes
								exists = top_liege
							}
							OR = {
								capital_county.title_province = { geographical_region = world_westeros_seven_kingdoms }
								top_liege = scope:the_royal_leader
								trigger_if = {
									limit = { exists = scope:the_royal_leader.top_liege }
									top_liege = scope:the_royal_leader.top_liege
								}
							}
						}
						trigger_else = {
							exists = domicile.domicile_location.county.holder.top_liege
							OR = {
								domicile.domicile_location = { geographical_region = world_westeros_seven_kingdoms }
								domicile.domicile_location.county.holder.top_liege = scope:the_royal_leader
								trigger_if = {
									limit = { exists = scope:the_royal_leader.top_liege }
									domicile.domicile_location.county.holder.top_liege = scope:the_royal_leader.top_liege
								}
							}
						}
					}
					trigger_event = agot_trial_of_seven.1002
				}
			}
			#Or to the accuser if no player observer exists
			else = {
				trigger_event = agot_trial_of_seven.1001
			}
		}

		}

			limit = {

				if = {
					limit = {
						scope:leader_lost_before_started = {
							is_playable_character = no
							NOT = { is_courtier_of = scope:final_victor }
						}
					}
					scope:final_victor = { add_courtier = scope:leader_lost_before_started }
				}
			}
		}

			limit = {
				exists = scope:final_victor.primary_title

			}

				remove_variable = trial_of_seven_cooldown

					name = trial_of_seven_cooldown

			}
		}
	}
