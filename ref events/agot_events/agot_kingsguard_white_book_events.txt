namespace = agot_kingsguard_white_book

agot_kingsguard_white_book.0001 = {
	type = character_event
	title = agot_kingsguard_white_book.0001.t
	desc = agot_kingsguard_white_book.0001.desc
	window = agot_white_book_event

	theme = white_book

	left_portrait = {
		character = root
		animation = reading
	}

	immediate = {
		every_in_global_list = {
			variable = kingsguard_white_book_pages
			root = {
				if = {
					limit = {
						NOT = { exists = var:selected_white_book_page }
					}
					set_variable = { name = selected_white_book_page value = prev }
				}
			}
		}
	}

	option = {
		name = agot_kingsguard_white_book.0001.a
		var:selected_white_book_page.var:kingsguard_member ?= { save_scope_as = selected_kingsguard }
		trigger_event = agot_kingsguard_white_book.0002
	}

	option = {
		name = agot_kingsguard_white_book.0001.b
		remove_variable = selected_white_book_page
	}
}

agot_kingsguard_white_book.0002 = {
	type = character_event
	title = agot_kingsguard_white_book.0002.t
	desc = agot_kingsguard_white_book.0002.desc
	window = agot_white_book_event

	theme = white_book

	right_portrait = {
		character = scope:selected_kingsguard
	}

	immediate = {
		agot_kingsguard_get_page_effect = { KINGSGUARD = scope:selected_kingsguard }
	}

	option = {
		name = agot_kingsguard_white_book.0002.a
		trigger_event = agot_kingsguard_white_book.0001
	}

	option = {
		name = agot_kingsguard_white_book.0002.b
	}

	after = {
		remove_variable = selected_white_book_page
	}
}

agot_kingsguard_white_book.1000 = {
	hidden = yes

	immediate = {
		if = {
			limit = { exists = scope:dead_kingsguard }
			agot_kingsguard_update_white_book_page_died_update = { KINGSGUARD = scope:dead_kingsguard KING = scope:king }
		}
		else_if = {
			limit = {
				exists = scope:removed_kg
				scope:removed_kg = { has_trait = nightswatch }
			}
			agot_kingsguard_update_white_book_page_sent_to_nightswatch_update = { KINGSGUARD = scope:removed_kg KING = scope:king }
		}
		else_if = {
			limit = {
				exists = scope:removed_kg
			}
			agot_kingsguard_update_white_book_page_removed_update = { KINGSGUARD = scope:removed_kg KING = scope:king }
		}
	}
}
