	group = animal_skull_type
	trigger = {
		trigger_if = {
			limit = { exists = scope:owner.var:animal_type }
			OR = {
				scope:owner.var:animal_type = flag:lizard_lion
				scope:owner.var:animal_type = flag:none
			}
		}
	}
	weight = 4
}
