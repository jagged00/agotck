# Assigns circumstantial data to character to call when determining title, name, and format
assign_character_parameter_effect = {
	$SCOPE$ ?= {

		### GENERAL CHARACTER PARAMETERS ###
		#1. Lowborn/Highborn
		if = {
			limit = { is_lowborn = yes }
			save_temporary_scope_value_as = {
				name = lowborn
				value = yes
			}
		}

		#2. Bastardy
		if = {
			limit = { agot_bastard_surname_trigger = yes }
			save_temporary_scope_value_as = {
				name = bastardy
				value = yes
			}
		}

		#3. Knighthood
		if = {
			limit = { is_agot_knight_trigger = yes }
			save_temporary_scope_value_as = {
				name = knighthood
				value = yes
			}
		}

		#4. Maester
		if = {
			limit = {
				OR = {
					has_trait = maester
					has_trait = maester_historical
				}
				is_ruler = no
			}
			save_temporary_scope_value_as = {
				name = maester
				value = yes
			}
		}

		#5. Silent Sister
		if = {
			limit = {
				has_trait = silent_sister
				is_ruler = no
			}
			save_temporary_scope_value_as = {
				name = silent_sister
				value = yes
			}
		}

		#6. Septon/Septa
		if = {
			limit = {
				has_trait = septon
				NOT = { primary_title ?= title:k_the_most_devout }
			}
			save_temporary_scope_value_as = {
				name = septon
				value = yes
			}
		}

		#7. Councillor
		if = {
			limit = {
				is_councillor = yes
				is_human = yes
				liege.primary_title.tier ?= tier_hegemony
			}
			save_temporary_scope_value_as = {
				name = councillor
				value = yes
			}
		}

		#8. High Septon
		if = {
			limit = { primary_title ?= title:k_the_most_devout }
			save_temporary_scope_value_as = {
				name = high_septon
				value = yes
			}
		}

		#9. Name is Title
		if = {
			limit = {
				OR = {
					this = character:Warg_King_1
					this = character:NightsQueen_1
				}
			}
			save_temporary_scope_value_as = {
				name = name_is_title
				value = yes
			}
		}

		#10. Regents
		if = {
			limit = {
				is_independent_ruler = no
				NOR = {
					exists = scope:high_septon
					primary_title ?= title:d_kingsguard
				}
				liege ?= {
					OR = {
						has_government = lp_feudal_government
						has_government = feudal_government
					}
				}
				is_diarch_of_target = liege
			}
			save_temporary_scope_value_as = {
				name = regent_status
				value = yes
			}
		}

		### RULER/SPOUSE/SCION CHARACTER PARAMETERS ###
		#1. Parental Ranks
		if = {
			limit = {
				any_parent = {
					even_if_dead = yes
					parent_is_royal = { PARENT = this }
				}
			}
			if = {
				limit = { parent_is_royal = { PARENT = $SCOPE$.father } }
				assign_parent_data = { PARENT = $SCOPE$.father }
				save_temporary_scope_value_as = {
					name = scion_status
					value = flag:royal
				}
			}
			else_if = {
				limit = { parent_is_royal = { PARENT = $SCOPE$.mother } }
				assign_parent_data = { PARENT = $SCOPE$.mother }
				save_temporary_scope_value_as = {
					name = scion_status
					value = flag:royal
				}
			}
		}
		else_if = {
			limit = {
				any_parent = {
					even_if_dead = yes
					OR = {
						any_parent = {
							even_if_dead = yes
							ruler_is_royal = yes
						}
						exists = primary_title
					}
				}
				is_female = yes
				NOT = { exists = scope:bastardy }
			}
			save_temporary_scope_value_as = {
				name = scion_status
				value = flag:active
			}
		}

		#2. Grandparent Ranks
		if = {
			limit = {
				grandparent_is_royal = {
					SCOPE = $SCOPE$
				}
			}
			assign_grandparent_data = { SCOPE = $SCOPE$ }
			if = {
				limit = {
					exists = scope:scion_2_title
				}
				save_temporary_scope_value_as = {
					name = scion_status
					value = flag:royal
				}
			}
		}

		#3. Bestowal Data
		if = {
			limit = {
				is_royal_scion = yes
				is_bestowed_royal_scion = { SCOPE = $SCOPE$ }
			}
			assign_bestowal_data = { SCOPE = $SCOPE$ }
			if = {
				limit = {
					exists = scope:scion_3_title
				}
				save_temporary_scope_value_as = {
					name = scion_status
					value = flag:royal
				}
			}
		}

		#6. Determine Spouse Ranks
		assign_spouse_data = { SCOPE = $SCOPE$ }
		if = {
			limit = { spouse_is_royal = yes }
			save_temporary_scope_value_as = {
				name = spouse_status
				value = flag:royal
			}
		}
		else_if = {
			limit = { exists = scope:spouse_scope.primary_title }
			save_temporary_scope_value_as = {
				name = spouse_status
				value = flag:active
			}
		}

		#7. Determine Personal Ranks
		if = {
			limit = { ruler_is_royal = yes }
			save_temporary_scope_value_as = {
				name = ruler_status
				value = flag:royal
			}
		}
		else_if = {
			limit = { highest_held_title_tier >= tier_barony }
			save_temporary_scope_value_as = {
				name = ruler_status
				value = flag:active
			}
		}
	}
}

# Assign Parent Data
assign_parent_data = {
	if = {
		limit = {
			exists = $PARENT$
			NOR = {
				scope:scion_status ?= flag:royal
				has_trait = bastard
			}
		}
		$PARENT$ ?= { save_temporary_scope_as = parent_scope }
		scope:parent_scope.primary_title ?= { save_temporary_scope_as = scion_1_title }
	}
}

# Assign Grandparent Data
assign_grandparent_data = {
	if = {
		limit = {
			any_close_family_member = {
				even_if_dead = yes
				exists = this
				primary_title.tier >= tier_barony
				is_grandparent_of = $SCOPE$
				NOT = { primary_title ?= title:k_dragonstone }
				house ?= $SCOPE$.house
			}
			NOT = { scope:scion_status ?= flag:royal }
			NOT = { has_trait = bastard }
		}
		ordered_close_family_member = {
			even_if_dead = yes
			limit = {
				exists = primary_title
				is_grandparent_of = $SCOPE$
				NOT = { primary_title ?= title:k_dragonstone }
				house ?= $SCOPE$.house
			}
			order_by = primary_title.tier
			save_temporary_scope_as = grandparent_scope
		}
		scope:grandparent_scope.primary_title ?= { save_temporary_scope_as = scion_2_title }
	}
}

# Assign Historical Scion Data
assign_bestowal_data = {
	character:Ruin_Empress ?= {
		ordered_owned_story = {
			type = scion_title_data
			limit = {
				has_variable_list = scoped_scions_list
				is_target_in_variable_list = { name = scoped_scions_list target = $SCOPE$ }
			}
			order_by = var:scoped_title.tier
			var:scoped_royal ?= { save_temporary_scope_as = relative_scope }
			var:scoped_title ?= { save_temporary_scope_as = scion_3_title }
		}
	}
}

# Assign Spousal Data
assign_spouse_data = {
	if = {
		limit = {
			any_spouse = {
				even_if_dead = yes
				exists = primary_title
			}
		}
		$SCOPE$ = {
			if = {
				limit = {
					any_spouse = {
						exists = this.primary_title
						is_alive = yes
					}
				}
				primary_spouse.primary_title ?= { save_temporary_scope_as = spouse_title }
				primary_spouse ?= { save_temporary_scope_as = spouse_scope }
			}
			else = {
				ordered_spouse = {
					even_if_dead = yes
					limit = {
						exists = this.primary_title
						is_alive = no
					}
					order_by = {
						value = days_since_death
						multiply = -1
					}
					primary_title ?= { save_temporary_scope_as = spouse_title }
					save_temporary_scope_as = spouse_scope
				}
				if = {
					limit = {
						is_female = yes
						is_alive = yes
						any_child = {
							is_alive = yes
							primary_title ?= scope:spouse_title
							agot_flavour_hegemony_is_kingdom_trigger = { CHAR = this }
						}
					}
					save_temporary_scope_value_as = {
						name = queen_mother
						value = flag:active
					}
				}
				else_if = {
					limit = {
						is_female = yes
						is_alive = yes
						trigger_if = {
							limit = { scope:scion_status ?= flag:royal }
							scope:spouse_scope = { ruler_is_royal = yes }
						}
						trigger_else_if = {
							limit = { scope:spouse_scope = { ruler_is_royal = yes } }
							always = yes
						}
						trigger_else = { always = no }
						scope:spouse_scope = {
							OR = {
								agot_is_historical_government_trigger = { GOV = lp_feudal }
								agot_is_historical_government_trigger = { GOV = feudal }
							}
						}
					}
					save_temporary_scope_value_as = {
						name = dowager_status
						value = flag:active
					}
				}
			}
		}
	}
}

	every_county = {
		every_past_holder = {
			limit = { is_alive = no }
			if = { ### WESTEROS ###
				limit = { primary_title.title_capital_county.title_province ?= { geographical_region = world_westeros_seven_kingdoms_without_wall } }
				if = {
					limit = { days_since_death < days_since_conquest }
					save_temporary_scope_as = temp_ruler_scope
					assign_government_data_effect = {
						GOV = lp_feudal
						SCOPE = scope:temp_ruler_scope
					}
				}
				else = {
					save_temporary_scope_as = temp_ruler_scope
					assign_government_data_effect = {
						GOV = feudal
						SCOPE = scope:temp_ruler_scope
					}
					if = {
						limit = { primary_title.tier >= tier_empire }
						save_temporary_scope_as = temp_ruler_scope
						add_to_global_variable_list = {
							name = historical_independent_ruler_list
							target = scope:temp_ruler_scope
						}
						make_trait_inactive = agot_was_king
					}
					else_if = {
						limit = {
							primary_title.tier = tier_kingdom
							trigger_if = {
								limit = {
									primary_title.title_province ?= { geographical_region = world_westeros_dorne }
								}
								days_since_death > days_since_nymerias_war
							}
							trigger_else = {
								NOT = { agot_flavour_is_lord_in_westeros_trigger = { CHAR = this } }
							}
						}
						save_temporary_scope_as = temp_ruler_scope
						add_to_global_variable_list = {
							name = historical_independent_ruler_list
							target = scope:temp_ruler_scope
						}
						make_trait_inactive = agot_was_king
					}
					primary_title ?= {

							limit = {
								is_alive = no
								days_since_death >= days_since_conquest
							}
							save_temporary_scope_as = temp_ruler_scope
							assign_government_data_effect = {
								GOV = feudal
								SCOPE = scope:temp_ruler_scope
							}
							if = {
								limit = { primary_title.tier >= tier_empire }
								save_temporary_scope_as = temp_ruler_scope

									name = historical_independent_ruler_list
									target = scope:temp_ruler_scope
								}

							}
							else_if = {

										}

									}
									trigger_else = {
										NOT = { agot_flavour_is_lord_in_westeros_trigger = { CHAR = this } }
									}
								}
								save_temporary_scope_as = temp_ruler_scope
								add_to_global_variable_list = {
									name = historical_independent_ruler_list
									target = scope:temp_ruler_scope
								}
								make_trait_inactive = agot_was_king
							}
						}
					}
				}
			}
			else_if = {
				limit = { primary_title.title_capital_county.title_province ?= { geographical_region = world_essos } }
				if = {
					limit = { primary_title.holder ?= { agot_ruler_is_government_trigger = { GOV = administrative } } }

					assign_government_data_effect = {
						GOV = administrative
						SCOPE = scope:temp_ruler_scope
					}
				}
				else_if = {
					limit = { primary_title ?= title:k_the_stepstones }
					save_temporary_scope_as = temp_ruler_scope
					if = {
						limit = {
							scope:temp_ruler_scope ?= {
								OR = {
									this = character:Drahar_8
									this = character:Saan_13
									this = character:Naelor_9
									this = character:Daemon_Successor_1
									this = character:Daemon_Successor_2
									this = character:Daemon_Successor_3
									this = character:Daemon_Successor_4
								}
							}
						}
						assign_government_data_effect = {

					}

						}
						assign_government_data_effect = {
							GOV = feudal

agot_bestow_scion_status_effect = {
	if = {
		limit = {
			exists = $SCOPE$
			NOT = { is_bestowed_royal_scion = { SCOPE = $SCOPE$ } }
		}
		custom_description = {
			text = AGOT_CREATE_PRINCELING
			subject = $SCOPE$
			object = $SCOPED_TITLE$
			if = {
				limit = { exists = $SCOPED_TITLE$ }
				character:Ruin_Empress ?= {
					set_variable = {
						name = scoped_scion
						value = $SCOPE$
					}
					set_variable = {
						name = scoped_royal
						value = $SCOPED_ROYAL$
					}
					set_variable = {
						name = scoped_title
						value = $SCOPED_TITLE$
					}
					if = {
						limit = {
							any_owned_story = {
								type = scion_title_data
								var:scoped_royal ?= $SCOPED_ROYAL$
								var:scoped_title ?= $SCOPED_TITLE$
							}
						}
						random_owned_story = {
							limit = {
								story_type = scion_title_data
								var:scoped_royal ?= $SCOPED_ROYAL$
								var:scoped_title ?= $SCOPED_TITLE$
							}
							add_to_variable_list = {
								name = scoped_scions_list
								target = $SCOPE$
							}
						}
					}
					else = {
						create_story = {
							type = scion_title_data
							save_scope_as = scion_title_data
						}
					}
					remove_variable = scoped_scion
					remove_variable = scoped_royal
					remove_variable = scoped_title
				}
			}
		}
		if = {
			limit = { $SCOPE$ = { is_alive = yes } }
			$SCOPE$ ?= {
				add_prestige = 300
				if = {
					limit = { always = $OPINION$ }
					switch = {
						trigger = is_male
						yes = {
							add_opinion = {
								target = $SCOPED_ROYAL$
								modifier = bestowed_scion_status_male_opinion
							}
						}
						no = {
							add_opinion = {
								target = $SCOPED_ROYAL$
								modifier = bestowed_scion_status_female_opinion
							}
						}
					}
				}
			}
		}
	}
}

agot_on_birth_bestow_scion_status_effect = {
	if = {
		limit = {
			NOT = { $CHILD$ = { is_royal_scion = yes } }
		}
		$CHILD$ ?= {
			if = { # Grandchild (Iron Throne)
				limit = {
					exists = title:h_the_iron_throne
					exists = title:h_the_iron_throne.holder
					any_parent = {
						even_if_dead = yes
						primary_title ?= title:k_dragonstone
						is_primary_heir_of = title:h_the_iron_throne.holder
					}
					is_grandchild_of = title:h_the_iron_throne.holder
				}
				title:k_dragonstone.holder = { save_scope_as = scoped_relative }
				title:h_the_iron_throne.holder ?= { save_scope_as = scoped_royal }
			}
			else_if = { # Great-Grandchild (Iron Throne)
				limit = {
					exists = title:h_the_iron_throne
					exists = title:h_the_iron_throne.holder
					any_close_family_member = {
						even_if_dead = yes
						OR = {
							is_parent_of = $CHILD$
							is_grandparent_of = $CHILD$
						}
						OR = {
							is_child_of = title:h_the_iron_throne.holder
							is_grandchild_of = title:h_the_iron_throne.holder
						}
						agot_flavour_is_prince_dragonstone_trigger = { CHAR = this }
						primary_title ?= title:k_dragonstone
					}
					title:h_the_iron_throne.holder ?= {
						NOR = {
							is_parent_of = $CHILD$
							is_grandparent_of = $CHILD$
						}
					}
				}
				random_close_family_member = {
					even_if_dead = yes
					limit = {
						exists = title:h_the_iron_throne
						exists = title:h_the_iron_throne.holder
						OR = {
							is_parent_of = $CHILD$
							is_grandparent_of = $CHILD$
						}
						OR = {
							is_child_of = title:h_the_iron_throne.holder
							is_grandchild_of = title:h_the_iron_throne.holder
						}
						agot_flavour_is_prince_dragonstone_trigger = { CHAR = this }
						primary_title ?= title:k_dragonstone
					}
					save_scope_as = scoped_relative
				}
				title:h_the_iron_throne.holder ?= { save_scope_as = scoped_royal }
			}
			else_if = { # Nibling
				limit = {
					any_close_or_extended_family_member = {
						even_if_dead = yes
						ruler_is_royal = yes
						NOT = { primary_title ?= title:k_dragonstone }
						is_uncle_or_aunt_of = $CHILD$
						house ?= $CHILD$.house
						any_sibling = {
							even_if_dead = yes
							is_parent_of = $CHILD$
							is_royal_scion = yes
							NOT = { ruler_is_royal = yes }
							house ?= prev.house
						}
					}
				}
				ordered_close_or_extended_family_member = {
					even_if_dead = yes
					limit = {
						ruler_is_royal = yes
						NOT = { primary_title ?= title:k_dragonstone }
						is_uncle_or_aunt_of = $CHILD$
						house ?= $CHILD$.house
						any_sibling = {
							even_if_dead = yes
							is_parent_of = $CHILD$
							is_royal_scion = yes
							NOT = { ruler_is_royal = yes }
							house ?= prev.house
						}
					}
					order_by = primary_title.tier
					save_scope_as = scoped_royal
					random_sibling = {
						even_if_dead = yes
						limit = {
							is_parent_of = $CHILD$
							is_royal_scion = yes
							NOT = { ruler_is_royal = yes }
							house ?= prev.house
						}
						save_scope_as = scoped_relative
					}
				}
			}
		}
		scope:scoped_royal ?= {
			trigger_event = {
				id = agot_title_and_name_events.0001
				days = 1
			}
		}
	}
	else_if = {
		limit = { is_royal_scion_by_birth = { SCOPE = $CHILD$ } }
		if = {
			limit = { parent_is_royal = { PARENT = $CHILD$.father } }
			$CHILD$.father ?= { save_scope_as = scoped_royal }
		}
		else_if = {
			limit = { parent_is_royal = { PARENT = $CHILD$.mother } }
			$CHILD$.mother ?= { save_scope_as = scoped_royal }
		}
		else_if = {
			limit = { grandparent_is_royal = { SCOPE = $CHILD$ } }
			assign_grandparent_data = { SCOPE = $CHILD$ }
			scope:grandparent_scope ?= { save_scope_as = scoped_royal }
		}
		agot_bestow_scion_status_effect = {
			SCOPE = $CHILD$
			SCOPED_ROYAL = scope:scoped_royal
			SCOPED_TITLE = scope:scoped_royal.primary_title
			OPINION = no
		}
	}

# Apply character title localization
apply_character_title_data = {
	if = {
		limit = {
			exists = $SCOPED_TITLE$
			exists = $SCOPE$
		}
		$SCOPE$ ?= {
			if = {
				limit = { is_titled_character = yes }

				### RULERS & SPOUSES ###
				if = {
					limit = { is_titled_ruler_or_spouse = yes }

					### CLAIMANT OVERRIDE ###
					if = {
						limit = {
							OR = {
								has_character_flag = agot_claimant_king
								scope:spouse_scope ?= { has_character_flag = agot_claimant_king }
							}
						}
						if = {
							limit = {
								OR = {
									any_character_war = {
										OR = {
											casus_belli = {
												any_target_title = {
													this = title:e_the_stormlands
												}
											}
											AND = {
												prev = { has_title = title:e_the_stormlands }
												OR = {
													using_cb = independence_war
													using_cb = agot_independence_war
												}
											}
										}
									}
									any_spouse = {
										even_if_dead = yes
										any_character_war = {
											OR = {
												casus_belli = {
													any_target_title = {
														this = title:e_the_stormlands
													}
												}
												AND = {
													prev = { has_title = title:e_the_stormlands }
													OR = {
														using_cb = independence_war
														using_cb = agot_independence_war
													}
												}
											}
										}
									}
								}
							}

							set_title_localization_effect = { SCOPE = $SCOPE$ KEY = "king_stormlands" GENDER = male }
							set_title_localization_effect = { SCOPE = $SCOPE$ KEY = "queen_stormlands" GENDER = female }
						}
						else_if = {
							limit = {
								OR = {
									AND = {
										culture = {
											OR = {
												has_name_list = name_list_salt_dornish
												has_name_list = name_list_sand_dornish
												has_name_list = name_list_rhoynish
												has_cultural_pillar = heritage_rhoynar
											}
										}
										any_character_war = {
											OR = {
												casus_belli = {
													any_target_title = {
														this = title:e_dorne
													}
												}
												AND = {
													prev = { has_title = title:e_dorne }
													OR = {
														using_cb = independence_war
														using_cb = agot_independence_war
													}
												}
											}
										}
									}
									any_spouse = {
										even_if_dead = yes
										culture = {
											OR = {
												has_name_list = name_list_salt_dornish
												has_name_list = name_list_sand_dornish
												has_name_list = name_list_rhoynish
												has_cultural_pillar = heritage_rhoynar
											}
										}
										any_character_war = {
											OR = {
												casus_belli = {
													any_target_title = {
														this = title:e_dorne
													}
												}
												AND = {
													prev = { has_title = title:e_dorne }
													OR = {
														using_cb = independence_war
														using_cb = agot_independence_war
													}
												}
											}
										}
									}
								}
							}

							set_title_localization_effect = { SCOPE = $SCOPE$ KEY = "prince" GENDER = male }
							set_title_localization_effect = { SCOPE = $SCOPE$ KEY = "princess" GENDER = female }
						}
						else_if = {
							limit = { always = yes }

							set_title_localization_effect = { SCOPE = $SCOPE$ KEY = "king" GENDER = male }
							set_title_localization_effect = { SCOPE = $SCOPE$ KEY = "queen" GENDER = female }
						}
					}

					### LP_FEUDAL ###
					else_if = {
						limit = {
							OR = {
								agot_localization_government_trigger = { GOV = lp_feudal SPOUSE_SCOPE = scope:spouse_scope }
								AND = {
									agot_localization_government_trigger = { GOV = feudal SPOUSE_SCOPE = scope:spouse_scope }
									OR = {
										agot_is_part_of_paramountcy_realm = yes
										scope:spouse_scope ?= { agot_is_part_of_paramountcy_realm = yes }
									}
								}
							}
						}

apply_character_title_suffix_data = {
	### SET SCOPED HOLDER ###
	if = {
		limit = { scope:ruler_status ?= flag:royal }
		$SCOPE$ = { save_temporary_scope_as = scoped_title_holder }
	}
	else_if = {
		limit = { scope:spouse_status ?= flag:royal }
		scope:spouse_scope ?= { save_temporary_scope_as = scoped_title_holder }
	}
	else_if = {
		limit = {
			scope:scion_status ?= flag:royal
			$SCOPE$ ?= {
				is_titled_ruler_or_spouse = no
				NOR = {
					exists = scope:maester
					exists = scope:silent_sister
					exists = scope:septon
					agot_localization_government_trigger = { GOV = nights_watch SPOUSE_SCOPE = scope:spouse_scope }
					agot_localization_government_trigger = { GOV = first_ranger SPOUSE_SCOPE = scope:spouse_scope }
				}
			}
		}
		if = {
			limit = { exists = scope:scion_1_title }
			scope:parent_scope ?= { save_temporary_scope_as = scoped_title_holder }
		}
		else_if = {
			limit = { exists = scope:scion_2_title }
			scope:grandparent_scope ?= { save_temporary_scope_as = scoped_title_holder }
		}
		else_if = {
			limit = { exists = scope:scion_3_title }
			scope:relative_scope ?= { save_temporary_scope_as = scoped_title_holder }
		}
	}
	else = {
		if = {
			limit = {
				OR = {
					AND = {
						scope:ruler_status ?= flag:active
						OR = {
							agot_is_historical_government_trigger = { GOV = lp_feudal }
							agot_is_historical_government_trigger = { GOV = feudal }
						}
					}
					AND = {
						scope:spouse_status ?= flag:active
						scope:spouse_scope ?= {
							OR = {
								agot_is_historical_government_trigger = { GOV = lp_feudal }
								agot_is_historical_government_trigger = { GOV = feudal }
							}
						}
					}
					scope:scion_status ?= flag:active
					AND = {
						exists = scope:councillor
						NOR = {
							exists = scope:knighthood
							exists = scope:maester
							exists = scope:silent_sister
							exists = scope:septon
						}
					}
				}
				exists = scope:regent_status
			}
			$SCOPE$.liege ?= { save_temporary_scope_as = scoped_title_holder }
		}
		else_if = {
			limit = { scope:ruler_status ?= flag:active }
			$SCOPE$ = { save_temporary_scope_as = scoped_title_holder }
		}
		else_if = {
			limit = { scope:spouse_status ?= flag:active }
			scope:spouse_scope ?= { save_temporary_scope_as = scoped_title_holder }
		}
	}

	### SET SUFFIX DATA ###
	if = {
		limit = {
			scope:scoped_title_holder ?= { is_alive = no }
			$SCOPED_TITLE$ ?= {
				has_variable_list = previous_title_names
				any_in_list = {
					variable = previous_title_names
					days_since_title_name_change <= scope:scoped_title_holder.days_since_death
				}
			}
		}
		$SCOPED_TITLE$ = {
			ordered_in_list = {
				variable = previous_title_names
				limit = {
					days_since_title_name_change <= scope:scoped_title_holder.days_since_death
				}
				order_by = {
					value = days_since_title_name_change
					multiply = 1
				}
				save_temporary_scope_as = previous_title_name
			}
			if = {
				limit = {
					exists = scope:previous_title_name
					exists = $SCOPED_TITLE$
				}
				if = {
					limit = { scope:suffix_format = yes }
					custom_description_no_bullet = {
						text = PAST_TITLE_SUFFIX
						subject = $SCOPE$
						object = scope:previous_title_name
					}
				}
				else = {
					custom_description_no_bullet = {
						text = PAST_TITLE_SUFFIX_NO_FORMAT
						subject = $SCOPE$
						object = scope:previous_title_name
					}
				}
			}
		}
	}
	else = {
		if = {
			limit = { exists = $SCOPED_TITLE$ }
			if = {
				limit = { scope:suffix_format = yes }
				custom_description_no_bullet = {
					text = TITLE_SUFFIX
					subject = $SCOPE$
					object = $SCOPED_TITLE$
				}
			}
			else = {
				custom_description_no_bullet = {
					text = TITLE_SUFFIX_NO_FORMAT
					subject = $SCOPE$
					object = $SCOPED_TITLE$
				}
			}
		}
	}
}

					if = {

				limit = { exists = $SCOPE$.mother }
				$SCOPE$.mother ?= {

						limit = { exists = house }

			}

			if = {
				limit = {
					NOR = {
						has_trait = bastard
						trigger_if = {
							limit = { has_trait = twin }
							any_sibling = {
								even_if_dead = yes
								NOT = { has_trait = bastard }
								trigger_if = {
									limit = { $SCOPE$ ?= { is_female = yes } }
									is_female = yes
								}
								trigger_else = {
									is_female = no
								}
								house ?= $SCOPE$.house
								agot_years_since_birth = $SCOPE$.agot_years_since_birth
							}
						}
					}

					custom_description_no_bullet = {
						text = BASTARD_BORN
						subject = $SCOPE$
					}
				}
				else = {
					custom_description_no_bullet = {
						text = BASTARD_BORN
						subject = $SCOPE$
					}
				}
			}
		}
	}
