agot_destroy_or_downgrade_pirate_ship_building_3_effect = {
	if = {
		limit = { has_domicile_building = $BUILDING$_03 }
		lower_domicile_building_no_refund = $BUILDING$_03
	}
	else_if = {
		limit = { has_domicile_building = $BUILDING$_02 }
		lower_domicile_building_no_refund = $BUILDING$_02
	}
	else = { remove_domicile_building_no_refund = $BUILDING$_01 }
}

agot_destroy_random_pirate_ship_building_variable_effect = { # Domicile scope
	save_scope_as = raid_domicile
	if = {
		limit = { has_variable = destroyed_building_variable }
		remove_variable = destroyed_building_variable
	}
	random_list = {
		10 = {
			trigger = { has_domicile_building_or_higher = agot_captains_quarters_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 0
				months = 1
			}
		}
		10 = {
			trigger = { has_domicile_building_or_higher = agot_ship_galley_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 1
				months = 1
			}
		}
		10 = {
			trigger = { has_domicile_building_or_higher = agot_forecastle_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 2
				months = 1
			}
		}
		10 = {
			trigger = { has_domicile_building_or_higher = agot_ships_armory_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 3
				months = 1
			}
		}
		10 = {
			trigger = { has_domicile_building_or_higher = agot_ships_brig_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 4
				months = 1
			}
		}
		10 = {
			trigger = { has_domicile_building_or_higher = agot_berth_deck_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 5
				months = 1

		}
		10 = {

				name = destroyed_building_variable
				value = 6
				months = 1
			}

		10 = {
			trigger = { has_domicile_building_or_higher = agot_ships_hold_01 }
			set_variable = {
				name = destroyed_building_variable
				value = 7
				months = 1
			}
		}
	}
}

agot_destroy_random_pirate_ship_building_effect = {
	if = {
		limit = { has_variable = destroyed_building_variable }
		switch = {
			trigger = var:destroyed_building_variable
			0 = {
				agot_destroy_or_downgrade_pirate_ship_building_3_effect = { BUILDING = agot_captains_quarters }
			}
			1 = {
				agot_destroy_or_downgrade_pirate_ship_building_3_effect = { BUILDING = agot_ship_galley }
			}
			2 = {
				agot_destroy_or_downgrade_pirate_ship_building_3_effect = { BUILDING = agot_forecastle }
			}
