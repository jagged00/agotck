### LEADER SETUP
# There has to be 2 leaders
# This should be the first of the Multi Duel System
# SIDE_A_LEADER - leader of the A side
# SIDE_B_LEADER - leader of the B side
# DUEL_BACKGROUND - terrain_type / army_camp / battlefield
# EVENT_BACKGROUND_REFERENCE_CHARACTER - this character's location will be used for certain event backgrounds (currently used only if DUEL_BACKGROUND is "tos_bg") (Needs to be landed.)
# terrain_type - Terrain of your current location.
# army_camp - Usual army camp background.
# battlefield - Usual battlefield background.
# SIDE_A_TEAM_SELECTION - Team setup event for team A. Should be unique for every feature.
# SIDE_B_TEAM_SELECTION - Team setup event for team B. Should be unique for every feature.
# UNIQUE_STORY_NAME - This is the story identifier. Should be unique for every feature.
# STORY_VARIANT - #Put "none" if not needed.
# END_STORY_ON - dead_leader / dead_every
# dead_leader - Ends the the story for everyone when one of the leader dies.
# dead_every - Ends the story when everyone died on one side.

multi_duel_leaders_effect = {
	$SIDE_A_LEADER$ = { save_scope_as = side_a_leader }
	$SIDE_B_LEADER$ = { save_scope_as = side_b_leader }
	save_scope_value_as = {
		name = leader_a_event
		value = event_id:$SIDE_A_TEAM_SELECTION$
	}
	save_scope_value_as = {
		name = leader_b_event
		value = event_id:$SIDE_B_TEAM_SELECTION$
	}

	multi_duel_participant_effect = {
		DUELIST = scope:side_a_leader
		LEADER = yes
		SIDE = side_a
		DUELIST_NUM = 0
	}

	multi_duel_participant_effect = {
		DUELIST = scope:side_b_leader
		LEADER = yes
		SIDE = side_b
		DUELIST_NUM = 0
	}

	scope:side_a_leader = {
		add_character_flag = multi_duel_sudden_death_protection
		set_variable = {
			name = opponent_multi_duel_leader
			value = scope:side_b_leader
		}
		set_variable = {
			name = match_rule
			value = flag:$END_STORY_ON$
		}
		set_variable = {
			name = unqiue_story_name
			value = flag:$UNIQUE_STORY_NAME$
		}
		set_variable = {
			name = count_of_team_members
			value = 1
		}
		set_variable = {
			name = duel_background
			value = flag:$DUEL_BACKGROUND$
		}
		set_variable = {
			name = event_bg_ref_char_capital
			value = $EVENT_BACKGROUND_REFERENCE_CHARACTER$.capital_county
			years = 3
		}
		set_variable = {
			name = og_side_a_leader
			value = scope:side_a_leader
		}
		set_variable = {
			name = og_side_b_leader
			value = scope:side_b_leader
		}
		set_variable = {
			name = story_variant
			value = flag:$STORY_VARIANT$
		}
		#ToS
		if = {
			limit = {
				exists = var:unqiue_story_name
				var:unqiue_story_name = flag:tos_story
			}
			add_to_variable_list = {
				name = potential_multi_duel_bonding_list
				target = scope:side_a_leader
				years = 5
			}
			add_to_variable_list = {
				name = potential_multi_duel_bonding_opponent_list
				target = scope:side_b_leader
				years = 5
			}
		}
		#ToS Succession Crisis

			limit = {

		}

			limit = {
				exists = var:story_variant
				var:story_variant = flag:usurpation

			}

				every_in_list = {
					variable = usurpation_titles
					scope:side_a_leader = {
						add_to_variable_list = {
							name = usurpation_titles
							target = prev
						}
					}
				}
			}
		}
	}
	scope:side_b_leader = {
		add_character_flag = multi_duel_sudden_death_protection
		set_variable = {
			name = opponent_multi_duel_leader
			value = scope:side_a_leader
		}
		set_variable = {
			name = match_rule
			value = flag:$END_STORY_ON$
		}

			name = unqiue_story_name

			value = flag:$DUEL_BACKGROUND$

				target = scope:side_b_leader

			}
			add_to_variable_list = {
				name = potential_multi_duel_bonding_opponent_list
				target = scope:side_a_leader
				years = 5
			}

		if = {
			limit = {

	scope:side_a_leader = { trigger_event = { saved_event_id = scope:leader_a_event } }

# $DUELIST$ is character
# $LEADER$ is only yes/no, sets a flag
# $SIDE$ is only side_a/side_b, sets a flag
# $DUELIST_NUM$ is a number, the identifier of the duelist for easier check, set to 0 if not needed

						limit = { any_owned_story = { story_type = multi_duel_leader_story } }

								name = count_of_team_members

							}
						}
					}

					if = {
						limit = {
							exists = scope:side_a_leader.var:unqiue_story_name

								years = 5

					}

								exists = var:story_variant

multi_duel_battle_theme_effect = {
	#Handle Observers
	if = {
		limit = {
			any_player = {
				has_character_flag = multi_duel_just_observed
			}
		}
		every_player = {
			limit = { has_character_flag = multi_duel_just_observed }
			remove_character_flag = multi_duel_just_observed
		}
	}

	#TOS STORY
	if = {
		limit = {
			exists = var:unqiue_story_name
			var:unqiue_story_name = flag:tos_story
			exists = var:story_variant
			OR = {

		name = duel_yield

		30 = {
			trigger = {
				$SIDE_A_LEADER$ = {
					NOT = { agot_multi_duelist_cant_continue = yes }
					NOT = { has_character_flag = in_multi_duel }
					NOT = { has_character_flag = last_man_standing }
					NOT = { has_character_flag = multi_duel_sudden_death_protection }
				}
				$SIDE_B_LEADER$ = {
					NOT = { agot_multi_duelist_cant_continue = yes }
					NOT = { has_character_flag = in_multi_duel }
					NOT = { has_character_flag = last_man_standing }
					NOT = { has_character_flag = multi_duel_sudden_death_protection }
				}
			}
			$SIDE_A_LEADER$ = {
				add_character_flag = {
					flag = chance_to_fight_in_multiduel
					days = 1
				}
			}
			$SIDE_B_LEADER$ = {
				add_character_flag = {
					flag = chance_to_fight_in_multiduel
					days = 1
				}
			}
		}
		#70% chance for Duelist v Duelist (including the Leaders)
		70 = {
			$SIDE_A_LEADER$ = {
				if = {
					limit = { any_owned_story = { story_type = multi_duel_leader_story } }
					random_owned_story = {
						limit = {
							story_type = multi_duel_leader_story
							has_variable_list = side_a_duelists
						}
						every_in_list = {
							variable = side_a_duelists
							limit = {
								NOT = { agot_multi_duelist_cant_continue = yes }
								NOT = { has_character_flag = in_multi_duel }
								NOT = { has_character_flag = multi_duel_sudden_death_protection }
							}
							add_character_flag = {
								flag = chance_to_fight_in_multiduel
								days = 1
							}
						}
					}
				}
			}
			$SIDE_B_LEADER$ = {
				if = {
					limit = { any_owned_story = { story_type = multi_duel_leader_story } }
					random_owned_story = {
						limit = {
							story_type = multi_duel_leader_story

								NOT = { has_character_flag = in_multi_duel }
								NOT = { has_character_flag = multi_duel_sudden_death_protection }
							}
							add_character_flag = {
								flag = chance_to_fight_in_multiduel
								days = 1
							}
						}
					}
				}
			}
		}
	}

	$SIDE_A_LEADER$ = {
		save_scope_as = side_a_leader

			limit = { any_owned_story = { story_type = multi_duel_leader_story } }

				limit = { has_character_flag = last_man_standing }

						}
						add_character_flag = in_multi_duel
						save_scope_as = duelist_in_side_a
					}
					every_in_list = {
						variable = side_a_duelists
						limit = {
							NOT = { agot_multi_duelist_cant_continue = yes }
							NOT = { has_character_flag = in_multi_duel }
						}
						add_character_flag = idle_in_multi_duel
						if = {
							limit = { has_character_flag = chance_to_fight_in_multiduel }
							remove_character_flag = chance_to_fight_in_multiduel
						}
						if = {
							limit = { has_character_flag = multi_duel_sudden_death_protection }
							remove_character_flag = multi_duel_sudden_death_protection
						}
					}
				}
			}
			scope:duelist_in_side_a = {
				if = {
					limit = {
						this = scope:side_a_leader
						NOT = { has_character_flag = last_man_standing }
					}
					set_new_multi_duel_leader = yes
				}
			}
		}
	}
	$SIDE_B_LEADER$ = {
		save_scope_as = side_b_leader
		if = {
			limit = { any_owned_story = { story_type = multi_duel_leader_story } }
			if = {
				limit = { has_character_flag = last_man_standing }
				save_scope_as = duelist_in_side_b
			}
			else = {
				random_owned_story = {
					limit = {
						story_type = multi_duel_leader_story
						has_variable_list = side_b_duelists
					}
					random_in_list = {
						variable = side_b_duelists
						limit = {
							NOT = { agot_multi_duelist_cant_continue = yes }
							NOT = { has_character_flag = in_multi_duel }
							has_character_flag = chance_to_fight_in_multiduel
						}
						add_character_flag = in_multi_duel
						save_scope_as = duelist_in_side_b
					}
					every_in_list = {
						variable = side_b_duelists
						limit = {
							NOT = { agot_multi_duelist_cant_continue = yes }
							NOT = { has_character_flag = in_multi_duel }
						}
						add_character_flag = idle_in_multi_duel
						if = {
							limit = { has_character_flag = chance_to_fight_in_multiduel }
							remove_character_flag = chance_to_fight_in_multiduel
						}
						if = {
							limit = { has_character_flag = multi_duel_sudden_death_protection }
							remove_character_flag = multi_duel_sudden_death_protection
						}
					}
				}
			}
			scope:duelist_in_side_b = {
				if = {
					limit = {
						this = scope:side_b_leader
						NOT = { has_character_flag = last_man_standing }
					}
					set_new_multi_duel_leader = yes
				}
			}
		}
	}
	scope:duelist_in_side_a.var:opponent_multi_side_leader = { save_scope_as = side_b_leader }
	scope:duelist_in_side_b.var:opponent_multi_side_leader = { save_scope_as = side_a_leader }
	scope:duelist_in_side_a = {
		if = {
			limit = { has_character_flag = last_man_standing }
			scope:side_b_leader = { add_character_flag = check_for_dead_leader }
		}
	}
	scope:duelist_in_side_b = {
		if = {
			limit = { has_character_flag = last_man_standing }
			scope:side_a_leader = { add_character_flag = check_for_dead_leader }
		}
	}

	save_scope_value_as = {
		name = player_duel_event
		value = event_id:$PLAYER_DUEL_EVENT$
	}

	#Duel Events
	if = {
		limit = {
			OR = {
				AND = {
					exists = scope:duelist_in_side_a
					scope:duelist_in_side_a = { is_ai = no }
				}
				AND = {
					exists = scope:duelist_in_side_b
					scope:duelist_in_side_b = { is_ai = no }
				}
			}
		}
		if = {
			limit = {
				exists = scope:duelist_in_side_a
				scope:duelist_in_side_a = { is_ai = no }
			}
			scope:duelist_in_side_a = { trigger_event = { saved_event_id = scope:player_duel_event } }
		}
		else_if = {
			limit = {
				exists = scope:duelist_in_side_b
				scope:duelist_in_side_b = { is_ai = no }
			}
			scope:duelist_in_side_b = { trigger_event = { saved_event_id = scope:player_duel_event } }
		}
	}
	else_if = {
		limit = {
			AND = {
				exists = scope:duelist_in_side_a
				scope:duelist_in_side_a = { is_ai = yes }
				exists = scope:duelist_in_side_b
				scope:duelist_in_side_b = { is_ai = yes }
			}
		}
		scope:duelist_in_side_a = { multi_duel_battle_ai = yes }
	}
}

set_new_multi_duel_leader = {
	if = {
		limit = {
			exists = var:duelist_heir

			var:match_rule = flag:dead_every
		}
		if = {
			limit = { has_character_flag = last_man_standing }
			remove_variable = duelist_heir
		}
		else = {
			save_scope_as = multi_duel_leader_story_owner
			var:opponent_multi_duel_leader = { save_scope_as = opponent_multi_duel_leader }
			var:duelist_heir = { save_scope_as = duelist_heir }
			#Next Leader
			scope:duelist_heir = {
				if = {
					limit = {
						exists = var:duelist_liege
						exists = var:duelist_role
					}
					remove_variable = duelist_liege
					remove_variable = duelist_role
				}
				set_variable = {
					name = duelist_role
					value = flag:yes
				}
				set_variable = {
					name = opponent_multi_duel_leader
					value = scope:opponent_multi_duel_leader
				}
				set_variable = {
					name = opponent_multi_side_leader
					value = scope:opponent_multi_duel_leader
				}
				set_variable = {
					name = count_of_team_members
					value = scope:multi_duel_leader_story_owner.var:count_of_team_members
				}
				if = {
					limit = {
						exists = var:duelist_side
						var:duelist_side = flag:side_b
					}
					add_character_flag = duel_starting_soon
				}
				else = {
					add_character_flag = {
						flag = duel_starting_soon
						days = 5
					}
				}
			}
			#Opponent Leader
			scope:opponent_multi_duel_leader = {
				remove_variable = opponent_multi_duel_leader
				set_variable = {
					name = opponent_multi_duel_leader
					value = scope:duelist_heir
				}
				set_variable = {
					name = opponent_multi_side_leader
					value = scope:duelist_heir
				}

				if = {
					limit = {
						any_owned_story = {
							story_type = multi_duel_leader_story
							any_in_list = {
								variable = side_a_duelists
								is_alive = yes
							}
						}
					}
					random_owned_story = {
						limit = { story_type = multi_duel_leader_story }
						every_in_list = {
							variable = side_a_duelists
							limit = { is_alive = yes }
							remove_variable = opponent_multi_side_leader

							set_variable = {
								name = opponent_multi_side_leader
								value = scope:duelist_heir
							}
						}
					}
				}
				else_if = {
					limit = {
						any_owned_story = {
							story_type = multi_duel_leader_story
							any_in_list = {
								variable = side_b_duelists
								is_alive = yes
							}
						}
					}
					random_owned_story = {
						limit = { story_type = multi_duel_leader_story }
						every_in_list = {
							variable = side_b_duelists
							limit = { is_alive = yes }
							remove_variable = opponent_multi_side_leader

							set_variable = {
								name = opponent_multi_side_leader
								value = scope:duelist_heir
							}
						}
					}
				}
			}
			#Current Leader
			scope:multi_duel_leader_story_owner = {
				remove_variable = duelist_heir
				remove_variable = opponent_multi_duel_leader
				remove_character_flag = duel_starting_soon
				remove_character_flag = has_duelist_heir
				remove_variable = duelist_role
				remove_variable = count_of_team_members
				set_variable = {
					name = duelist_role
					value = flag:no
				}

				if = {
					limit = {
						any_owned_story = {
							story_type = multi_duel_leader_story
							any_in_list = {
								variable = side_a_duelists
								is_alive = yes
							}
						}
					}
					random_owned_story = {
						limit = { story_type = multi_duel_leader_story }
						every_in_list = {
							variable = side_a_duelists
							limit = { is_alive = yes }
							remove_variable = my_multi_side_leader

							scope:duelist_heir = {
								add_to_variable_list = {
									name = side_a_duelists_temp
									target = prev
								}
							}

							set_variable = {
								name = my_multi_side_leader
								value = scope:duelist_heir
							}
						}
						make_story_owner = scope:duelist_heir
					}
				}
				else_if = {
					limit = {
						any_owned_story = {
							story_type = multi_duel_leader_story
							any_in_list = {
								variable = side_b_duelists
								is_alive = yes
							}
						}
					}
					random_owned_story = {
						limit = { story_type = multi_duel_leader_story }
						every_in_list = {
							variable = side_b_duelists
							limit = { is_alive = yes }
							remove_variable = my_multi_side_leader

							scope:duelist_heir = {
								add_to_variable_list = {
									name = side_b_duelists_temp
									target = prev
								}
							}

							set_variable = {
								name = my_multi_side_leader
								value = scope:duelist_heir
							}
						}
						make_story_owner = scope:duelist_heir
					}
				}
			}
			#Next Leader Again
			scope:duelist_heir = {
				if = {
					limit = {
						any_owned_story = {
							story_type = multi_duel_leader_story
						}

						has_variable_list = side_a_duelists_temp
					}

						limit = { story_type = multi_duel_leader_story }
						save_scope_as = new_leader_story
						scope:duelist_heir = {
							every_in_list = {
								variable = side_a_duelists_temp
								limit = { is_alive = yes }
								scope:new_leader_story = {
									add_to_variable_list = {
										name = side_a_duelists
										target = prev
									}
								}
							}
							clear_variable_list = side_a_duelists_temp
						}
					}
				}
				else_if = {
					limit = {
						any_owned_story = {
							story_type = multi_duel_leader_story
						}
						exists = var:duelist_side
						var:duelist_side = flag:side_b
						has_variable_list = side_b_duelists_temp
					}

preparation_for_multi_duel_notification = {
	location ?= { save_scope_as = background_terrain_scope }

	scope:duelist_in_side_a = {
		if = {
			limit = { has_character_flag = in_multi_duel }
			remove_character_flag = in_multi_duel
		}
	}
	scope:duelist_in_side_b = {
		if = {
			limit = { has_character_flag = in_multi_duel }
			remove_character_flag = in_multi_duel
		}
	}

	if = {
		limit = {
			scope:duelist_in_side_a = {
				NOT = { agot_multi_duelist_cant_continue = yes }
			}
			scope:duelist_in_side_b = {
				agot_multi_duelist_cant_continue = yes
			}
		}
		scope:duelist_in_side_a = { save_scope_as = winner_duelist }
		scope:duelist_in_side_b = { save_scope_as = loser_duelist }
	}
	else_if = {
		limit = {
			scope:duelist_in_side_a = {
				agot_multi_duelist_cant_continue = yes
			}
			scope:duelist_in_side_b = {
				NOT = { agot_multi_duelist_cant_continue = yes }
			}
		}
		scope:duelist_in_side_a = { save_scope_as = loser_duelist }
		scope:duelist_in_side_b = { save_scope_as = winner_duelist }
	}
	else = {
		random_list = {
			10 = {
				scope:duelist_in_side_a = { save_scope_as = winner_duelist }
				scope:duelist_in_side_b = { save_scope_as = loser_duelist }
			}
			10 = {
				scope:duelist_in_side_a = { save_scope_as = loser_duelist }
				scope:duelist_in_side_b = { save_scope_as = winner_duelist }
			}
		}
	}
	save_scope_as = this_multi_duel_participant
	random_flavor_event_effect = yes
}

post_multi_duel_handler = {
	scope:side_a_leader = {
		if = {
			limit = {
				NOT = { agot_multi_duelist_cant_continue = yes }
			}
			add_character_flag = {
				flag = duel_starting_soon
				days = 5
			}
		}
	}
	scope:side_a_leader = {
		if = {
			limit = {
				has_character_flag = check_for_dead_leader
				scope:duelist_in_side_b = scope:side_b_leader
				scope:side_b_leader = {
					agot_multi_duelist_cant_continue = yes
				}
			}
			add_character_flag = {
				flag = dead_every_note
				days = 3
			}
			trigger_event = {
				id = agot_multi_duel.0100
				days = 1
			}
		}
	}
	scope:side_b_leader = {
		if = {
			limit = {
				has_character_flag = check_for_dead_leader
				scope:duelist_in_side_a = scope:side_a_leader
				scope:side_a_leader = {
					agot_multi_duelist_cant_continue = yes
				}
			}
			add_character_flag = {
				flag = dead_every_note
				days = 3
			}
			trigger_event = {
				id = agot_multi_duel.0100
				days = 1
			}
		}
	}
	if = {
		limit = {
			scope:duelist_in_side_b = scope:side_b_leader
			scope:side_b_leader = {
				agot_multi_duelist_cant_continue = yes
			}
		}
		scope:side_a_leader = {
			if = {
				limit = { NOT = { has_character_flag = dead_every_note } }
				trigger_event = {
					id = agot_multi_duel.0100
					days = 1
				}
			}
		}
		if = {
			limit = { scope:side_b_leader = scope:side_a_leader.var:opponent_multi_duel_leader }
			send_duel_story_events_effect = yes
		}
	}
	else_if = {
		limit = {
			scope:duelist_in_side_a = scope:side_a_leader
			scope:side_a_leader = {
				agot_multi_duelist_cant_continue = yes
			}
		}
		scope:side_b_leader = {
			if = {
				limit = { NOT = { has_character_flag = dead_every_note } }
				trigger_event = {
					id = agot_multi_duel.0100
					days = 1
				}
			}
		}
		if = {
			limit = { scope:side_a_leader = scope:side_b_leader.var:opponent_multi_duel_leader }
			send_duel_story_events_effect = yes
		}
	}
	else = {
		send_duel_story_events_effect = yes
	}
}

agot_trial_of_seven_outcome = {
	if = {
		limit = {
			exists = var:og_side_a_leader
			exists = var:og_side_b_leader
			exists = var:match_rule
			exists = var:duelist_side
			var:match_rule = flag:dead_leader
			exists = var:tos_accuser
			exists = var:tos_accused
			exists = var:unqiue_story_name
			var:unqiue_story_name = flag:tos_story
		}
		if = {
			limit = { var:duelist_side = flag:side_a }
			var:og_side_a_leader = { save_scope_as = main_winner_leader }
			var:og_side_b_leader = { save_scope_as = main_loser_leader }
			var:og_side_a_leader = { save_scope_as = left_side_character }
		}
		else_if = {
			limit = { var:duelist_side = flag:side_b }
			var:og_side_b_leader = { save_scope_as = main_winner_leader }
			var:og_side_a_leader = { save_scope_as = main_loser_leader }
			var:og_side_b_leader = { save_scope_as = left_side_character }
		}
		#ToS Outcome
		if = {
			limit = {
				exists = scope:main_winner_leader
				exists = scope:main_loser_leader
				exists = scope:left_side_character
			}
			if = {
				limit = {
					OR = {
						scope:main_loser_leader = var:tos_accuser
						trigger_if = {
							limit = {
								exists = var:story_variant
								var:story_variant = flag:usurpation
							}
							scope:main_loser_leader = var:og_side_b_leader
						}
						trigger_if = {
							limit = {
								exists = var:story_variant
								var:story_variant = flag:dishonorable_imprisonment
							}
							scope:main_loser_leader = var:og_side_a_leader
						}
					}
				}
				if = {
					limit = { scope:main_loser_leader = { has_character_flag = agot_multi_duel_downed } }
					scope:main_loser_leader = { save_scope_as = loser_accuser_downed }
				}
				else_if = {
					limit = { scope:main_loser_leader = { has_character_flag = agot_multi_duel_yielded } }
					scope:main_loser_leader = { save_scope_as = loser_accuser_yielded }
				}
				else = { scope:main_loser_leader = { save_scope_as = loser_accuser_dead } }
			}
			else_if = {
				limit = {
					OR = {
						scope:main_loser_leader = var:tos_accused
						trigger_if = {
							limit = {
								exists = var:story_variant
								var:story_variant = flag:usurpation
							}
							scope:main_loser_leader = var:og_side_a_leader
						}
						trigger_if = {
							limit = {
								exists = var:story_variant
								var:story_variant = flag:dishonorable_imprisonment
							}
							scope:main_loser_leader = var:og_side_b_leader
						}
					}
				}
				if = {
					limit = { scope:main_loser_leader = { has_character_flag = agot_multi_duel_downed } }
					scope:main_loser_leader = { save_scope_as = loser_accused_downed }
				}
				else = { scope:main_loser_leader = { save_scope_as = loser_accused_dead } }
			}
			#ToS - SUCCESSION CRISIS Outcome
			if = {
				limit = {
					exists = var:story_variant
					var:story_variant = flag:usurpation
					OR = {
						scope:main_winner_leader = var:tos_accuser
						scope:main_winner_leader = var:og_side_b_leader
					}
				}
				#Helper accuser pushed ToS for the Legit Heir
				if = {
					limit = { exists = var:tos_true_heir }
					if = {
						limit = { var:tos_true_heir = { is_alive = yes } }
						save_scope_as = tos_story_usurpation_legit_heir_helper_won
					}
					else = { save_scope_as = tos_story_usurpation_legit_heir_helper_won_need_new_heir }
				}
				#Legit Heir was the accuser
				else = { save_scope_as = tos_story_usurpation_legit_heir_won }

				agot_save_tos_usurpation_outcome_variables = yes
			}
			else_if = {
				limit = {
					exists = var:story_variant
					var:story_variant = flag:usurpation
					OR = {
						scope:main_winner_leader = var:tos_accused
						scope:main_winner_leader = var:og_side_a_leader
					}
				}
				save_scope_as = tos_story_usurpation_usurper_won

				agot_save_tos_usurpation_outcome_variables = yes
			}
			#ToS - DISHONORABLE IMPRISONMENT Outcome
			else_if = {
				limit = {
					exists = var:story_variant
					var:story_variant = flag:dishonorable_imprisonment
					OR = {
						scope:main_winner_leader = var:tos_accused
						scope:main_winner_leader = var:og_side_b_leader
					}
				}
				save_scope_as = tos_story_dishonorable_imprisonment_defender_won

				agot_save_tos_imprisonment_outcome_variables = yes
			}
			else_if = {
				limit = {
					exists = var:story_variant
					var:story_variant = flag:dishonorable_imprisonment
					OR = {
						scope:main_winner_leader = var:tos_accuser
						scope:main_winner_leader = var:og_side_a_leader
					}
				}
				save_scope_as = tos_story_dishonorable_imprisonment_accuser_won

				agot_save_tos_imprisonment_outcome_variables = yes
			}
		}
	}
}

tos_sudden_death_succession_crisis_effect = {
	$OG_USURPER$ = { save_scope_as = og_usurper }
	$CURRENT_USURPER$ = { save_scope_as = current_usurper }
	$THE_LEGIT_HEIR$ = { save_scope_as = the_legit_heir }
	$THE_LEGIT_COMBATANT$ = { save_scope_as = the_legit_combatant }

	agot_save_tos_usurpation_outcome_variables = yes

	#LEGITIMATE SIDE WON
	if = {
		limit = {
			scope:og_usurper = { NOT = { is_alive = yes } }
		}
		if = {
			limit = {
				NOT = { scope:current_usurper = scope:the_legit_heir }
				scope:the_legit_heir = { is_alive = yes }
			}
			#Titles and Artifacts
			scope:current_usurper = {
				create_title_and_vassal_change = {
					type = conquest_claim
					save_scope_as = change
					add_claim_on_loss = yes
				}
				every_held_title = {
					limit = {
						tier > tier_barony
						save_temporary_scope_as = t_title
						scope:claim_keeper = { is_target_in_variable_list = { name = final_usurpation_titles target = scope:t_title } }
					}
					change_title_holder_include_vassals = {
						holder = scope:the_legit_heir
						change = scope:change
						take_baronies = yes
					}
				}
				resolve_title_and_vassal_change = scope:change

				agot_new_house_head_effect = { NEW_HOUSE_HEAD = scope:the_legit_heir }
				save_scope_as = wos_usurper
				scope:the_legit_heir = { save_scope_as = wos_king }
				usurper_gives_back_artifacts = yes
				unsave_usurper_artifact_after_war = yes
				if = {
					limit = { agot_is_independent_ruler = yes }
					create_title_and_vassal_change = {
						type = swear_fealty
						save_scope_as = change
						add_claim_on_loss = no
					}
					change_liege = {
						liege = scope:the_legit_heir
						change = scope:change
					}
					resolve_title_and_vassal_change = scope:change
				}
			}
		}
		if = {
			limit = { scope:the_legit_heir = { is_alive = yes } }
			scope:the_legit_heir = {
				add_prestige = 500
				add_legitimacy = 500
			}
		}

		#Memories
		if = {
			limit = {
				scope:the_legit_combatant = { is_alive = yes }
			}
			scope:the_legit_combatant = {
				create_character_memory = {
					type = agot_trial_of_seven_winner_leader
					participants = {
						opponent = scope:og_usurper
					}
				}
			}
		}

		#Bonding
		if = {
			limit = { scope:the_legit_combatant = { is_alive = yes } }
			scope:the_legit_combatant = {
				add_character_flag = { flag = we_won_the_tos days = 3 }
			}
			scope:the_legit_combatant = {
				add_character_flag = {
					flag = tos_bonding_effect_received
					days = 40
				}
				agot_tos_bonding_effect = yes
			}
		}
	}
	#USURPER SIDE WON
	else_if = {
		limit = {
			scope:the_legit_combatant = { NOT = { is_alive = yes } }
		}
		if = {
			limit = {
				NOT = { scope:current_usurper = scope:the_legit_heir }
			}
			if = {
				limit = {
					exists = scope:the_legit_combatant
					scope:the_legit_combatant = { NOT = { is_imprisoned = yes } }
				}
				scope:og_usurper = {
					hard_imprison_character_effect = {
						TARGET = scope:the_legit_combatant
						IMPRISONER = scope:og_usurper
					}
				}
			}
			add_prestige = 500
			add_legitimacy = 500
		}

		#Memories
		if = {
			limit = {
				scope:og_usurper = { is_alive = yes }
			}
			scope:og_usurper = {
				create_character_memory = {
					type = agot_trial_of_seven_winner_leader
					participants = {
						opponent = scope:the_legit_combatant
					}
				}
			}
		}

		#Bonding
		if = {
			limit = { scope:og_usurper = { is_alive = yes } }
			scope:og_usurper = {
				add_character_flag = { flag = we_won_the_tos days = 3 }
			}
			scope:og_usurper = { agot_tos_bonding_effect = yes }
		}
	}
}

agot_remove_from_activity_and_travel_effect = {
	save_scope_as = activity_owner
	if = {
		limit = {
			exists = involved_activity
			involved_activity = { NOT = { has_activity_type = activity_wedding } }
			involved_activity.activity_host = scope:activity_owner
		}
		invalidate_activity = scope:activity_owner.involved_activity
	}
	else_if = {
		limit = {
			exists = involved_activity
			involved_activity = { NOT = { has_activity_type = activity_wedding } }
		}
		remove_from_activity = involved_activity
	}

	current_travel_plan ?= {
		if = {
			limit = {
				travel_plan_owner = scope:activity_owner
			}
			abort_travel_plan = yes
		}
		else = {
			remove_character = scope:activity_owner
		}
	}
}
