agot_kingsguard_get_page_effect = {
	hidden_effect = {
		$KINGSGUARD$ = { save_scope_as = getting_page_for }
		random_in_global_list = {
			variable = kingsguard_white_book_pages
			limit = {
				has_variable = kingsguard_member
				var:kingsguard_member ?= scope:getting_page_for
			}
			save_scope_as = white_book_page
		}
	}
}

agot_kingsguard_create_historical_white_book_page = {
	hidden_effect = {
		$KINGSGUARD$ = { save_scope_as = selected_kingsguard }
		create_dynamic_title = {
			tier = duchy
			name = agot_kingsguard_white_book_page
		}
		scope:new_title ?= {
			set_capital_county = title:c_ruins
			set_landless_title = yes
			set_delete_on_destroy = no
			set_coa = d_kingsguard
			save_scope_as = white_book_page
			if = {
				limit = { NOT = { has_variable = kingsguard_member } }
				set_variable = {
					name = kingsguard_member
					value = scope:selected_kingsguard
				}
				add_to_global_variable_list = {
					name = kingsguard_white_book_pages
					target = scope:white_book_page
				}
			}
		}
		scope:selected_kingsguard = {
			if = {
				limit = {
					is_alive = no
				}
				make_unprunable = yes
			}
		}
	}
}

agot_kingsguard_create_dynamic_white_book_page_join = {
	hidden_effect = {
		$KINGSGUARD$ = { save_scope_as = selected_kingsguard }
		$KING$ = { save_scope_as = king }
		add_to_global_variable_list = {
			name = kingsguard_white_book
			target = scope:selected_kingsguard
		}
		create_dynamic_title = {
			tier = duchy
			name = agot_kingsguard_white_book_page_join_paragraph
		}
		scope:new_title ?= {
			set_capital_county = title:c_ruins
			set_landless_title = yes
			set_delete_on_destroy = no
			set_coa = d_kingsguard
			save_scope_as = white_book_page
			if = {
				limit = { NOT = { has_variable = kingsguard_member } }
				set_variable = {
					name = kingsguard_member
					value = scope:selected_kingsguard
				}
				add_to_global_variable_list = {
					name = kingsguard_white_book_pages
					target = scope:white_book_page
				}
			}
		}
		
		if = {
			limit = { exists = $KINGSGUARD$.var:my_personal_coa }
			agot_kingsguard_update_white_book_page_coa = { KINGSGUARD = $KINGSGUARD$ COAT_OF_ARMS = $KINGSGUARD$.var:my_personal_coa.house }
		}
	}
}
