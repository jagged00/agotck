# These are a select few that join upon Daemon's Landing, the other's a short while later.
agot_fourth_blackfyre_loyalists_alliance_effect = {
	character:Blackfyre_11 ?= {
		save_scope_as = invader
		random_owned_story = {
			limit = {
				story_type = story_agot_fourth_blackfyre_rebellion
			}
			save_scope_as = fourth_blackfyre_rebellion_story
		}
	}

	# Only Yronwoods join on Historical Outcomes
	if = {
		limit = { has_game_rule = agot_story_historical_events_historical_outcomes }
		dynasty:dynn_Yronwood.dynasty_founder.house.house_head ?= {
			add_to_list = potential_allies
		}
	}
	# lim approaches weighted/random
	else_if = {
		limit = { agot_has_historical_events_trigger = yes }
		dynasty:dynn_Massey.dynasty_founder.house.house_head ?= {
			add_to_list = potential_allies
		}
		dynasty:dynn_Blackmont.dynasty_founder.house.house_head ?= {
			add_to_list = potential_allies
		}
		dynasty:dynn_Wendwater.dynasty_founder.house.house_head ?= {
			add_to_list = potential_allies
		}
		dynasty:dynn_Bar_Emmon.dynasty_founder.house.house_head ?= {
			add_to_list = potential_allies
		}
		dynasty:dynn_Fell.dynasty_founder.house.house_head ?= {
			add_to_list = potential_allies
		}
		# 50/50 of iterating through loyalists
		random = {
			chance = 50
			# If the 50/50 rolls in favour of adding loyalists, iterate & give each a 80% chance of leaving Daemons side.
			every_in_list = {
				list = potential_allies
				random = {
					chance = 80
					remove_from_list = potential_allies
				}
			}
		}
		# Weighted Outcome gamerule, add 50/50
		if = {
			limit = { has_game_rule = agot_story_historical_events_weighted_outcomes }
			# extra separate 50/50 for getting Yronwoods to join Daemon guaranteed
			random = {
				chance = 50
				dynasty:dynn_Yronwood.dynasty_founder.house.house_head ?= {
					add_to_list = potential_allies
				}
			}
		}
	}
	every_in_list = {
		list = potential_allies
		limit = {
			is_alive = yes
			is_landed = yes
			is_ai = yes
			NOT = {
				invader_ally_builder_trigger = {
					INVADER = scope:invader
					TITLE_HOLDER = title:h_the_iron_throne.holder
				}
			}
		}
		save_temporary_scope_as = ally
		scope:fourth_blackfyre_rebellion_story = {
			add_to_variable_list = {
				name = blackfyre_loyalist_story_content_ally
				target = scope:ally
			}
		}
		scope:invader = {
			random_character_war = {
				limit = {
					using_cb = agot_blackfyre_claim
					NOT = { scope:ally = { is_participant_in_war = prev } }
				}
				add_attacker = scope:ally
			}
		}
	}
	# Assign MW loyalty variable:
	scope:fourth_blackfyre_rebellion_story = {
		every_in_list = {
			variable = blackfyre_loyalist_story_content_ally
			# Set MW Loyal stance
			set_variable = { name = mw_is_loyal_to value = scope:daemon }
		}
	}
}

# fired from: agot_on_game_start_historical_scenario_setup
agot_8236_on_game_start_setup = {
	#Fourth Blackfyre Rebellion Story Cycle
	agot_fourth_blackfyre_rebellion_scopes_effect = yes

	# If no DLC, land Daemon
	if = {
		limit = {
			character:Blackfyre_11 ?= { is_alive = yes }
		}
		if = {
			limit = { NOT = { has_dlc_feature = roads_to_power } }
			# If no DLC, land Daemon
			character:Blackfyre_11 = {
				if = {
					limit = {
						NOT = { has_trait = adventurer }
					}
					add_trait = adventurer
				}
				# Grant Daemon the d_muricosh title
				create_title_and_vassal_change = {
					type = granted
					save_scope_as = title_change
					add_claim_on_loss = no
				}
				title:d_muricosh.holder = {
					every_held_title = {
						limit = { tier >= tier_county }
						add_to_list = granted_titles
					}
				}
				every_in_list = {
					list = granted_titles
					change_title_holder_include_vassals = {
						holder = scope:daemon
						change = scope:title_change
					}
				}
				resolve_title_and_vassal_change = scope:title_change
				# Grant Daemon the e_blackfyre_rebellion title
				create_title_and_vassal_change = {
					type = granted
					save_scope_as = change
					add_claim_on_loss = no
				}
				title:e_blackfyre_rebellion = {
					change_title_holder = {
						holder = scope:daemon
						change = scope:change
					}
				}
				resolve_title_and_vassal_change = scope:change
				# Ensure his independence
				if = {
					limit = {
						is_independent_ruler = no
					}
					create_title_and_vassal_change = {
						type = independency
						save_scope_as = indep_change
						add_claim_on_loss = no
					}
					change_liege_or_become_independent = {
						CHANGE = scope:indep_change
						VASSAL = scope:daemon
					}
					resolve_title_and_vassal_change = scope:indep_change
				}
				set_location_to_default = yes
				change_government = feudal_government
				# Set coronation laws
				if = {
					limit = {
						agot_uses_agot_coronations_trigger = no
					}
					create_character_memory = {
						type = was_coronated
						participants = {
							officiant = character:Targaryen_95
						}
					}
					character:Targaryen_95 = {
						create_character_memory = {
							type = performed_coronation
							participants = {
								coronation_target = character:Blackfyre_11
							}
						}
					}
					add_realm_law_skip_effects = crowned_emperor
					set_variable = crowned_emperor_var
				}
			}
		}
		else = {
			# To prevent crashes from Blackfyre host title conflicting with landless, reset landless gov
			character:Blackfyre_11 = {
				title:d_dummy_blackfyre = { copy_title_history = title:e_blackfyre_rebellion }
				destroy_title = title:e_blackfyre_rebellion
				title:e_blackfyre_rebellion = { copy_title_history = title:d_dummy_blackfyre }
				change_government = landless_adventurer_government
				if = {
					limit = {
						NOT = { has_realm_law = landless_adventurer_succession_law }
					}
					add_realm_law = landless_adventurer_succession_law
				}
				# Style Daemon as "King" because he's coronated in this time
				add_character_flag = agot_claimant_king
				if = {
					limit = {
						NOT = { has_realm_law = camp_purpose_legitimists }
					}
					add_realm_law_skip_effects = camp_purpose_legitimists
				}
				set_location_to_default = yes
				domicile = { change_provisions = starting_provisions_value }
			}
		}
	}
}
