agot_generate_one_random_wilderness_county = {

		random_in_de_jure_hierarchy = {
			limit = {
				tier = tier_county
				holder = { has_character_flag = enhanced_freefolk_flag }
			}
			save_scope_as = siren_distribution_title
			title:c_wilderness.holder = {
				get_title = scope:siren_distribution_title
			}
		}
	}
}

agot_generate_duchy_tribe = {
	$DUCHY$ = { save_scope_as = wilderness_duchy }
	scope:wilderness_duchy = {
		random_in_de_jure_hierarchy = {
			limit = {
				tier = tier_county
				holder = { has_character_flag = enhanced_freefolk_flag }
			}
			save_scope_as = siren_distribution_title
			create_character = {
				location = character:Local_Rulers.capital_province
				random_traits = yes
				culture = scope:siren_distribution_title.title_province.culture
				faith = scope:siren_distribution_title.title_province.faith
				age = { 16 55 }
				gender = male
				dynasty = generate
				after_creation = {
					save_scope_as = tribal_chief
					get_title = scope:siren_distribution_title
					add_gold = { minor_gold_value_check medium_gold_value_check }
					add_prestige = { minor_prestige_gain medium_prestige_gain }
					add_piety = { minor_piety_gain medium_piety_gain }
				}
			}
			scope:tribal_chief = {
				get_title = scope:wilderness_duchy
			}
		}
		every_in_de_jure_hierarchy = {
			limit = {
				tier = tier_county
				holder = { has_character_flag = enhanced_freefolk_flag }
			}
			save_scope_as = siren_distribution_title
			create_character = {
				location = character:Local_Rulers.capital_province
				random_traits = yes
				culture = scope:siren_distribution_title.title_province.culture
				faith = scope:siren_distribution_title.title_province.faith
				age = { 16 55 }
				gender = male
				dynasty = generate
				after_creation = {
					get_title = scope:siren_distribution_title
					add_gold = { minor_gold_value_check medium_gold_value_check }
					add_prestige = { minor_prestige_gain medium_prestige_gain }
					add_piety = { minor_piety_gain medium_piety_gain }
					create_title_and_vassal_change = {
						type = swear_fealty
						save_scope_as = change
						add_claim_on_loss = no
					}
					change_liege = {
						liege = scope:tribal_chief
						change = scope:change
					}
					resolve_title_and_vassal_change = scope:change
				}
			}
		}
	}
}

agot_setup_beyond_the_wall = {
	title:e_beyond_the_wall = {
		every_in_de_jure_hierarchy = {
			limit = {
				tier = tier_duchy
				NOT = { exists = holder }
			}
			save_scope_as = siren_distribution_duchy
			random_list = {
				2 = {
					agot_generate_remaining_counties_wilderness = { DUCHY = scope:siren_distribution_duchy }
				}
				4 = { #single tribe duchy
					agot_generate_duchy_tribe = { DUCHY = scope:siren_distribution_duchy }
				}
				4 = {
					agot_generate_four_county_tribe = { DUCHY = scope:siren_distribution_duchy }
					agot_generate_remaining_one_county_tribes = { DUCHY = scope:siren_distribution_duchy }

					agot_generate_remaining_counties_wilderness = { DUCHY = scope:siren_distribution_duchy }
				}

					agot_generate_three_county_tribe = { DUCHY = scope:siren_distribution_duchy }
					agot_generate_remaining_one_county_tribes = { DUCHY = scope:siren_distribution_duchy }

agot_update_wildling_resistance = {
	if = {
		limit = {
			holder = {
				government_has_flag = government_is_uninteractable
			}
		}
		if = {
			limit = { has_county_modifier = btw_wildlings_resistance }
			remove_county_modifier = btw_wildlings_resistance
		}
	}
	else_if = {
		limit = { #wildling county, not wildling holder
			culture = { agot_is_wildling_culture = yes }
			holder = {
				NOT = {
					culture = { agot_is_wildling_culture = yes }
				}
			}
			NOT = {
				has_county_modifier = btw_wildlings_resistance
			}
		}
		add_county_modifier = btw_wildlings_resistance
	}
	else_if = {
		limit = { #wildling county, wildling holder, no resistance
			culture = { agot_is_wildling_culture = yes }
			holder = {
				culture = { agot_is_wildling_culture = yes }
			}
			has_county_modifier = btw_wildlings_resistance
		}
		remove_county_modifier = btw_wildlings_resistance
	}
	else_if = {
		limit = { #not wildling county, not wildling holder, no resistance
			NOT = {
				culture = { agot_is_wildling_culture = yes }
			}
			holder = {
				NOT = {
					culture = { agot_is_wildling_culture = yes }
				}
			}
			has_county_modifier = btw_wildlings_resistance
		}
		remove_county_modifier = btw_wildlings_resistance
	}
	else_if = {
		limit = { #not wildling county, wildling holder, no resistance
			culture = { agot_is_wildling_culture = yes }
			holder = {
				culture = { agot_is_wildling_culture = yes }
			}
			has_county_modifier = btw_wildlings_resistance
		}
		remove_county_modifier = btw_wildlings_resistance
	}
}

agot_remove_permafrost_modifiers = {
	if = {
		limit = {
			has_county_modifier = mpo_siberian_permafrost_modifier
		}
		remove_county_modifier = mpo_siberian_permafrost_modifier
	}
	else_if = {
		limit = {
			has_county_modifier = mpo_siberian_permafrost_modifier_bad
		}
		remove_county_modifier = mpo_siberian_permafrost_modifier_bad
	}
