agot_war_victory_effects = {
	scope:attacker = {
		if = {
			limit = {
				culture = {
					has_cultural_parameter = gain_piety_from_aggro_war_victory
				}
			}
			add_piety = {
				add = minor_piety_gain
				multiply = scope:defender.primary_title.tier
			}
		}
	}
}

blackfyre_war_declared = {
	if = {
		limit = {
			exists = global_var:number_of_blackfyre_rebellions
		}
		change_global_variable = {
			name = number_of_blackfyre_rebellions
			add = 1
		}
	}
}
