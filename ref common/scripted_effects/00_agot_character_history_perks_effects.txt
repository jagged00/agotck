agot_safe_refund_all_perks_history_effect = {
	if = {
		limit = {
			NOT = { has_character_flag = history_engine_perks_purged }
		}
		refund_all_perks = yes
		remove_trait = diplomat
		remove_trait = family_first
		remove_trait = august
		remove_trait = strategist
		remove_trait = overseer
		remove_trait = gallant
		remove_trait = architect
		remove_trait = administrator
		remove_trait = avaricious
		remove_trait = schemer
		remove_trait = seducer
		remove_trait = torturer
		remove_trait = whole_of_body
		remove_trait = scholar
		remove_trait = theologian
		remove_trait = lifestyle_surveyor
		remove_trait = lifestyle_wayfarer
		remove_trait = lifestyle_voyager
		add_character_flag = {
			flag = history_engine_perks_purged
			days = 1
		}
	}
}
agot_safe_spend_perk_point_history_effect = {
	if = {
		limit = { $LIFESTYLE$_perk_points >= 1 }
		add_$LIFESTYLE$_perk_points = -1
	}
	else = {
		random_list = {
			1 = {
				trigger = { diplomacy_lifestyle_perk_points >= 1 }
				add_diplomacy_lifestyle_perk_points = -1
			}
			1 = {
				trigger = { martial_lifestyle_perk_points >= 1 }
				add_martial_lifestyle_perk_points = -1
			}
			1 = {
				trigger = { stewardship_lifestyle_perk_points >= 1 }
				add_stewardship_lifestyle_perk_points = -1
			}
			1 = {
				trigger = { intrigue_lifestyle_perk_points >= 1 }
				add_intrigue_lifestyle_perk_points = -1
			}
			1 = {
				trigger = { learning_lifestyle_perk_points >= 1 }
				add_learning_lifestyle_perk_points = -1
			}
			1 = {
				trigger = { wanderer_lifestyle_perk_points >= 1 }
				add_wanderer_lifestyle_perk_points = -1
			}
		}
	}
}
agot_safe_buy_perk_history_effect = { # Exported for history clarity
	agot_safe_refund_all_perks_history_effect = yes
	agot_safe_buy_$PERK$_history_effect = yes
}
