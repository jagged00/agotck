# Ignored order/advice opinion
agot_dragon_siege_negative_opinion_value = { # how does commander feel about his order/advice being ignored?
	value = -10 # base value

	# These traits incur opinion loss
	if = {
		limit = { scope:commander = { has_trait = arrogant } }
		add = -7
	}
	if = {
		limit = { scope:commander = { has_trait = authoritative } }
		add = -7
	}
	if = {
		limit = { scope:commander = { has_trait = bossy } }
		add = -4
	}
	if = {
		limit = { scope:commander = { has_trait = wrathful } }
		add = -7
	}
	if = {
		limit = { scope:commander = { has_trait = stubborn } }
		add = -4
	}

	multiply = {
		value = 1

		if = { # If you are bound --> extra loss
			limit = { scope:demand_dragon_use ?= flag:bound }
			add = 0.5
		}

		# Relations
		# Negative relations with commander --> increased opinion loss
		if = {
			limit = { scope:commander = { any_relation = { type = friend this = root } } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = best_friend this = root } } }
			subtract = 0.7
		}
		if = {
			limit = { scope:commander = { any_relation = { type = lover this = root } } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = soulmate this = root } } }
			subtract = 1
		}

		# Positive relations with commander --> decreased opinion loss
		if = {
			limit = { scope:commander = { any_relation = { type = victim this = root } } }
			add = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = grudge this = root } } }
			add = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = rival this = root } } }
			add = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = nemesis this = root } } }
			add = 1
		}

		# Traits
		# Commander is not too serious about his command/advice being ignored --> reduced opinion loss
		if = {
			limit = { scope:commander = { has_trait = humble } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { has_trait = calm } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { has_trait = content } }
			subtract = 0.3
		}
		if = {
			limit = { scope:commander = { has_trait = shy } }
			subtract = 0.3
		}
		if = {
			limit = { scope:commander = { has_trait = forgiving } }
			subtract = 0.3
		}
	}

	max = -5 # We always want to remove at least 5; must be at end of value!!!!!
}

# Followed order/advice opinion
agot_dragon_siege_positive_opinion_value = {
	value = 10 # base value

	# These traits incur opinion gain
	if = {
		limit = { scope:commander = { has_trait = humble } }
		add = 7
	}
	if = {
		limit = { scope:commander = { has_trait = compassionate } }
		add = 4
	}

	multiply = {
		value = 1

		if = { # If you are bound, you do not improve opinion by following commands as much since rider is "COMMANDED" --> only half the positive gain
			limit = { scope:demand_dragon_use ?= flag:bound }
			subtract = 0.6
		}

		# Relations
		# Positive relations with commander --> increased opinion gain
		if = {
			limit = { scope:commander = { any_relation = { type = friend this = root } } }
			add = 0.3
		}
		if = {
			limit = { scope:commander = { any_relation = { type = best_friend this = root } } }
			add = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = lover this = root } } }
			add = 0.3
		}
		if = {
			limit = { scope:commander = { any_relation = { type = soulmate this = root } } }
			add = 0.7
		}

		# Negative relations with commander --> reduced opinion gain
		if = {
			limit = { scope:commander = { any_relation = { type = victim this = root } } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = grudge this = root } } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = rival this = root } } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { any_relation = { type = nemesis this = root } } }
			subtract = 1
		}

		# Traits
		# Commander expects you to follow his order/advice --> reduced opinion gain
		if = {
			limit = { scope:commander = { has_trait = callous } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { has_trait = arrogant } }
			subtract = 0.8
		}
		if = {
			limit = { scope:commander = { has_trait = authoritative } }
			subtract = 0.5
		}
		if = {
			limit = { scope:commander = { has_trait = bossy } }
			subtract = 0.3
		}
	}

	min = 5 # We always want to give at least 5; must be at end of value!!!!!
}
