### Choose stance values - used in 00_agot_scripted_modifiers_mega_wars.txt

mw_base_loyalist_value = {
	value = 0

	if = { # Fetch base value by checking legitimate house.
		limit = {
			any_in_list = {
				list = target_titles
				NOT = { scope:mw_rebel_leader.house = scope:mw_crown.house } # If both sides are of the same house this value is kinda outta place, so don't use it
				OR = {
					var:legitimate_house ?= scope:mw_crown.house
					var:current_house ?= scope:mw_crown.house
				}
			}
		}

		add = {
			if = {
				limit = {
					scope:mw_crown = { save_temporary_scope_as = target_scope }
				}
				every_in_list = {
					list = target_titles
					add = mw_legitimate_house_base_value # Honour is considered in the value!
				}
			}
			multiply = { # How much do we want to reduce the impact: if rebel has relations to root, lower legitimate house impact.
				value = 1
				if = {
					limit = {
						scope:mw_ruler_scope.var:mw_is_loyal_to ?= scope:mw_rebel_leader
					}
					add = -0.1
				}
				if = {
					limit = {
						agot_mw_has_alliance_with_rebel_leader_trigger = yes
					}
					add = -0.25
				}
				if = {
					limit = {
						scope:s_relation_t_rebel_leader = yes
					}
					add = -0.5
				}
			}
		}
	}
	else_if = { # In some instances honour needs to be applied to the rebel side as the rebel leader is considered legitimate.
		limit = {
			scope:mw_ruler_scope ?= {
				OR = {
					scope:mw_rebel_story.var:mw_war_cb ?= flag:succession_war
					#AND = {
					#	NOT = { scope:s_relation_t_crown = yes }
					#	scope:s_relation_t_rebel_leader = yes
					#}
					agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
				}
			}
		}
		add = -100
	}
	else = { add = 100 }

	# Consider honour
	if = { # In some instances honour needs to be applied to the rebel side as the rebel leader is considered legitimate.
		limit = {
			scope:mw_ruler_scope ?= {
				NOR = {
					scope:mw_rebel_story.var:mw_war_cb ?= flag:succession_war
					#AND = {
					#	NOT = { scope:s_relation_t_crown = yes }
					#	scope:s_relation_t_rebel_leader = yes
					#}
					agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
				}
			}
		}

		multiply = {
			add = 1

			if = { # base factor is 1; here we increase that factor by by positive honour
				limit = { scope:mw_ruler_scope.ai_honor > 0 }
				add = {
					add = scope:mw_ruler_scope.ai_honor
					multiply = 0.09
				}
			}
		}
	}
	else = { multiply = 0.5 }

	add = {
		add = scope:mw_crown.dread
		multiply = mw_dread_factor
	}
	add = {
		subtract = scope:mw_crown.tyranny
		multiply = mw_tyranny_factor
	}
	if = { # # If you already are in loyalist list, you are less likely to switch
		limit = { scope:mw_crown_story = { is_target_in_variable_list = { name = mw_loyalist_list target = scope:mw_ruler_scope } } }

		add = mw_stay_loyalist_value
	}

	# At the end to avoid factors
	if = { # if mw lite: there is no neutral stance to favour --> boost loyal stance
		limit = { scope:mw_crown_story.var:mw_mode ?= flag:mw_lite }
		add = 250
	}
}
mw_base_neutral_value = {
	value = 75 # increased weight on this one since it being 0 lead to rulers only staying neutral if they had a reason; they should also have a chance to stay neutral if they do not care enough

	if = { # If you have a historical loyalty to one of the sides, don't stay neutral
		limit = {
			OR = {
				AND = {
					agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_crown.house }
					NOT = { agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house } }
				}
				AND = {
					NOT = { agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_crown.house } }
					agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
				}
				agot_mw_is_in_another_rebel_list_trigger = { RULER = scope:mw_ruler_scope }
			}
		}

		add = -75
	}
	else_if = { # external wars should incentivise everyone to pick a side
		limit = { scope:mw_rebel_story.var:mw_source ?= flag:mw_external }

		add = -50
	}


	subtract = { # depending on your energy you want more or less to actually choose a side, "move yourself"
		add = scope:mw_ruler_scope.ai_energy
		multiply = 0.5
	}
	#add = scope:mw_ruler_scope.ai_greed # staying neutral might open new opportunities to gain something out of this # AGOT TODO enable this once can actually get something for your support
	subtract = { # joining rebel side is always more risky than staying loyal
		add = scope:mw_ruler_scope.ai_boldness
		#multiply = 0.5
	}
	add = {
		add = scope:mw_crown.tyranny
		multiply = 1.5
	}
	if = { # Fallback if no stance is valid --> if you can't decide what to do you stay neutral
		limit = {
			scope:mw_ruler_scope ?= {
				NOT = { has_character_flag = mw_fallback_checker }
				NOR = {
					has_character_flag = mw_loyalist
					has_character_flag = mw_neutral
					has_character_flag = mw_rebel
					has_character_flag = mw_independence
				}
			}
		}
		add = 10000
	}
}
mw_base_rebel_value = {
	value = 0

	if = { # Fetch base value by checking legitimate house.
		limit = {
			any_in_list = {
				list = target_titles
				NOT = { scope:mw_rebel_leader.house = scope:mw_crown.house } # If both sides are of the same house this value is kinda outta place, so don't use it
				OR = {
					var:legitimate_house ?= scope:mw_rebel_leader.house
					var:current_house ?= scope:mw_rebel_leader.house
				}
			}
		}

		add = {
			if = {
				limit = {
					scope:mw_rebel_leader = { save_temporary_scope_as = target_scope }
				}
				every_in_list = {
					list = target_titles
					add = mw_legitimate_house_base_value
				}
			}
			multiply = {
				value = 1
				if = {
					limit = {
						scope:mw_ruler_scope.var:mw_is_loyal_to ?= scope:mw_crown
					}
					add = -0.1
				}
				if = {
					limit = {
						agot_mw_has_alliance_with_crown_trigger = yes
					}
					add = -0.25
				}
				if = {
					limit = {
						scope:s_relation_t_crown = yes
					}
					add = -0.5
				}
			}
		}
	}
	else_if = { # There are some cases where we want a character's honour to apply for rebel instead of crown; e. g.: succession wars (rebel leader is legit heir)
		limit = {
			scope:mw_ruler_scope ?= {
				OR = {
					scope:mw_rebel_story.var:mw_war_cb ?= flag:succession_war
					#AND = {
					#	NOT = { scope:s_relation_t_crown = yes }
					#	scope:s_relation_t_rebel_leader = yes
					#}
					agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
				}
			}
		}
		add = 100
	}
	else_if = { # if rebel leader is claimant, give him bonus!
		limit = {
			any_in_list = {
				list = target_titles
				scope:mw_rebel_leader = { has_claim_on = prev }
			}
		}

		every_in_list = {
			list = target_titles
			limit = { scope:mw_rebel_leader = { has_claim_on = prev } }
			if = {
				limit = { tier = tier_empire }
				add = 50
			}
			else_if = {
				limit = { tier = tier_kingdom }
				add = 40
			}
			else_if = {
				limit = { tier = tier_duchy }
				add = 30
			}
			else = {
				add = 20
			}
		}
	}
	else = { # else there is no real reason to have a good base value
		add = -100
	}

	# Consider honour
	if = { # There are some cases where we want a character's honour to apply for rebel instead of crown; e. g.: succession wars (rebel leader is legit heir)
		limit = {
			scope:mw_ruler_scope ?= {
				OR = {
					scope:mw_rebel_story.var:mw_war_cb ?= flag:succession_war
					#scope:s_relation_t_rebel_leader = yes
					agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
				}
			}
		}

		multiply = {
			add = 1

			if = { # Base factor is 1; here we increase that factor by by positive honour
				limit = { scope:mw_ruler_scope.ai_honor > 0 }
				add = {
					add = scope:mw_ruler_scope.ai_honor
					multiply = 0.05
				}
			}
		}
	}

	add = {
		add = scope:mw_crown.tyranny
		multiply = mw_tyranny_factor
	}
}

mw_follow_lieges_choice_value = {
	value = 300 # base value
	multiply = {
		value = 1

		scope:mw_ruler_scope ?= {
			if = { # If you have no special relation to one of the sides, increase follow liege factor.
				limit = {
					NOR = {
						has_relation_friend = scope:mw_crown
						has_relation_lover = scope:mw_crown
						has_relation_friend = scope:mw_rebel_leader
						has_relation_lover = scope:mw_rebel_leader
						scope:s_relation_t_crown = yes
						scope:s_relation_t_rebel_leader = yes
						agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_crown.house }
						agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
					}
				}

				## Add base multipliers
				if = { # Additional bonus if liege is independence leader
					limit = {
						OR = {
							# War triggered by declaing independence war or faction demand
							AND = {
								scope:mw_rebel_story.var:mw_war_cb ?= flag:independence_war
								OR = {
									top_liege ?= scope:mw_rebel_leader
									liege.var:pre_war_liege ?= scope:mw_rebel_leader
								}
							}
							# War triggered by demand independence during mega war
							scope:mw_crown_story = { is_target_in_variable_list = { name = mw_independence_rebel_leader_list target = scope:mw_ruler_scope.liege } }
							scope:mw_ruler_scope = { any_liege_or_above = { is_target_in_variable_list = { name = mw_independence_rebels_list target = scope:mw_ruler_scope.liege } } }
						}
					}
					add = 1
				}
				if = { # Counts are more likely to follow their liege.
					limit = { highest_held_title_tier = tier_county }
					add = {
						add = 8
						if = {
							limit = { NOT = { scope:mw_crown_story.var:mw_mode ?= flag:mw_lite } }
							multiply = 0.5
						}
					}
				}
				else_if = { # Dukes are still inclined to follow liege, but not as much as counts
					limit = { highest_held_title_tier = tier_duchy }
					add = 2
				}
				else_if = { # Kings are still inclined to follow liege, but not as much as counts or dukes
					limit = { highest_held_title_tier = tier_kingdom }
					add = 1.5
				}

				# Character based multipliers
				if = { # If your liege's legitimacy level is lower than the expected level, you care less about his choice or position # AGOT TODO rework dis as it basically means that count joins loyalists just because his duke liege is a stupid asshole and has no legitimacy.
					limit = {
						liege = { save_temporary_scope_as = liege }
						default_ai_legitimacy_expectation != scope:liege.legitimacy_level
						default_ai_legitimacy_expectation != 0
					}
					add = {
						add = scope:liege.legitimacy_level
						divide = default_ai_legitimacy_expectation
					}
				}
				if = { # consider character's opinion of his liege
					limit = {
						save_temporary_opinion_value_as = { name = opinion_of_liege target = liege }
						NOR = { # only non side-lieges
							liege = scope:mw_crown
							liege = scope:mw_rebel_leader
						}
					}

					add = { # e.g. opinion +50, compassion -100 --> -50 --> -0.5
						add = scope:opinion_of_liege
						add = ai_compassion
						divide = 100
					}
				}
				if = { # consider leverage liege has over ruler based on diplomacy and stewardship skill of rulers and castellans and chancellors
					limit = {
						NOR = {
							liege = scope:mw_crown
							liege = scope:mw_rebel_leader
						}
					}

					add = {
						if = {
							limit = {
								mw_leverage_value < liege.mw_leverage_value
								mw_leverage_value != 0
							}
							add = liege.mw_leverage_value
							divide = mw_leverage_value
						}
						else_if = {
							limit = { liege.mw_leverage_value != 0 }
							subtract = mw_leverage_value
							divide = liege.mw_leverage_value
						}
						if = {
							limit = {
								highest_held_title_tier = tier_duchy
							}
							multiply = 1.6
						}
						else = { multiply = 0.5 }
					}
				}

				# consider level of dread
				if = { # ruler is intimidated by liege
					limit = {
						has_dread_level_towards = {
							target = scope:mw_ruler_scope.liege
							level = 1
						}
					}
					add = 3
				}
				else_if = { # ruler is cowed by liege
					limit = {
						has_dread_level_towards = {
							target = scope:mw_ruler_scope.liege
							level = 2
						}
					}
					add = 5
				}
			}
			else = { add = -0.75 } # Else, reduce incentive as you prefer joining your relation target.
		}
	}
}
mw_leverage_value = {
	if = {
		limit = { exists = diarch }
		add = diarch.diplomacy
		add = diarch.stewardship
	}
	else = {
		add = diplomacy
		add = stewardship
	}
	if = {
		limit = { NOT = { cp:councillor_chancellor ?= scope:mw_ruler_scope } }
		cp:councillor_chancellor ?= {
			add = {
				add = diplomacy
				multiply = 0.25
			}
		}
	}
	if = {
		limit = { NOT = { cp:councillor_castellan ?= scope:mw_ruler_scope } }
		cp:councillor_castellan ?= {
			add = {
				add = stewardship
				multiply = 0.25
			}
		}
	}
}

### Opinions and relations
mw_opinion_value_factor_crown = {
	value = 1
	if = {
		limit = {
			scope:mw_crown = { save_temporary_scope_as = target_scope }
		}
		multiply = mw_opinion_value_factor_base
	}
}
mw_opinion_value_factor_rebel_leader = {
	value = 1
	if = {
		limit = {
			scope:mw_rebel_leader = { save_temporary_scope_as = target_scope }
		}
		multiply = mw_opinion_value_factor_base
	}
}
mw_opinion_value_factor_base = {
	value = 2.5
	if = { # Ruler has positive opinion of target.
		limit = { opinion = { target = scope:target_scope value > 0 } }

		if = {
			limit = {
				scope:mw_ruler_scope = {
					OR = {
						has_relation_soldier_friend = scope:target_scope
						has_relation_friend = scope:target_scope
					}
				}
			}
			add = 2
		}
		if = {
			limit = { scope:mw_ruler_scope = { has_relation_best_friend = scope:target_scope } }
			add = 3
		}
		if = {
			limit = { scope:mw_ruler_scope = { has_relation_lover = scope:target_scope } }
			add = 5
		}
		if = {
			limit = { scope:mw_ruler_scope = { has_relation_soulmate = scope:target_scope } }
			add = 7
		}
	}
	else_if = { # Ruler has negative opinion of target.
		limit = { opinion = { target = scope:target_scope value < 0 } }

		if = {
			limit = { scope:mw_ruler_scope = { has_relation_rival = scope:target_scope } }
			add = 4
		}
		if = {
			limit = { scope:mw_ruler_scope = { has_relation_nemesis = scope:target_scope } }
			add = 7
		}
	}
	if = { # You care less about whether you like the leader, if he is not your direct liege.
		limit = { NOT = { scope:mw_rebel_story.var:mw_war_cb ?= flag:succession_war } } # only if we want opinion not matter that much.

		if = { # e.g.: you are a duke, your liege is a kingdom --> TARGET is emperor
			limit = {
				NOT = { scope:mw_ruler_scope.liege = scope:target_scope }
				OR = {
					scope:mw_ruler_scope.liege.liege = scope:target_scope
					scope:mw_ruler_scope.liege.var:pre_war_liege ?= scope:target_scope
				}
			}
			multiply = 0.5
		}
		else_if = { # e.g.: you are a count, your liege is duke, then kingdom --> TARGET is emperor
			limit = {
				NOT = {
					scope:mw_ruler_scope.liege = scope:target_scope
					OR = {
						scope:mw_ruler_scope.liege.liege = scope:target_scope
						scope:mw_ruler_scope.liege.var:pre_war_liege ?= scope:target_scope
					}
				}
				OR = {
					scope:mw_ruler_scope.liege.liege.liege = scope:target_scope
					scope:mw_ruler_scope.liege.liege.var:pre_war_liege ?= scope:target_scope # scope:target_scope.holder
					scope:mw_ruler_scope.liege.var:pre_war_liege.liege ?= scope:target_scope
					scope:mw_ruler_scope.liege.var:pre_war_liege.var:pre_war_liege ?= scope:target_scope
				}
			}
			multiply = 0.25
		}
	}
	else = { # In case of a succession war, sligthly increase the opinion impact.
		add = 1.5
	}
}
mw_close_family_value = {
	value = 0

	# From ruler scope's perspective: are there close family members that are related/married to target scope ir his family?
	scope:mw_ruler_scope = {
		every_close_family_member = {
			limit = {
				save_temporary_scope_as = check_scope
				agot_mw_family_relations_trigger = yes
			}

			save_temporary_scope_as = check_scope

			scope:target_scope ?= {
				if = { # target is close family of ruler
					limit = {
						this = scope:check_scope
						is_close_family_of = scope:mw_ruler_scope
					}
					add = 1000
				}
				else_if = { # target is extended family of ruler
					limit = {
						this = scope:check_scope
						is_close_or_extended_family_of = scope:mw_ruler_scope
					}
					add = 500
				}

				if = {
					limit = {
						scope:check_scope = {
							is_child_of = scope:target_scope
							is_close_family_of = scope:mw_ruler_scope
						}
					}
					add = 1200
				}
				else_if = {
					limit = {
						scope:check_scope = {
							is_child_of = scope:target_scope
							is_close_or_extended_family_of = scope:mw_ruler_scope
						}
					}
					add = 800
				}

				if = { # spouse of target is close family of ruler
					limit = {
						is_spouse_of = scope:check_scope
						scope:check_scope = { is_close_family_of = scope:mw_ruler_scope }
					}
					add = 1000
				}
				else_if = { # spouse of target is extended family of ruler
					limit = {
						is_spouse_of = scope:check_scope
						scope:check_scope = { is_close_or_extended_family_of = scope:mw_ruler_scope }
					}
					add = 800
				}

				if = { # close family of target is child of ruler
					limit = {
						is_close_family_of = scope:check_scope
						scope:check_scope = { is_child_of = scope:mw_ruler_scope }

						### Filter corss relations between crown and rebel
						trigger_if = {
							limit = { this ?= scope:mw_crown }
							NOT = { scope:check_scope = { is_child_of = scope:mw_rebel_leader } }
						}
						trigger_else_if = {
							limit = { this ?= scope:mw_rebel_leader }
							NOT = { scope:check_scope = { is_child_of = scope:mw_crown } }
						}
						trigger_else = { always = yes }
					}
					add = 1000
				}
				else_if = { # extended family of target is child of ruler
					limit = {
						is_close_or_extended_family_of = scope:check_scope
						scope:check_scope = { is_child_of = scope:mw_ruler_scope }

						### Filter corss relations between crown and rebel
						trigger_if = {
							limit = { this ?= scope:mw_crown }
							NOT = { scope:check_scope = { is_child_of = scope:mw_rebel_leader } }
						}
						trigger_else_if = {
							limit = { this ?= scope:mw_rebel_leader }
							NOT = { scope:check_scope = { is_child_of = scope:mw_crown } }
						}
						trigger_else = { always = yes }
					}
					add = 600
				}

				if = { # close family of target is spouse of ruler
					limit = {
						is_close_family_of = scope:check_scope
						scope:check_scope = { is_spouse_of = scope:mw_ruler_scope }
					}
					add = 800
				}
				else_if = { # extended family of target is spouse of ruler
					limit = {
						is_close_or_extended_family_of = scope:check_scope
						scope:check_scope = { is_spouse_of = scope:mw_ruler_scope }
					}
					add = 500
				}

				if = { # close family of target is close family of ruler
					limit = {
						is_close_family_of = scope:check_scope
						scope:check_scope = { is_close_family_of = scope:mw_ruler_scope }

						### Filter corss relations between crown and rebel
						trigger_if = {
							limit = { this ?= scope:mw_crown }
							NOT = { scope:check_scope = { is_close_family_of = scope:mw_rebel_leader } }
						}
						trigger_else_if = {
							limit = { this ?= scope:mw_rebel_leader }
							NOT = { scope:check_scope = { is_close_family_of = scope:mw_crown } }
						}
						trigger_else = { always = yes }
					}
					add = 800
				}
				else_if = { # extended family of target is close family of ruler
					limit = {
						is_close_or_extended_family_of = scope:check_scope
						scope:check_scope = { is_close_family_of = scope:mw_ruler_scope }

						### Filter corss relations between crown and rebel
						trigger_if = {
							limit = { this ?= scope:mw_crown }
							NOT = { scope:check_scope = { is_close_family_of = scope:mw_rebel_leader } }
						}
						trigger_else_if = {
							limit = { this ?= scope:mw_rebel_leader }
							NOT = { scope:check_scope = { is_close_family_of = scope:mw_crown } }
						}
						trigger_else = { always = yes }
					}
					add = 500
				}

				if = { # target's primary heir is close family of ruler
					limit = {
						primary_heir ?= scope:check_scope
						primary_heir ?= { is_close_family_of = scope:mw_ruler_scope }
					}
					add = 800
				}

				if = { # target's primary heir's spouse is close family of ruler
					limit = {
						primary_heir ?= {
							is_spouse_of = scope:check_scope
							any_spouse = { is_close_family_of = scope:mw_ruler_scope }
						}
					}
					add = 600
				}
			}
		}
	}

	# From target scope's perspective: are there cousin relations (which are excluded from above secction) or marriages to ruler scope
	scope:target_scope = {
		every_child = {
			limit = {
				OR = {
					is_close_or_extended_family_of = scope:mw_ruler_scope
					is_spouse_of = scope:mw_ruler_scope
				}
			}
			if = {
				limit = { is_close_family_of = scope:mw_ruler_scope }
				add = 1200
			}
			if = {
				limit = { is_close_or_extended_family_of = scope:mw_ruler_scope }
				add = 600
			}
			if = {
				limit = { is_spouse_of = scope:mw_ruler_scope }
				add = 400
			}
		}
	}
}
