# Standard colonization costs
@base_colonize_cost = 50
@count_colonize_cost = 25
@duke_colonize_cost = 50
@king_colonize_cost = 75
@emporer_colonize_cost = 100

base_colonize_cost = @base_colonize_cost
count_colonize_cost = @count_colonize_cost
duke_colonize_cost = @duke_colonize_cost
king_colonize_cost = @king_colonize_cost
emporer_colonize_cost = @emporer_colonize_cost

# Determines cost for colonization for provinces on mainland.
colonize_cost_val = {
	value = 0

	add = { # Adding base cost for count
		value = base_colonize_cost
		desc = "BASE_VALUE"
		format = "BASE_VALUE_FORMAT"
	}

	if = { # Adding cost for count
		limit = { highest_held_title_tier = tier_county }
		add = { # Adding base cost for count
			value = count_colonize_cost
			desc = "colonization_cost_count_tier"
			format = "BASE_VALUE_FORMAT"
		}
	}
	else_if = { # Adding cost for duke
		limit = { highest_held_title_tier = tier_duchy }
		add = {
			value = duke_colonize_cost
			desc = "colonization_cost_duke_tier"
			format = "BASE_VALUE_FORMAT"
		}
	}
	else_if = {# Adding cost for king
		limit = { highest_held_title_tier = tier_kingdom }
		add = {
			value = king_colonize_cost
			desc = "colonization_cost_king_tier"
			format = "BASE_VALUE_FORMAT"
		}
	}
	else = { # Adding base cost for emporer
		add = {
			value = emporer_colonize_cost
			desc = "colonization_cost_emperor_tier"
			format = "BASE_VALUE_FORMAT"
		}
	}
}

feudalize_tribal_holding_interaction_cost = {
	value = 500
	if = {
		limit = {
			trigger_if = {
				limit = { exists = scope:actor }
				scope:actor = {
					is_ai = yes
					capital_province = { has_holding_type = tribal_holding }
				}
			} trigger_else = {
				is_ai = yes
				capital_province = { has_holding_type = tribal_holding }
			}
		}
		add = -250
	}
	if = {
		limit = {
			scope:target ?= {
				OR = {
					has_county_modifier = mpo_siberian_permafrost_modifier
					has_county_modifier = mpo_siberian_permafrost_modifier_bad
				}
			}
		}
		add = 2000
	}
	if = {
		limit = {
			trigger_if = {
				limit = { exists = scope:actor }
				scope:actor = { highest_held_title_tier = tier_county }
			} trigger_else = {
				highest_held_title_tier = tier_county
			}
		}
		subtract = 450
	}
	else_if = {
		limit = {
			trigger_if = {
				limit = { exists = scope:actor }
				scope:actor = { highest_held_title_tier = tier_duchy }
			} trigger_else = {
				highest_held_title_tier = tier_duchy
			}
		}
		subtract = 400
	}
	else_if = {
		limit = {
				trigger_if = {
					limit = { exists = scope:actor }
					scope:actor = { highest_held_title_tier = tier_kingdom }
				} trigger_else = {
					highest_held_title_tier = tier_kingdom
				}
		}
		subtract = 350
	}
	else_if = {
		limit = {
			trigger_if = {
				limit = { exists = scope:actor }
				scope:actor = { highest_held_title_tier = tier_empire }
			} trigger_else = {
				highest_held_title_tier = tier_empire
			}
		}
		subtract = 300
	}
	else_if = {
		limit = {
			trigger_if = {
				limit = { exists = scope:actor }
				scope:actor = { highest_held_title_tier = tier_hegemony }
			} trigger_else = {
				highest_held_title_tier = tier_hegemony
			}
		}
		subtract = 250
	}
}
