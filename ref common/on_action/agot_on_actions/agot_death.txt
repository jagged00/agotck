agot_after_death_on_action = {
	trigger = {
		always = yes
	}
	effect = {
		if = {
			limit = { has_trait = dragon }
			save_scope_as = dead_dragon
			if = {
				limit = {
					agot_has_dragon_storage_system_global_list = yes
				}
				every_in_global_list = {
					variable = gl_dragon_variable_storage
					limit = {
						var:dragon_id ?= root
					}
					save_scope_as = dragon_var_story_val
				}
				every_in_list = {
					variable = past_riders
					scope:dragon_var_story_val = {
						add_to_variable_list = {
							name = past_riders
							target = prev
						}
					}
				}
			}
			every_player = {
				limit = { always = yes }
				send_interface_message = {
					type = msg_dragon_death
					title = {
						first_valid = {
							triggered_desc = {
								trigger = {
									agot_is_any_dragon_alive = no
								}
								desc = agot_dragon_maintenance.0904.t_last_dragon_died
							}
							triggered_desc = {
								trigger = {
									any_in_global_list = {
										variable = living_dragons
										count <= 5
									}
								}
								desc = agot_dragon_maintenance.0904.t
							}
							triggered_desc = {
								trigger = {
									any_in_global_list = {
										variable = living_dragons
										count > 5
									}
								}
								desc = agot_dragon_maintenance.0904.t_many_dragons_extant
							}
						}
					}
					desc = agot_dragon_maintenance_msg_dragon_death_placeholder
					left_icon = scope:dead_dragon
				}
			}
			agot_dragon_death_magic_effect = yes
			agot_dead_dragon_effect = yes
		}
		else_if = {
			limit = {
				exists = var:current_dragon
			}
			random_equipped_character_artifact = {
				limit = {
					artifact_type = dragon_horn
				}
				clear_variable_list = tamed_dragons
			}
			save_scope_as = dead_rider
			var:current_dragon = {
				save_scope_as = dead_riders_dragon
			}
			scope:dead_riders_dragon = { trigger_event = { id = agot_dragon.0002 } } # Needs to fire on the dragon because of diseases causing on_death orders of operation trickery.
		}
		else_if = {
			limit = {
				is_human = yes
				any_relation = {
					type = agot_dragon
				}
			}
			save_scope_as = dead_rider
			random_relation = {
				type = agot_dragon
				save_scope_as = dead_riders_dragon
			}
			scope:dead_riders_dragon = { trigger_event = { id = agot_dragon.0002 } } # Needs to fire on the dragon because of diseases causing on_death orders of operation trickery.
		}
		save_scope_as = dead_relative
		every_close_family_member = {
			if = {
				limit = { is_ruler = yes }
				random_list = {
					100 = {}
					1 = {
						trigger_event = {
							id = agot_silent_sisters.0007 # Find Silent Sister defiling relative
							days = { 5 7 }
						}
					}
					1 = {
						trigger_event = {
							id = agot_silent_sisters.0005 # Find Silent Sister Aluring
							days = { 2 4 }
						}
					}
				}
			}
		}
	}
}

# remove dead ruler from story cycle list and add the new ruler
# change pre_war_liege variable
# story owner succession is dealt with in the respective story cycles this, this includes mw_rebel_leader_list
agot_mw_death_on_action = {
	trigger = {
		OR = {
			has_variable = pre_war_liege
			any_liege_or_above = {
				OR = {
					agot_mw_crown_trigger = yes
					has_variable = pre_war_liege
					agot_mw_rebel_leader_trigger = yes
				}
			}
		}
	}
	effect = {
		### Fetch Scopes ###
		agot_mw_fetch_crown_story_from_char_scope = { CHECK_CHAR = root }
		agot_mw_fetch_rebel_story_from_char_scope = { CHECK_CHAR = root }
		### Fetching Scope End ###

		if = {
			limit = {
				NOT = { exists = scope:mw_crown_story }
			}
			debug_log = "ERROR MW_0100: Failed to fetch scope 'mw_crown_story'"
			debug_log_date = yes
		}
		if = {
			limit = {
				NOT = { exists = scope:mw_rebel_story }
			}
			debug_log = "ERROR MW_0100: Failed to fetch scope 'mw_rebel_story'"
			debug_log_date = yes
		}

		# Update mega war lists if necessary
		if = {
			limit = {
				scope:mw_crown_story ?= {
					OR = {
						is_target_in_variable_list = { name = mw_loyalist_list target = root }
						is_target_in_variable_list = { name = mw_neutral_list target = root }
					}
				}
			}

			every_heir = {
				limit = {
					any_heir_title = { tier > tier_barony }
				}
				if = {
					limit = {
						NOR = {
							agot_mw_is_in_loyalist_list_trigger = { TARGET = scope:this_scope }	# don't add him, if he is already in loyalist list
							agot_mw_is_in_neutral_list_trigger = { TARGET = scope:this_scope }	# don't add him, if he is already in neutral list
							agot_mw_is_in_rebel_list_trigger = { TARGET = scope:this_scope }	# don't add him, if he is already in rebel list
						}
					}
					scope:mw_crown_story ?= {
						if = {
							limit = { is_target_in_variable_list = { name = mw_loyalist_list target = root } }

							add_to_variable_list = { name = mw_loyalist_list target = prev} # add new ruler
						}
						if = {
							limit = { is_target_in_variable_list = { name = mw_neutral_list target = root } }

							add_to_variable_list = { name = mw_neutral_list target = prev } # add new ruler
						}
					}
				}
				else_if = { # if primary heir already was in a list, we now mark this, so know whether we also need to update pre war liege below and whether to update de facto liege
					limit = {
						OR = {
							agot_mw_is_in_loyalist_list_trigger = { TARGET = scope:this_scope }
							agot_mw_is_in_neutral_list_trigger = { TARGET = scope:this_scope }
							agot_mw_is_in_rebel_list_trigger = { TARGET = scope:this_scope }
						}
					}

					set_variable = { name = mw_pwl_update_exempt days = 1 }
				}
			}
			scope:mw_crown_story ?= { # remove dead ruler
				remove_list_variable = { name = mw_loyalist_list target = root }
				remove_list_variable = { name = mw_neutral_list target = root }
				remove_list_variable = { name = mw_rebel_leader_list target = root }
			}
		}
		else_if = {
			limit = {
				scope:mw_rebel_story ?= {
					is_target_in_variable_list = { name = mw_rebel_supporter_list target = root }
				}
				exists = root.primary_heir
			}
			scope:mw_rebel_story ?= {
				remove_list_variable = { name = mw_rebel_supporter_list target = root } # remove dead ruler
				add_to_variable_list = { name = mw_rebel_supporter_list target = root.primary_heir } # add new ruler
			}
		}

		# Update pre_war_liege if necessary
		if = {
			limit = {
				any_ruler = {
					var:pre_war_liege ?= root
				}
			}

			every_ruler = { # every ruler that has the variable with root as value
				limit = {
					var:pre_war_liege ?= root
					NOT = { this = root.primary_heir }
				}
				set_variable = { name = pre_war_liege value = root.primary_heir }
			}
		}
		every_heir = {
			limit = {
				any_heir_title = { tier = root.highest_held_title_tier }
			}

			if = { # we have a pre_war_liege set for dead char and may transfer it to heir
				limit = {
					exists = root.var:pre_war_liege
					NOT = { has_variable = mw_pwl_update_exempt }
				}
				set_variable = { name = pre_war_liege value = root.var:pre_war_liege }
			}
			else_if = { # there is no pre_war_liege set or we had heir being extempt from variable transfer --> we need to make sure his current can still be vassalized by his set pre_war_liege
				limit = {
					has_variable = mw_pwl_update_exempt
				}
				if = {
					limit = { # if transferred title rank is greater or equal to pre_war_liege rank, then we need to step up pre_war_liege
						exists = var:pre_war_liege.highest_held_title_tier
						root.highest_held_title_tier >= var:pre_war_liege.highest_held_title_tier
					}

					set_variable = { name = pre_war_liege value = var:pre_war_liege.liege }
				}
			}
		}
	}
}

agot_landless_pirate_domicile_transfer = {
	trigger = {
		any_held_title = {
			agot_is_pirate_domicile_title_trigger = yes
		}
	}
	effect = {
		if = {
			limit = {
				player_heir ?= {
					agot_pirate_landless_succession_jank_courtier_checker = { DYING_PIRATE = ROOT }
					trigger_if = {
						limit = {
							has_variable = pirate_jank_succession_variable
						}
						var:pirate_jank_succession_variable = root
					}
				}
			}
			player_heir = {
				set_variable = pirate_jank_succession_variable
				save_scope_as = chosen_successor
			}
		}
		else_if = { # Any Designated Pirate Ship Heirs
			limit = {
				any_courtier = {
					var:pirate_jank_succession_variable ?= root
					agot_pirate_landless_succession_jank_courtier_checker = { DYING_PIRATE = root }
				}
			}
			random_courtier = {
				limit = {
					var:pirate_jank_succession_variable ?= root
					agot_pirate_landless_succession_jank_courtier_checker = { DYING_PIRATE = root }
				}
				save_scope_as = chosen_successor
			}
		}
		else_if = {
			limit = {
				any_child = {
					agot_pirate_landless_succession_jank_courtier_checker = { DYING_PIRATE = root }
					is_alive = yes
				}
			}
			ordered_child = {
				limit = {
					is_alive = yes
					agot_pirate_landless_succession_jank_courtier_checker = { DYING_PIRATE = root }
					is_adult = yes
				}
				alternative_limit = {
					is_alive = yes
					agot_pirate_landless_succession_jank_courtier_checker = { DYING_PIRATE = root }
				}
				position = 0
				order_by = prowess
				set_variable = pirate_jank_succession_variable
				save_scope_as = chosen_successor
			}
		}
		else_if = { # Prioritize Primary Heirs
			limit = {
				exists = primary_heir
			}
			primary_heir = {
				set_variable = pirate_jank_succession_variable
				save_scope_as = chosen_successor
			}
		}
		else = {
			ordered_courtier = {
				position = 0
				order_by = prowess
				set_variable = pirate_jank_succession_variable
				save_scope_as = chosen_successor
			}
		}
		if = {
			limit = {
				exists = scope:chosen_successor
			}
			# Maa Transfer
			if = {
				limit = {
					is_landed = no
					number_of_maa_regiments > 0
				}
				create_character = {
					name = Aegon
					gender = male
					employer = title:c_ruins.holder
					trait = agot_dummy_trait
					random_traits = no
					faith = faith:fots_seven
					culture = culture:northman
					dynasty = none
					save_scope_as = pirate_dummy_to_transfer_maas
				}
				agot_create_pirate_domicile_effect = { PIRATE = scope:pirate_dummy_to_transfer_maas }
				scope:pirate_dummy_to_transfer_maas = {
					if = {
						limit = {
							number_of_maa_regiments > 0
						}
						every_maa_regiment = {
							destroy_maa_regiment = yes
						}
					}
				}
				every_maa_regiment = {
					save_scope_as = maa_regiment_to_transfer
					save_scope_value_as = {
						name = maa_size_scope_value
						value = scope:maa_regiment_to_transfer.maa_size
					}
					scope:pirate_dummy_to_transfer_maas = {
						create_maa_regiment = {
							type_of = scope:maa_regiment_to_transfer
							check_can_recruit = no
							size = 1
						}
						scope:maa_regiment_to_transfer = {
							destroy_maa_regiment = yes
						}
					}
				}
			}
			# Maa Transfer End
			random_held_title = {
				limit = {
					agot_is_pirate_domicile_title_trigger = yes
				}
				save_scope_as = domicile_being_passed_down
				create_title_and_vassal_change = {
					type = granted
					save_scope_as = change
				}
				change_title_holder_include_vassals = {
					holder = scope:chosen_successor
					change = scope:change
				}
			}
			resolve_title_and_vassal_change = scope:change
			agot_create_pirate_domicile_effect = { PIRATE = scope:chosen_successor }
			# Maa transfer 2
			clear_saved_scope = maa_size_scope_value
			clear_saved_scope = maa_regiment_to_transfer

				every_maa_regiment = {
					save_scope_as = maa_regiment_to_transfer
					save_scope_value_as = {
						name = maa_size_scope_value
						value = scope:maa_regiment_to_transfer.maa_size
					}
					scope:chosen_successor = {
						create_maa_regiment = {
							type_of = scope:maa_regiment_to_transfer
							check_can_recruit = no
							size = 1
						}
						scope:maa_regiment_to_transfer = {
							destroy_maa_regiment = yes
						}
						add_character_modifier = {
							modifier = maa_recruitment_boon
							years = 3
						}
					}
				}

					prev = {

					}
				}

			}

			if = {

				}

				}
			}

			if = {

				}
