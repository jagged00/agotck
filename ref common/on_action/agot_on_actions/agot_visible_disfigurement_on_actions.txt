random_yearly_everyone_pulse = {
	on_actions = {
		agot_visible_disfigurement_pulse # Used for Flavor and Special Events
		agot_one_eyed_pulse
	}
}

agot_visible_disfigurement_pulse = {
	trigger = {
		has_trait = disfigured
		agot_has_visible_disfigurement_trigger = no
		NOT = { has_character_flag = is_Blackfyre_14 }
	}
	effect = {
		trigger_event = {
			id = agot_health_events.3000
		}
	}
}

agot_one_eyed_pulse = {
	trigger = {
		has_trait = one_eyed
		NOT = { has_character_flag = one_eyed_healed }
	}
	effect = {
		if = {
			limit = { is_ai = yes }
			agot_random_one_eyed_fashion_effect = yes
			add_character_flag = one_eyed_healed
		}
		else = {
			trigger_event = agot_health_events.4000
		}
	}
}
