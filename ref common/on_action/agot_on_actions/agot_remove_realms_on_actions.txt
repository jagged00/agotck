agot_remove_realms = {
	trigger = {
		NOT = { has_game_rule = agot_essos_enabled }
	}
	effect = {
		agot_remove_realms_scenario_effect = yes
		if = {
			limit = { agot_realm_three_daughters_enabled = no }
			agot_remove_realm_effect = { REALM = title:e_three_daughters }
		}
		if = {
			limit = { agot_realm_pentos_enabled = no }
			agot_remove_realm_effect = { REALM = title:e_pentos }
		}
		if = {
			limit = { agot_realm_braavos_enabled = no }
			agot_remove_realm_effect = { REALM = title:e_braavos }
		}
		title:c_unknown.holder = {
			every_held_title = {
				set_color_from_title = title:c_unknown
			}
			every_courtier_or_guest = {
				death = {
					death_reason = death_vanished
				}
			}
		}
	}
}
