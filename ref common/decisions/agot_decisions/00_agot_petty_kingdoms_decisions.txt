# Runestone
agot_proclaim_pk_runestone = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_found_kingdom.dds"
	}
	decision_group_type = major
	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 120
		kingdom = 120
		empire = 120
		hegemony = 120
	}
	desc = agot_proclaim_pk_runestone_desc

	is_shown = {
		is_independent_ruler = yes
		primary_title.tier = tier_duchy
		has_title = title:d_runestone
		NOR = {
			has_variable = pre_war_liege
			has_global_variable = pk_runestone_decision
		}
	}

	is_valid = {
		custom_description = {
			text = agot_proclaim_pk
			any_held_title = {
				tier = tier_duchy
				count >= 2
				save_temporary_scope_as = de_jure_duchy
				root = { completely_controls = scope:de_jure_duchy }
			}
		}
		gold >= 400
		is_imprisoned = no
		agot_is_independent_ruler = yes
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 1000
	}

	effect = {
		add_prestige = 400
		get_title = title:k_runestone

		title:e_the_vale = { save_temporary_scope_as = empire }

		title:k_runestone = {
			set_de_jure_liege_title = scope:empire
			save_temporary_scope_as = k_runestone
		}
		every_held_title = {
			limit = { tier = tier_duchy }
			set_de_jure_liege_title = scope:k_runestone
		}
		set_global_variable = pk_runestone_decision
		save_scope_as = i_am_the_king
		every_player = {
			limit = { NOT = { this = scope:i_am_the_king } }
			trigger_event = {
				id = agot_proclaim.0100
			}
		}
	}

	cost = { gold = 400 }
}
