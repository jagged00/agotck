	picture = { reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds" }
	desc = reclaim_gift_decision_desc
	selection_tooltip = reclaim_gift_decision_tooltip
	sort_order = 200
	decision_group_type = major

	is_shown = {
		NOT = { exists = global_var:new_gift_reclaimed }
		NOT = { exists = global_var:new_gift_reclaim_cd }

			government_has_flag = government_is_nw
		}

			any_de_jure_county = {

				holder = {
					top_liege = title:k_the_wall.holder

			}
		}
	}

	is_valid_showing_failures_only = {
		is_at_war = no
		custom_tooltip = {
			text = reclaim_gift_decision.requirements

				is_allied_to = title:k_the_wall.holder
				has_perk = diplomat_perk
				title:k_the_wall.holder = {
					opinion = {
						target = root
						value = 100
					}
				}
			}
		}
		trigger_if = {
			limit = {
				agot_is_independent_ruler = no
			}
			has_usable_hook = top_liege
		}
		title:k_the_wall.holder = {
			is_at_war = no
		}
		gold >= 3000
	}

	is_valid = {}

	effect = {
		custom_tooltip = reclaim_gift_decision_desc_effect
		title:k_the_wall.holder = {
			trigger_event = agot_nights_watch.3000
		}
		if = {
			limit = {
				agot_is_independent_ruler = no
			}
			custom_tooltip = reclaim_gift_decision_desc_use_hook_tt
			hidden_effect = { use_hook = top_liege }
		}
	}

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0

		empire = 120
		hegemony = 120

	}

	ai_will_do = {

			}

donate_land_to_nw_decision = {
	picture = { reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds" }
	desc = donate_land_to_nw_decision_desc
	selection_tooltip = donate_land_to_nw_decision_tooltip
	sort_order = 200
	decision_group_type = major
	cooldown = { years = 1 }

	is_shown = {
		OR = {
			completely_controls = title:d_brandons_gift
			completely_controls = title:d_new_gift
		}
		NOR = {
			government_has_flag = government_is_nw
			title:k_the_wall.holder ?= root
		}
		title:k_the_wall.holder ?= {
			government_has_flag = government_is_nw
		}
	}

	is_valid_showing_failures_only = {
		is_at_war = no
		trigger_if = {
			limit = {
				agot_is_independent_ruler = no
			}
			has_usable_hook = top_liege
		}
		title:k_the_wall.holder = {
			is_at_war = no
		}
		trigger_if = {
			limit = {
				exists = global_var:old_gift_reclaimed
			}
			title:c_weatherback_ridge.holder = root
			title:c_moles_town.holder = root
			title:c_hoarburn.holder = root
		}
		trigger_else_if = {
			limit = {
				NOT = {
					exists = global_var:old_gift_reclaimed
				}
				exists = global_var:new_gift_reclaimed
			}
			title:c_norlinton.holder = root
			title:c_boreharrow.holder = root
			title:c_whitecross.holder = root
			title:c_giantsbeard.holder = root
		}
		trigger_else = { always = yes }
	}

	effect = {
		if = {
			limit = {
				exists = global_var:old_gift_reclaimed
			}
			add_to_variable_list = {
				name = brandons_gift_counties_list
				target = title:c_weatherback_ridge
			}
			add_to_variable_list = {
				name = brandons_gift_counties_list
				target = title:c_moles_town
			}
			add_to_variable_list = {
				name = brandons_gift_counties_list
				target = title:c_hoarburn
			}
			create_title_and_vassal_change = {
				type = granted
				save_scope_as = change
			}
			every_in_list = {
				variable = brandons_gift_counties_list
				change_title_holder_include_vassals = {
					holder = title:k_the_wall.holder
					change = scope:change
				}
				set_coa = c_castle_black
				if = {
					limit = {
						any_county_province = {
							has_holding_type = city_holding
						}
					}
					random_county_province = {
						limit = {
							has_holding_type = city_holding
						}
						barony = {
							set_capital_barony = yes
						}
					}
				}
				else = {
					title_province = {
						set_holding_type = city_holding
					}
				}
			}
			resolve_title_and_vassal_change = scope:change
			remove_global_variable = old_gift_reclaimed
			save_scope_as = reclaim_brandons_gift
			set_global_variable = {
				name = brandons_gift_reclaim_cd
				years = 100
			}
			custom_tooltip = brandons_gift_will_be_returned_to_nw_tt
		}
		else_if = {
			limit = {
				NOT = {
					exists = global_var:old_gift_reclaimed
				}
				exists = global_var:new_gift_reclaimed
			}
			custom_tooltip = new_gift_will_be_returned_to_nw_tt
			add_to_variable_list = {
				name = new_gift_counties_list
				target = title:c_norlinton
			}
			add_to_variable_list = {
				name = new_gift_counties_list
				target = title:c_boreharrow
			}
			add_to_variable_list = {
				name = new_gift_counties_list
				target = title:c_whitecross
			}
			add_to_variable_list = {
				name = new_gift_counties_list
				target = title:c_giantsbeard
			}
			create_title_and_vassal_change = {
				type = granted
				save_scope_as = change
			}
			every_in_list = {
				variable = new_gift_counties_list
				change_title_holder_include_vassals = {
					holder = title:k_the_wall.holder
					change = scope:change
				}
				set_coa = c_castle_black
				if = {
					limit = {
						any_county_province = {
							has_holding_type = city_holding
						}
					}
					random_county_province = {
						limit = {
							has_holding_type = city_holding
						}
						barony = {
							set_capital_barony = yes
						}
					}
				}
				else = {
					title_province = {
						set_holding_type = city_holding
					}
				}
			}
			resolve_title_and_vassal_change = scope:change
			remove_global_variable = new_gift_reclaimed
			set_global_variable = {
				name = new_gift_reclaim_cd
				years = 100
			}
		}
		title:k_the_wall.holder = {
			trigger_event = agot_nights_watch.3005
		}
	}

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 120
		empire = 120
		hegemony = 120
	}

	ai_potential = {
		is_available = yes
	}

	ai_will_do = {
		base = 5
	}
}
