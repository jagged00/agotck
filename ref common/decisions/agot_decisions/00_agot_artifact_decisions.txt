agot_commission_special_artifact_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_smith.dds"
	}
	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 12
		empire = 12
		hegemony = 12
	}
	desc = agot_commission_special_artifact_decision_desc
	selection_tooltip = agot_commission_special_artifact_decision_tooltip
	sort_order = 80

	decision_group_type = artifact

	is_shown = {
		NOT = {
			government_has_flag = government_is_nw
		}
	}

	is_valid_showing_failures_only = {
		# Only valid for characters with a court smith.
		employs_court_position = court_smith_court_position

		custom_tooltip = {
			text = agot_in_coronation
			NOT = {
				involved_activity ?= { has_activity_type = activity_agot_coronation }
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 0

		modifier = {
			add = 100

			primary_title.tier >= tier_kingdom
			agot_ai_unique_crown_character_trigger = yes
			agot_is_independent_ruler = yes
			agot_can_make_any_crown_trigger = yes
			NOT = {
				has_variable = agot_recently_commissioned_crown
			}
			NOT = {
				any_equipped_character_artifact = {
					artifact_slot_type = helmet
				}
			}
		}

		modifier = {
			factor = 0
			short_term_gold < agot_special_commission_cost
		}
	}

	widget = {
		gui = "decision_view_widget_commission_artifact"
		controller = decision_option_list_controller
		decision_to_second_step_button = "agot_commission_special_artifact_button"

		# List
		item = {
			value = agot_commission_special_artifact_crown_choice
			localization = {
				desc = agot_commission_special_artifact_crown_choice_loc
			}
			icon = "gfx/interface/icons/duchy_crown.dds"
			is_valid = {
				primary_title.tier >= tier_kingdom
				agot_ai_unique_crown_character_trigger = yes
				agot_is_independent_ruler = yes
				agot_can_make_any_crown_trigger = yes
				custom_tooltip = {
					text = agot_commissioned_personal_crown_recently
					NOT = {
						has_variable = agot_recently_commissioned_crown
					}
				}
			}
		}

		item = {
			value = agot_commission_special_artifact_crown_spouse_choice
			localization = {
				desc = agot_commission_special_artifact_crown_spouse_choice_loc
			}
			icon = "gfx/interface/icons/character_interactions/icon_marriage.dds"
			is_valid = {
				is_married = yes
				is_ai = no
				primary_title.tier >= tier_kingdom
				agot_ai_unique_crown_character_trigger = yes
				agot_is_independent_ruler = yes
				primary_spouse ?= {
					agot_can_make_any_crown_trigger = yes
					save_temporary_scope_as = spouse_scope
					custom_tooltip = {
						text = agot_commissioned_spouse_crown_recently
						NOT = {
							has_variable = agot_recently_commissioned_crown
						}
					}
				}
			}
		}

		item = {
			value = agot_commission_special_artifact_armor_choice
			localization = {
				desc = agot_commission_special_artifact_armor_choice_loc
			}
			icon = "gfx/interface/icons/inspirations/armor_inspiration.dds"
			is_valid = {
				is_ai = no
				custom_tooltip = {
					text = agot_commissioned_armor_recently
					NOT = {
						has_variable = agot_recently_commissioned_armor
					}
				}
			}
		}
	}

	effect = {
		custom_tooltip = agot_commission_special_artifact_decision_effects_tooltip
		random_court_position_holder = {
			type = court_smith_court_position
			save_scope_as = smith
		}
		if = {
			limit = { scope:agot_commission_special_artifact_crown_choice = yes }
			root = {
				save_scope_as = artifact_recipient
				trigger_event = agot_commission_special_artifact_crown.0001
			}
		}
		else_if = {
			limit = { scope:agot_commission_special_artifact_crown_spouse_choice = yes }
			root = {
				primary_spouse = { save_scope_as = artifact_recipient }
				trigger_event = agot_commission_special_artifact_crown.0001
			}
		}
		else_if = {
			limit = { scope:agot_commission_special_artifact_armor_choice = yes }
			root = {
				save_scope_as = artifact_recipient
				trigger_event = agot_commission_special_artifact_armor.0001
			}
		}
	}

	cost = { gold = agot_special_commission_cost }
}

agot_sell_artifacts_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}
	desc = agot_sell_artifacts_decision_desc
	sort_order = 75

	decision_group_type = artifact

	is_shown = {
		agot_can_trade_artifacts_trigger = yes
		NOT = { government_has_flag = government_is_first_ranger }
	}

	is_valid = {
		custom_description = {
			text = agot_stolen_artifact_from_artifact_keeper_dec_trigger
			trigger_if = {
				limit = {
					exists = house
				}
				NOT = {
					house = {
						exists = var:agot_stolen_artifact_from_artifact_keeper
					}
				}
			}
		}
		custom_description = {
			text = agot_not_wilding_desc_trigger
			culture = {
				agot_is_wildling_culture = no
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_check_interval_by_tier = {
		barony = 0
		county = 24
		duchy = 24
		kingdom = 24
		empire = 24
		hegemony = 24
	}

	effect = {
		custom_tooltip = agot_sell_artifacts_decision_custom_tooltip_1

		# Compatibility with old save
		hidden_effect = {
			if = {
				limit = {
					NOT = {
						exists = global_var:agot_artifact_keeper
					}
				}
				agot_get_artifact_keeper = yes
			}
			else_if = {
				limit = {
					global_var:agot_artifact_keeper = {
						OR = {
							is_alive = no
							is_ai = no
						}
					}
				}
				agot_reload_artifact_keeper = yes
			}
			else_if = {
				limit = {
					NOT = {
						global_var:agot_artifact_keeper_title.holder = global_var:agot_artifact_keeper
					}
				}
				global_var:agot_artifact_keeper_title = {
					create_title_and_vassal_change = {
						type = conquest
						save_scope_as = change
						add_claim_on_loss = no
					}
					change_title_holder_include_vassals = {
						holder = global_var:agot_artifact_keeper
						change = scope:change
					}

					resolve_title_and_vassal_change = scope:change
				}
			}
			if = {
				limit = {
					NOT = {
						global_var:agot_artifact_keeper = {
							is_independent_ruler = yes
						}
					}
				}
				global_var:agot_artifact_keeper = {
					create_title_and_vassal_change = {
						type = independency
						save_scope_as = change
						add_claim_on_loss = yes
					}
					becomes_independent = {
						change = scope:change
					}
					resolve_title_and_vassal_change = scope:change
				}
			}
		}

		# Refersh market and owned artifacts
		agot_get_market_artifacts_effect = yes
		agot_get_owned_artifacts_effect = yes

		if = {
			limit = {
				is_ai = yes
				NOT = { government_has_flag = government_is_first_ranger }
			}

			# Either sell or buy an artifact
			random_list = {
				1 = {} # Do nothing
				50 = { # Sell artifact
					trigger = {
						agot_ai_may_sell_artifact_trigger = yes
					}

					agot_ai_get_artifact_to_sell_effect = {
						ARTIFACT = artifact_to_sell
					}

					if = {
						limit = { exists = scope:artifact_to_sell }
						agot_sell_artifact_effect = {
							ARTIFACT = scope:artifact_to_sell
						}
					}
				}
				50 = { # Buy artifact
					trigger = {
						# Performance: Limit buying of artifacts to dukes and above
						highest_held_title_tier >= tier_duchy
						agot_ai_may_buy_artifact_trigger = yes
					}

					agot_ai_get_artifact_to_buy_effect = {
						ARTIFACT = artifact_to_buy
					}

					if = {
						limit = { exists = scope:artifact_to_buy }
						agot_buy_artifact_effect = {
							ARTIFACT = scope:artifact_to_buy
						}
					}
				}
			}
		}
		else = {
			# Show artifact market window
			set_variable = {
				name = agot_artifact_market_window_open
				value = 1
			}
		}
	}

	ai_will_do = {
		base = 100
	}
}
