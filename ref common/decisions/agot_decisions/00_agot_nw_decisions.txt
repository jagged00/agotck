## AI decision to start (or join) a wall reclamation war on behalf of NW
reclaim_wall_for_nw = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}

	# Never shown, so it doesn't matter what we write here
	title = placeholder
	desc = placeholder
	confirm_text = placeholder
	selection_tooltip = placeholder

	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 120
		duchy = 120
		kingdom = 120
		empire = 120
		hegemony = 120
	}

	is_shown = {
		is_available_ai_adult = yes
	}

	# Must be a landed ruler, not a nightswatch, nightswatch must exist, and one of NW de jure lands must be held by a non-vassal wildling
	is_valid = {
		is_landed = yes
		root.capital_barony.title_province = { geographical_region = world_westeros_seven_kingdoms }
		NOT = { has_trait = nightswatch }
		exists = title:k_the_wall.holder
		title:k_the_wall.holder = { government_has_flag = government_is_nw }

		title:k_the_wall = {
			any_in_de_jure_hierarchy = {
				exists = holder
				holder.top_liege = { culture = { agot_is_wildling_culture = yes } }
				NOT = { holder.top_liege = title:k_the_wall.holder }
			}
		}
	}

	is_valid_showing_failures_only = {
		is_at_war = no
	}

	effect = {
		# Find a wall de jure title that is held by a non-vassal wildling
		title:k_the_wall = {
			random_in_de_jure_hierarchy = {
				limit = {
					exists = holder
					holder.top_liege = { culture = { agot_is_wildling_culture = yes } }
					NOT = { holder.top_liege = title:k_the_wall.holder }
				}


				holder.top_liege = {
					if = {
						limit = {
							any_character_war = {
								is_defender = prev
								using_cb = agot_wall_reclamation_cb
							}
						}

						random_character_war = {
							limit = {
								is_defender = prev
								using_cb = agot_wall_reclamation_cb
							}

							add_attacker = root
						}
					}
					else = {
						root = {
							start_war = {
								cb = agot_wall_reclamation_cb
								target = prev
								claimant = this
							}
						}
					}
				}
			}
		}
	}

	ai_will_do = {
		base = -20

		ai_value_modifier = {
			who = root
			ai_boldness = 0.2
			ai_greed = -0.5
			ai_honor = 0.5
			ai_rationality = 0.2
			ai_zeal = 0.2
		}

		modifier = {
			add = 10
			is_landed = yes
			root.capital_barony.title_province = { geographical_region = world_westeros_the_north }
		}

		modifier = {
			factor = 0
			root = {
				debt_level > 0
			}
		}

		modifier = {
			factor = 0
			exists = title:k_the_wall.holder
			root.max_military_strength < title:k_the_wall.holder.max_military_strength
		}
	}
}

desert_the_nw = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_beyond_the_wall.dds"
	}

	title = desert_the_nw
	desc = desert_the_nw_desc
	confirm_text = desert_the_nw_confirm
	selection_tooltip = desert_the_nw_tooltip

	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 6
		duchy = 6
		kingdom = 0
		empire = 0
		hegemony = 0
	}

	is_shown = {
		agot_is_landless_nw_character = yes
		has_perk = twice_schemed_perk
	}

	is_valid = {
		is_at_war = no
		custom_description = {
			text = first_ranger_has_to_be_beyond_the_wall
			agot_ranger_beyond_the_wall = yes
		}
	}

	is_valid_showing_failures_only = {
		is_at_war = no
	}

	effect = {
		hidden_effect = { return_home = yes }
		save_scope_as = deserter

		hidden_effect = { title:d_dummy_nw_landless = { copy_title_history = title:d_nw_landless_first_ranger } }

		#Save deserters
		if = {
			limit = {
				any_courtier = {
					is_alive = yes
					has_trait = nightswatch
					save_temporary_opinion_value_as = {
						name = opinion_of_the_deserter
						target = scope:deserter
					}
					scope:opinion_of_the_deserter >= 25
					ai_honor < 50
				}
			}
			every_courtier = {
				limit = {
					is_alive = yes
					has_trait = nightswatch
					save_temporary_opinion_value_as = {
						name = opinion_of_the_deserter
						target = scope:deserter
					}
					scope:opinion_of_the_deserter >= 25
					ai_honor < 50
				}
				random = {
					chance = {
						value = 0
						add = { value = ai_honor multiply = -0.4 }
						add = { value = ai_boldness multiply = 0.2 }
						add = { value = ai_rationality multiply = -0.2 }
						add = { value = ai_greed multiply = 0.1 }
						add = { value = ai_zeal multiply = -0.1 }
					}
					remove_trait = nightswatch
					remove_trait = lifestyle_nw_ranger
					add_trait_force_tooltip = deserter
					add_character_flag = {
						flag = desert_with_me
						days = 10
					}
					add_to_list = also_deserters
				}
			}
		}

		#Become a deserter
		remove_trait = nightswatch
		remove_trait = lifestyle_nw_ranger
		add_trait_force_tooltip = deserter
		remove_character_flag = nw_ranger_title

		#Honorable friend's reaction
		if = {
			limit = {
				any_courtier = {
					is_alive = yes
					has_trait = nightswatch
					NOT = { has_character_flag = desert_with_me }
					has_relation_friend = scope:deserter
					ai_honor >= 50
				}
			}
			every_courtier = {
				limit = {
					is_alive = yes
					has_trait = nightswatch
					NOT = { has_character_flag = desert_with_me }
					has_relation_friend = scope:deserter
					ai_honor >= 50
				}
				remove_relation_friend = scope:deserter
				if = {
					limit = { can_set_relation_rival_trigger = { CHARACTER = scope:deserter } }
					set_relation_rival = {
						reason = rival_betrayed_friendship
						target = scope:deserter
					}
				}
			}
		}

		#End the story & send back the remaining rangers to the wall
		if = {
			limit = {
				any_courtier = {
					is_alive = yes
					has_trait = nightswatch
					NOT = { has_character_flag = desert_with_me }
				}
			}
			every_courtier = {
				limit = {
					is_alive = yes
					has_trait = nightswatch
					NOT = { has_character_flag = desert_with_me }
				}
				add_to_list = going_back_to_the_wall
			}
		}
		hidden_effect = {
			end_ranger_story = yes
			every_in_list = {
				list = going_back_to_the_wall
				title:k_the_wall.holder = { add_courtier = prev }
			}
		}

		#Save Family member's real home
		agot_ranger_deserts_pre_effect = yes

		#Create laamp & add deserters
		create_landless_adventurer_title_effect = {
			REASON = flag:voluntary
			FLAVOR_CHAR = root
		}
		every_in_list = {
			list = also_deserters
			scope:deserter = { add_courtier = prev }
		}

		#Send Family member back to their real home
		agot_ranger_deserts_post_effect = yes

		#Notife the LC
		title:k_the_wall.holder = {
			add_character_flag = fr_assign_block
			trigger_event = {
				id = agot_nw_ranger_events.0005
				days = 1
			}
		}

		#Create small ranger army
		save_scope_as = secret_child
		spawn_army = {
			levies = 0
			men_at_arms = {
				type = nw_rangers
				stacks = 2
			}
			inheritable = no
			location = scope:deserter.location
			name = secret_spawned_army
		}

		#LC hates me
		show_as_tooltip = {
			title:k_the_wall.holder = {
				add_opinion = {
					target = scope:deserter
					modifier = agot_deserted_nw
					opinion = -100
				}
			}
		}

		#Save a scope for possible faith and culture conversion
		hidden_effect = {
			if = {
				limit = {
					exists = location
					location.duchy = {
						any_de_jure_county_holder = {
							NOT = { government_has_flag = government_is_nw }
							NOT = { government_has_flag = government_is_uninteractable }
						}
					}
				}
				location.duchy = {
					random_de_jure_county_holder = {
						limit = {
							NOT = { government_has_flag = government_is_nw }
							NOT = { government_has_flag = government_is_uninteractable }
						}
						save_scope_as = culture_and_faith_base
						scope:deserter = {
							set_variable = {
								name = culture_and_faith
								value = scope:culture_and_faith_base
							}
							trigger_event = {
								id = agot_nw_ranger_events.1706
								days = { 60 120 }
							}
						}
					}
				}
			}
			else_if = {
				limit = {
					exists = location
					location.kingdom = {
						any_de_jure_county_holder = {
							NOT = { government_has_flag = government_is_nw }
							NOT = { government_has_flag = government_is_uninteractable }
						}
					}
				}
				location.kingdom = {
					random_de_jure_county_holder = {
						limit = {
							NOT = { government_has_flag = government_is_nw }
							NOT = { government_has_flag = government_is_uninteractable }
						}
						save_scope_as = culture_and_faith_base
						scope:deserter = {
							set_variable = {
								name = culture_and_faith
								value = scope:culture_and_faith_base
							}
							trigger_event = {
								id = agot_nw_ranger_events.1706
								days = { 60 120 }
							}
						}
					}
				}
			}
			else_if = {
				limit = { game_start_date >= 75.1.1 }
				scope:deserter = {
					set_variable = {
						name = culture_and_faith
						value = character:King_Beyond_The_Wall_1
					}
				}
			}
			#Just in case lol
			else = {
				create_character = {
					gender = male
					location = scope:deserter.location
					faith = faith:old_gods_btw_fnf
					culture = culture:forestman
					save_scope_as = wildling_dummy
					after_creation = {
						scope:deserter = {
							set_variable = {
								name = culture_and_faith
								value = scope:wildling_dummy
							}

				}
			}

	}

	ai_potential = {
		ai_honor < 50
	}

	ai_will_do = {
		base = 0

		ai_value_modifier = {
			who = root
			ai_honor = -0.4
			ai_boldness = 0.2
			ai_rationality = -0.2
			ai_greed = 0.1
			ai_zeal = -0.1

		opinion_modifier = {
