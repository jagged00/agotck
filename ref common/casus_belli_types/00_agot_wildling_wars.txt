wildling_diminish_cb = {
	group = struggle
	icon = struggle

	ai_only_against_neighbors = yes

	allow_hostages = no
	interface_priority = 110

	allowed_for_character = {
		# Defender must be Involved in BTW Struggle
		any_character_struggle = {
			involvement = involved
			is_struggle_type = btw_struggle
		}
	}

	allowed_against_character = {
		#Multi County
		OR = {
			any_vassal = {
				primary_title.tier >= tier_county
			}
			any_held_title = {
				tier = tier_county
				count > 1
			}
		}

		# Defender must be Involved in BTW Struggle
		any_character_struggle = {
			involvement = involved
			is_struggle_type = btw_struggle
		}

		# Not same realm
		scope:attacker = {
			NOR = {
				top_liege = scope:defender.top_liege
				liege = scope:defender
			}
		}

		# Attacker's vassals or lieges must not be at war with defender
		scope:attacker = {
			NOR = {
				any_vassal_or_below = {
					is_at_war_with = scope:defender
				}
				any_liege_or_above = {
					is_at_war_with = scope:defender
				}
			}
		}
	}

	white_peace_possible = yes

	ai_score = {
		value = {
			value = 25
			multiply = scope:attacker.ai_boldness
		}
	}

	ai_score_mult = {
		value = viking_conquest_ai_score_value
		add = {
			if = {
				limit = {
					scope:defender.highest_held_title_tier >= tier_kingdom
				}
				add = 3
			}
		}
	}

	should_invalidate = {
		scope:defender = {
			NOR = {
				any_vassal = {
					primary_title.tier >= tier_county
				}
				any_held_title = {
					tier = tier_county
					count > 1
				}
			}
		}
	}

	cost = {
		prestige = {
			add = {
				value = medium_prestige_value
				desc = CB_BASE_COST
			}
			if = {
				limit = {
					exists = scope:attacker.primary_title
					scope:attacker.primary_title = {
						tier = tier_kingdom
					}
				}
				multiply = {
					value = 1.5
					desc = CB_COST_RULES_KINGDOM
				}
			}
			else_if = {
				limit = {
					exists = scope:attacker.primary_title
					scope:attacker.primary_title = {
						tier = tier_empire
					}
				}
				multiply = {
					value = 2.5
					desc = CB_COST_RULES_EMPIRE
				}
			}
			multiply = common_cb_prestige_cost_multiplier
		}
	}

	on_declaration = {
		on_declared_war = yes
		debug_log = "Wildling Diminish War"
		if = {
			limit = {
				NOT = { exists = global_var:diminishment_wars }
			}
			set_global_variable = {
				name = diminishment_wars
				value = 1
			}
		}
		else = {
			change_global_variable = {
				name = diminishment_wars
				add = 1
			}
		}
	}

	on_victory_desc = {
		first_valid = {
			desc = wildling_diminish_victory_desc
		}
	}

	on_victory = {
		#EP2 accolade glory gain for winning against higher ranked enemy
		scope:attacker = { accolade_attacker_war_end_glory_gain_low_effect = yes }

		scope:defender.primary_title = { save_scope_as = defender_realm }
		# Destroy all defender titles in the same tier as their primary title
		custom_tooltip = {
			text = shatter_wildling_realm
			scope:defender = {
				every_vassal_or_below = {
					every_held_title = {
						limit = {
							tier = tier_county
							NOT = {
								this = prev.capital_county
							}
						}
						save_scope_as = target_county
						make_settlement_county_wilderness = { COUNTY = scope:target_county }
					}
					every_held_title = {
						limit = {
							tier > tier_county
						}
						scope:attacker = {
							destroy_title = prev
						}
					}
				}
				every_held_title = {
					limit = {
						tier = tier_county
						NOT = {
							this = prev.capital_county
						}
					}
					save_scope_as = target_county
					make_settlement_county_wilderness = { COUNTY = scope:target_county }
				}
				every_held_title = {
					limit = {
						tier > tier_county
					}
					scope:attacker = {
						destroy_title = prev
					}
				}
			}
		}

		#Prestige loss for defender
		scope:defender = {
			add_prestige = {
				add = {
					value = medium_prestige_loss
				}
				multiply = 2
				if = {
					limit = {
						exists = scope:attacker.primary_title
						scope:attacker.primary_title = {
							tier = tier_kingdom
						}
					}
					multiply = 1.5
				}
				else_if = {
					limit = {
						exists = scope:attacker.primary_title
						scope:attacker.primary_title = {
							tier = tier_empire
						}
					}
					multiply = 2.5
				}
			}
		}

		# Prestige for the attacker
		scope:attacker = {
			add_prestige = {
				add = {
					value = medium_prestige_value
				}
				multiply = 2
				if = {
					limit = {
						exists = scope:attacker.primary_title
						scope:attacker.primary_title = {
							tier = tier_kingdom
						}
					}
					multiply = 1.5
				}
				else_if = {
					limit = {
						exists = scope:attacker.primary_title
						scope:attacker.primary_title = {
							tier = tier_empire
						}
					}
					multiply = 2.5
				}
			}
		}

		# Prestige for the attacker's war allies
		add_from_contribution_attackers = {
			prestige = minor_prestige_value
			opinion = {
				modifier = contributed_in_war
			}
		}

		# Prestige for the defender's war allies
		add_from_contribution_defenders = {
			prestige = minor_prestige_value
			opinion = {
				modifier = contributed_in_war
			}
		}

		# Truce
		add_truce_attacker_victory_effect = yes
	}

	on_defeat_desc = {
		first_valid = {
			desc = great_ranging_defeat_desc
		}
	}

	on_defeat = {
		#EP2 accolade glory gain for winning against higher ranked enemy
		scope:defender = { accolade_defender_war_end_glory_gain_low_effect = yes }

		# Prestige loss for the attacker
		scope:attacker = {
			add_prestige = -500
		}

		# Prestige for the defender
		scope:defender = {
			add_prestige = 500
		}

		# Prestige for the attacker's war allies
		add_from_contribution_attackers = {
			prestige = minor_prestige_value
			opinion = {
				modifier = contributed_in_war
			}
		}

		# Prestige for the defender's war allies
		add_from_contribution_defenders = {
			prestige = minor_prestige_value
			opinion = {
				modifier = contributed_in_war
			}
		}

		# Truce
		add_truce_attacker_defeat_effect = yes

		agot_war_victory_effects = yes
	}

	on_white_peace_desc = {
		first_valid = {
			desc = wildling_diminish_white_peace_desc
		}
	}

	on_invalidated_desc = {
		first_valid = {
			desc = wildling_diminish_invalidated_desc
		}
	}

	transfer_behavior = transfer

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit

	attacker_allies_inherit = yes
	defender_allies_inherit = yes

	max_ai_diplo_distance_to_title = 500

	war_name = WILDLING_DIMINISH_WAR_NAME
	war_name_base = WILDLING_DIMINISH_WAR_NAME_BASE
	cb_name = WILDLING_DIMINISH_CB_NAME
}
