murder_attendee_intent = {
	icon = murder_intent

	is_shown = {
		has_dlc_feature = advanced_activities
		is_alive = yes
		is_adult = yes
		trigger_if = {
			limit = {
				OR = {
					scope:special_option ?= flag:hunt_type_standard
					scope:special_option ?= flag:hunt_type_falconry
					scope:special_option ?= flag:hunt_type_nerge
				}
			}
			can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = root }
		}
		NOT = { scope:special_option ?= flag:hunt_type_legendary }
	}

	on_invalidated = {
		trigger_event = activity_system.0090
	}

	ai_targets = {
		ai_recipients = scheme_targets
		ai_recipients = scripted_relations
	}

	# Handle target character invalidation
	is_target_valid = {
		scope:target = {
			is_ai = yes
		}
		OR = {
			can_start_scheme = {
				type = murder
				target_character = scope:target
			}
			is_scheming_against = {
				type = murder
				target = scope:target
			}
		}
	}

	on_target_invalidated = {
		trigger_event = activity_system.0091
	}

	ai_will_do = {
		value = 0
		add = {
			value = root.ai_honor
			multiply = -1
		}
		add = {
			value = root.ai_compassion
			multiply = -1
		}
		if = {
			limit = {
				ai_honor >= high_positive_ai_value
			}
			add = -1000
		}
	}

	ai_target_score = {
		value = 0
		if = {
			limit = {
				has_relation_rival = scope:target
			}
			add = 25
		}
		if = {
			limit = {
				any_scheme = {
					type = murder
					scheme_target_character = scope:target
				}
			}
			add = 150
		}
	}

	scripted_animation = {
		animation = spymaster
	}
}

					}

						ai_honor > 0
						ai_boldness < 0
					}
				}
				AND = {
					trait_is_shunned_in_faith_trigger = { TRAIT = adulterer FAITH = root.faith GENDER_CHARACTER = root }
					is_married = yes
					NOR = {
						is_consort_of = scope:target
						has_trait = adulterer
					}
					ai_honor > 0
				}
			}

			limit = {
				num_of_relation_soulmate = 0
				num_of_relation_lover = 1
				has_trait = chaste
			}
			add = -25
		}
		if = {
			limit = { is_unattractive_trigger = yes }
			add = -40
		}

			limit = {
				ai_boldness > 0

		}

					num_of_relation_soulmate > 0

						trait_is_shunned_in_faith_trigger = { TRAIT = fornicator FAITH = root.faith GENDER_CHARACTER = root }

							is_clergy = yes
							agot_has_religious_trait = yes

							NOR = {
								has_doctrine_parameter = clergy_can_marry
								trait_is_virtue = lustful
							}
						}
					}
