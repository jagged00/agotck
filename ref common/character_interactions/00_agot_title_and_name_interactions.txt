######################################
## Bestow Royal Status Interactions ##
######################################

# This interaction bestows the title of Prince/Princess upon relatives
agot_bestow_royal_status_interaction = {
	category = interaction_category_diplomacy
	desc = agot_bestow_royal_status_interaction_desc
	icon = icon_dynasty

	greeting = positive

	use_diplomatic_range = yes

	cost = {
		prestige = {
			value = medium_prestige_value
		}
	}

	is_shown = {
		scope:actor = {
			is_ai = no
			is_adult = yes
			ruler_is_royal = yes
			NOT = { primary_title ?= title:k_dragonstone }
			is_imprisoned = no
			NOT = { this = scope:recipient }
		}

		scope:recipient = {
			is_imprisoned = no
			is_human = yes
			OR = {
				house ?= scope:actor.house
				any_ancestor = {
					even_if_dead = yes
					this = scope:actor
				}
			}
			NOR = {
				has_trait = bastard
				has_trait = bastard_founder
				agot_bastard_surname_trigger = yes
			}
			is_royal_scion = no
		}
	}

	on_accept = {
		agot_bestow_scion_status_effect = {
			SCOPE = scope:recipient
			SCOPED_ROYAL = scope:actor
			SCOPED_TITLE = scope:actor.primary_title
			OPINION = yes
		}
	}

	ai_frequency = 1

	ai_targets = {
		ai_recipients = courtiers
		max = 10
	}

	ai_potential = {
		always = no
	}

	auto_accept = yes

	ai_will_do = {
		base = 0
	}
}
