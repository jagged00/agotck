FEUDAL_HEGEMONY = {
	type = character

	# Hegemony --> Seven Kingdoms/Iron Throne
	text = {
		trigger = {
			primary_title ?= title:h_the_iron_throne
		}
		localization_key = hegemony_feudal_iron_throne
	}

	# Hegemony --> Principality (Dornish/Rhoynish)
	text = {
		trigger = {
			agot_flavour_is_prince_rhoynish_trigger = { CHAR = this }
		}
		localization_key = title_principality
	}

	# Hegemony --> Kingdom
	text = {
		trigger = {
			agot_flavour_is_in_westeros_trigger = { CHAR = this }
		}
		localization_key = kingdom
	}

	# Fallback: Hegemony
	text = {
		fallback = yes
		localization_key = hegemony
	}
}

FEUDAL_EMPIRE = {
	type = character

	# Empire --> Principality
	text = {
		trigger = {
			OR = {
				# Dragonstone
				agot_flavour_is_prince_dragonstone_trigger = { CHAR = this }
				# Dornish/Rhoynish
				agot_flavour_is_prince_rhoynish_trigger = { CHAR = this }
			}
		}
		localization_key = title_principality
	}

	# Empire --> Claimant Kingdom
	text = {
		trigger = {
			has_character_flag = agot_claimant_king
		}
		localization_key = kingdom
	}

	# Empire --> Lord Paramountcy
	text = {
		trigger = {
			agot_is_part_of_paramountcy_realm = yes
		}
		localization_key = empire_feudal_paramountcy
	}

	# Empire --> Kingdom
	text = {
		trigger = {
			agot_flavour_is_lord_in_westeros_trigger = { CHAR = this }
		}
		localization_key = kingdom
	}

	# Fallback: Empire
	text = {
		fallback = yes
		localization_key = empire
	}
}
