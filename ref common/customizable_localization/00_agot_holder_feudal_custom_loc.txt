FEUDAL_HEGEMONY_MALE = { # Male
	type = character

	# Rhoynish Prince

		trigger = {
			agot_flavour_is_prince_rhoynish_trigger = { CHAR = this }
		}
		localization_key = king_rhoynish

		trigger = {
			agot_flavour_hegemony_is_kingdom_trigger = { CHAR = this }

}

# Ruler Children
PRINCE_MALE = { # Male
	type = character

	# Prince
	text = {
		trigger = {
			OR = {
				father ?= {
					OR = {
						# Dragonstone
						agot_flavour_is_prince_dragonstone_trigger = { CHAR = this }
						# Palatinate
						agot_flavour_palatinate_is_kingdom_trigger = { CHAR = this }
					}
				}
				mother ?= {
					OR = {
						# Dragonstone
						agot_flavour_is_prince_dragonstone_trigger = { CHAR = this }
						# Palatinate
						agot_flavour_palatinate_is_kingdom_trigger = { CHAR = this }
					}
				}
			}
		}
		localization_key = prince
	}

	# Lord
	text = {
		trigger = {
			OR = {
				father ?= { agot_flavour_is_lord_in_westeros_trigger = { CHAR = this } }
				mother ?= { agot_flavour_is_lord_in_westeros_trigger = { CHAR = this } }
			}
		}
		localization_key = child_lord
	}

	# Fallback: Prince
	text = {
		fallback = yes
		localization_key = prince
	}
}
