GetWhiteBookPageTitle = {
	type = character

	text = {
		trigger = {
			exists = house
			NOT = { agot_bastard_surname_trigger = yes }
		}
		localization_key = agot_kingsguard_white_book_title_house
	}

	text = {
		trigger = {
			agot_bastard_surname_trigger = yes
		}
		localization_key = agot_kingsguard_white_book_title_bastard
	}

	text = {
		trigger = {
			has_any_nickname = yes
		}
		localization_key = agot_kingsguard_white_book_title_nickname
	}

	text = {
		localization_key = agot_kingsguard_white_book_title_fallback
		fallback = yes
	}
}
