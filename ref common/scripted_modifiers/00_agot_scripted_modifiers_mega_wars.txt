agot_mw_choose_stance_modifier = {
	# General strategies:
	#

	### Base values
	modifier = { # basic stance values with weighting
		add = {
			if = { # generally you favor your crown
				limit = { $TARGET$ = scope:mw_crown.primary_title }

				add = mw_base_loyalist_value
			}
			else_if = { # basic weight for neutral
				limit = { $TARGET$ = scope:mw_ruler_scope.primary_title }

				add = mw_base_neutral_value
			}
			else_if = { # basic weight for rebel leader
				limit = { $TARGET$ = scope:mw_rebel_leader.primary_title }

				add = mw_base_rebel_value
			}
			else_if = { # basic weight for independence leader
				limit = { $TARGET$ = scope:mw_ruler_scope.capital_barony }

				add = mw_seek_independence_base_value
			}
		}
	}

	# Script value based modifiers
	modifier = { # Give additional weight to a certain stance if your liege chose it and you like him
		trigger = {
			NOR = { # If ruler has special relation we want him to follow that relation instead; this is to correspond to agot_mw_count_choice_is_unlocked_trigger disabling many heavy modifiers for counts
				scope:s_relation_t_crown = yes
				scope:s_relation_t_rebel_leader = yes
			}
		}
		NOT = { scope:mw_ruler_scope.liege ?= scope:mw_ruler_scope }
		OR = { # don't apply this to LPs
			AND = { # realm tier is hegemony
				scope:mw_crown.highest_held_title_tier = tier_hegemony
				scope:mw_ruler_scope.highest_held_title_tier < tier_empire
			}
			AND = { # realm tier is empire
				scope:mw_crown.highest_held_title_tier = tier_empire
				scope:mw_ruler_scope.highest_held_title_tier < tier_kingdom
			}
		}
		trigger_if = { # in mw lite oftimes most dukes are direct vassals of one side
			limit = { scope:mw_crown_story.var:mw_mode ?= flag:mw_lite }
			NOR = {
				scope:mw_ruler_scope.liege ?= scope:mw_crown
				scope:mw_ruler_scope.liege ?= scope:mw_rebel_leader
			}
		}
		trigger_else = { always = yes }

		mw_follow_lieges_choice_value > 0 # to ensure negative values invert choice

		add = {
			if = { # liege is crown, rebel leader or independence rebel leader
				limit = {
					OR = {
						scope:mw_ruler_scope.liege = scope:mw_crown			# check crown option
						scope:mw_ruler_scope.liege = scope:mw_rebel_leader	# check rebel option
					}
				}

				if = { # Direct liege is loyalist
					limit = { scope:mw_ruler_scope.liege = scope:mw_crown }
					switch = {
						trigger = $TARGET$
						scope:mw_crown.primary_title = { add = mw_follow_lieges_choice_value } # Check crown option
						scope:mw_ruler_scope.primary_title = { # Check neutral option
							if = {
								limit = { scope:mw_ruler_scope.mw_follow_lieges_choice_value > 0 }
								subtract = mw_follow_lieges_choice_value
							}
						}
						scope:mw_rebel_leader.primary_title = { subtract = mw_follow_lieges_choice_value } # Check rebel option
						scope:mw_ruler_scope.capital_barony = { subtract = mw_follow_lieges_choice_value } # Check independence leader option
					}
				}
				else_if = { # Direct liege is rebel
					limit = { scope:mw_ruler_scope.liege = scope:mw_rebel_leader }
					switch = {
						trigger = $TARGET$
						scope:mw_crown.primary_title = { subtract = mw_follow_lieges_choice_value } # Check crown option
						scope:mw_ruler_scope.primary_title = { # Check neutral option
							if = {
								limit = { scope:mw_ruler_scope.mw_follow_lieges_choice_value > 0 }
								subtract = mw_follow_lieges_choice_value
							}
						}
						scope:mw_rebel_leader.primary_title = { add = mw_follow_lieges_choice_value } # Check rebel option
						scope:mw_ruler_scope.capital_barony = { subtract = mw_follow_lieges_choice_value } # Check independence leader option
					}
				}
			}
			else_if = { # liege is neither crown nor rebel leader
				limit = {
					OR = {
						scope:mw_crown_story = { is_target_in_variable_list = { name = mw_loyalist_list target = scope:mw_ruler_scope.liege } } # check crown option
						scope:mw_crown_story = { is_target_in_variable_list = { name = mw_neutral_list target = scope:mw_ruler_scope.liege } } # check neutral option
						scope:mw_rebel_story = { is_target_in_variable_list = { name = mw_rebel_supporter_list target = scope:mw_ruler_scope.liege } } # check rebel option
						scope:mw_crown_story = { is_target_in_variable_list = { name = mw_independence_rebel_leader_list target = scope:mw_ruler_scope.liege } } # check independence option, liege
						scope:mw_ruler_scope = { any_liege_or_above = { is_target_in_variable_list = { name = mw_independence_rebels_list target = scope:mw_ruler_scope.liege } } } # check independence option, liege, liege
					}
				}

				if = { # Direct liege is loyalist
					limit = { scope:mw_crown_story = { is_target_in_variable_list = { name = mw_loyalist_list target = scope:mw_ruler_scope.liege } } }
					switch = {
						trigger = $TARGET$
						scope:mw_crown.primary_title = { add = mw_follow_lieges_choice_value } # Check crown option
						scope:mw_ruler_scope.primary_title = { # Check neutral option
							if = {
								limit = { scope:mw_ruler_scope.mw_follow_lieges_choice_value > 0 }
								subtract = mw_follow_lieges_choice_value
							}
						}
						scope:mw_rebel_leader.primary_title = { subtract = mw_follow_lieges_choice_value } # Check rebel option
						scope:mw_ruler_scope.capital_barony = { subtract = mw_follow_lieges_choice_value } # Check independence leader option
					}
				}
				else_if = { # Direct liege is neutral
					limit = { scope:mw_crown_story = { is_target_in_variable_list = { name = mw_neutral_list target = scope:mw_ruler_scope.liege } } }
					switch = {
						trigger = $TARGET$
						scope:mw_crown.primary_title = { subtract = mw_follow_lieges_choice_value } # Check crown option
						scope:mw_ruler_scope.primary_title = { # Check neutral option
							if = {
								limit = { scope:mw_ruler_scope.mw_follow_lieges_choice_value > 0 }
								add = mw_follow_lieges_choice_value
							}
						}
						scope:mw_rebel_leader.primary_title = { subtract = mw_follow_lieges_choice_value } # Check rebel option
						scope:mw_ruler_scope.capital_barony = { subtract = mw_follow_lieges_choice_value } # Check independence leader option
					}
				}
				else_if = { # Direct liege is rebel
					limit = { scope:mw_rebel_story = { is_target_in_variable_list = { name = mw_rebel_supporter_list target = scope:mw_ruler_scope.liege } } }
					switch = {
						trigger = $TARGET$
						scope:mw_crown.primary_title = { subtract = mw_follow_lieges_choice_value } # Check crown option
						scope:mw_ruler_scope.primary_title = { # Check neutral option
							if = {
								limit = { scope:mw_ruler_scope.mw_follow_lieges_choice_value > 0 }
								subtract = mw_follow_lieges_choice_value
							}
						}
						scope:mw_rebel_leader.primary_title = { add = mw_follow_lieges_choice_value } # Check rebel option
						scope:mw_ruler_scope.capital_barony = { subtract = mw_follow_lieges_choice_value } # Check independence leader option
					}
				}
				else_if = { # Direct liege is independence leader
					limit = {
						OR = {
							scope:mw_crown_story = { is_target_in_variable_list = { name = mw_independence_rebel_leader_list target = scope:mw_ruler_scope.liege } }
							scope:mw_ruler_scope = { any_liege_or_above = { is_target_in_variable_list = { name = mw_independence_rebels_list target = scope:mw_ruler_scope.liege } } }
						}
					}
					switch = {
						trigger = $TARGET$
						scope:mw_crown.primary_title = { subtract = mw_follow_lieges_choice_value } # Check crown option
						scope:mw_ruler_scope.primary_title = { # Check neutral option
							if = {
								limit = { scope:mw_ruler_scope.mw_follow_lieges_choice_value > 0 }
								subtract = mw_follow_lieges_choice_value
							}
						}
						scope:mw_rebel_leader.primary_title = { subtract = mw_follow_lieges_choice_value } # Check rebel option
						scope:mw_ruler_scope.capital_barony = { add = mw_follow_lieges_choice_value } # Check independence leader option
					}
				}
			}
		}
	}

	### Opinions and relations
	## Opinions
	opinion_modifier = { # Opinion of crown
		trigger = {
			agot_mw_count_choice_is_unlocked_trigger = yes
			OR = {
				$TARGET$ = scope:mw_crown.primary_title
				$TARGET$ = scope:mw_ruler_scope.primary_title
				$TARGET$ = scope:mw_ruler_scope.capital_barony
			}
		}
		opinion_target = scope:mw_crown
		who = scope:mw_ruler_scope
		multiplier = {
			add = mw_opinion_value_factor_crown # modify opinion weight by factor
			# modify value to be negative
			# --> if opinion is positive value becomes negative, thus decreasing chance for neutral, independence stance
			# --> if opinion is negative value becomes positive, thus increasing chance for neutral, independence stance
			switch = {
				trigger = $TARGET$
				scope:mw_ruler_scope.primary_title = { multiply = -0.3 }
				scope:mw_ruler_scope.capital_barony = { multiply = -0.7 }
			}
		}
	}
	opinion_modifier = { # Opinion of rebel leader
		trigger = {
			agot_mw_count_choice_is_unlocked_trigger = yes
			OR = {
				$TARGET$ = scope:mw_rebel_leader.primary_title
				$TARGET$ = scope:mw_ruler_scope.primary_title
				$TARGET$ = scope:mw_ruler_scope.capital_barony
			}
		}
		opinion_target = scope:mw_rebel_leader
		who = scope:mw_ruler_scope
		multiplier = {
			add = mw_opinion_value_factor_rebel_leader # modify opinion weight by factor
			# modify value to be negative
			# --> if opinion is positive, value becomes negative, thus decreasing chance for neutral, independence stance
			# --> if opinion is negative, value becomes positive, thus increasing chance for neutral, independence stance
			switch = {
				trigger = $TARGET$
				scope:mw_ruler_scope.primary_title = { multiply = -0.3 }
				scope:mw_ruler_scope.capital_barony = { multiply = -0.7 }
			}
		}
	}
	opinion_modifier = { # Opinion of old stance leader | modifier target effectively is neutral stance
		trigger = {
			agot_mw_count_choice_is_unlocked_trigger = yes
			exists = scope:mw_old_stance_leader
			agot_mw_is_in_another_rebel_list_bilateral_trigger = { RULER = scope:mw_ruler_scope TARGET = scope:mw_old_stance_leader }
			OR = {
				$TARGET$ = scope:mw_crown.capital_county		# crown
				$TARGET$ = scope:mw_ruler_scope.primary_title	# neutral
				$TARGET$ = scope:mw_rebel_leader.primary_title	# rebel leader
			}
		}
		opinion_target = scope:mw_old_stance_leader
		who = scope:mw_ruler_scope
		multiplier = {
			add = mw_opinion_value_factor_rebel_leader # modify opinion weight by factor
			# modify value to be negative
			# --> if opinion is positive, value becomes negative, thus decreasing chance for neutral, independence stance
			# --> if opinion is negative, value becomes positive, thus increasing chance for neutral, independence stance
			switch = {
				trigger = $TARGET$
				scope:mw_crown.capital_county = { multiply = -0.3 }
				scope:mw_rebel_leader.primary_title = { multiply = -0.7 }
			}
		}
	}

	## Relations
	## Family, house and dynasty
	modifier = { # Close relations to one of the sides
		OR = {
			scope:s_relation_t_crown = yes
			scope:s_relation_t_rebel_leader = yes
		}
		agot_mw_is_torn_between_sides_trigger = no # Not if char should rather go neutral!

		add = {
			if = { # Check for crown
				limit = { scope:s_relation_t_crown = yes }

				switch = {
					trigger = $TARGET$
					scope:mw_crown.primary_title = { add = scope:s_close_family_crown } # Check crown option
					scope:mw_rebel_leader.primary_title = { # Check rebel option
						subtract = {
							add = scope:s_close_family_crown
							multiply = 0.5
						}
					}
					scope:mw_ruler_scope.capital_barony = { # Check independence leader option
						subtract = {
							add = scope:s_close_family_crown
							multiply = 0.5
						}
					}
				}
			}
			if = { # Check for rebel leader
				limit = { scope:s_relation_t_rebel_leader = yes }

				switch = {
					trigger = $TARGET$
					scope:mw_crown.primary_title = { # Check crown option
						subtract = {
							add = scope:s_close_family_rebel_leader
							multiply = 0.5
						}
					}
					scope:mw_rebel_leader.primary_title = { add = scope:s_close_family_rebel_leader } # Check rebel option
					scope:mw_ruler_scope.capital_barony = { # Check independence leader option
						subtract = {
							add = scope:s_close_family_rebel_leader
							multiply = 0.5
						}
					}
				}
			}
		}
	}
	modifier = { # Same Dynasty
		agot_mw_count_choice_is_unlocked_trigger = yes
		exists = scope:mw_ruler_scope.dynasty
		exists = $TARGET$.holder.dynasty
		scope:mw_ruler_scope.dynasty ?= $TARGET$.holder.dynasty
		NOT = { $TARGET$.holder = { is_close_family_of = scope:mw_ruler_scope } } # We already give bonus for that above

		add = mw_same_dynasty_value
	}
	modifier = { # Check historical loyalties
		trigger = {
			agot_mw_count_choice_is_unlocked_trigger = yes
			exists = scope:mw_ruler_scope.house
			exists = scope:mw_rebel_leader.house
			NOT = { scope:mw_crown.house ?= scope:mw_rebel_leader.house }
			OR = { # Target is...
				$TARGET$ = scope:mw_crown.primary_title # ...crown
				$TARGET$ = scope:mw_rebel_leader.primary_title # ...rebel leader
			}
		}

		scope:mw_ruler_scope = {
			OR = {
				agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_crown.house }
				agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house }
			}
		}

		add = { # if having loyalty to both sides, a neutral bonus is granted via neutral base value
			if = { # Crown side
				limit = {
					scope:mw_ruler_scope = { agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_crown.house } }
				}
				switch = {
					trigger = $TARGET$
					scope:mw_crown.primary_title = { add = mw_historical_loyality_value }
					scope:mw_rebel_leader.primary_title = { subtract = mw_historical_loyality_value }
				}
			}
			if = { # Rebel side
				limit = {
					scope:mw_ruler_scope = { agot_mw_house_has_historical_loyality_to_tigger = { TO_HOUSE = scope:mw_rebel_leader.house } }
				}
				switch = {
					trigger = $TARGET$
					scope:mw_crown.primary_title = { subtract = mw_historical_loyality_value }
					scope:mw_rebel_leader.primary_title = { add = mw_historical_loyality_value }
				}
			}
		}
	}
	modifier = { # Check hostages in crown or rebel court
		trigger = {
			scope:mw_ruler_scope ?= {
				any_home_court_hostage = {
					OR = {
						warden ?= scope:mw_crown
						warden ?= scope:mw_rebel_leader
					}
				}
			}
		}

		agot_mw_is_torn_between_sides_trigger = no # Not if char should rather go neutral!

		add = {
			if = {
				limit = { scope:mw_ruler_scope = { any_home_court_hostage = { warden ?= scope:mw_crown } } }
				switch = {
					trigger = $TARGET$
					#scope:mw_crown.primary_title = { add = mw_has_hostage_crown_value } # Check crown option
					scope:mw_ruler_scope.primary_title = { add = mw_has_hostage_crown_value } # Check neutral option
					scope:mw_rebel_leader.primary_title = { subtract = mw_has_hostage_crown_value } # Check rebel option
					scope:mw_ruler_scope.capital_barony = { subtract = mw_has_hostage_crown_value } # Check independence leader option
				}
			}
			if = {
				limit = { scope:mw_ruler_scope = { any_home_court_hostage = { warden ?= scope:mw_rebel_leader } } }
				switch = {
					trigger = $TARGET$
					scope:mw_crown.primary_title = { subtract = mw_has_hostage_rebel_leader_value } # Check crown option
					scope:mw_ruler_scope.primary_title = { subtract = mw_has_hostage_rebel_leader_value } # Check neutral option
					#scope:mw_rebel_leader.primary_title = { add = mw_has_hostage_rebel_leader_value } # Check rebel option
					scope:mw_ruler_scope.capital_barony = { subtract = mw_has_hostage_rebel_leader_value } # Check independence leader option
				}
			}
		}
	}
	## Politics
	modifier = { # Alliances and oaths of loyalty
		trigger = {
			OR = {
				$TARGET$ = scope:mw_crown.primary_title
				$TARGET$ = scope:mw_ruler_scope.primary_title
				$TARGET$ = scope:mw_rebel_leader.primary_title
			}
			OR = {
				scope:mw_ruler_scope.var:mw_is_loyal_to ?= scope:mw_crown
				scope:mw_ruler_scope.var:mw_is_loyal_to ?= scope:mw_rebel_leader
				agot_mw_has_alliance_with_crown_trigger = yes
				agot_mw_has_alliance_with_rebel_leader_trigger = yes
			}
		}

		agot_mw_is_torn_between_sides_trigger = no # Not if char should rather go neutral!

		add = {
			if = { # Has alliance with one of the sides
				limit = {
					OR = {
						AND = {
							agot_mw_has_alliance_with_crown_trigger = yes
							agot_mw_has_alliance_with_rebel_leader_trigger = no
							NOT = { agot_mw_has_personal_relation_to_TARGET_trigger = { TARGET = scope:mw_rebel_leader } }
							NOT = { var:mw_is_loyal_to ?= scope:mw_rebel_leader } # conflict of conscience --> favour neutral stance
						}
						AND = {
							agot_mw_has_alliance_with_crown_trigger = no
							agot_mw_has_alliance_with_rebel_leader_trigger = yes
							NOT = { agot_mw_has_personal_relation_to_TARGET_trigger = { TARGET = scope:crown } }
							NOT = { var:mw_is_loyal_to ?= scope:mw_crown } # conflict of conscience --> favour neutral stance
						}
					}
				}

				if = { # Crown side
					limit = { agot_mw_has_alliance_with_crown_trigger = yes }
					if = {
						limit = { $TARGET$ = scope:mw_crown.primary_title }
						add = mw_has_alliance_with_side # alliances with TARGET/liege/liege.liege
					}
					if = {
						limit = { $TARGET$ = scope:mw_rebel_leader.primary_title }
						subtract = mw_has_alliance_with_side
					}
				}
				if = { # Rebel side
					limit = { agot_mw_has_alliance_with_rebel_leader_trigger = yes }
					if = {
						limit = { $TARGET$ = scope:mw_rebel_leader.primary_title }
						add = mw_has_alliance_with_side # alliances with TARGET/liege/liege.liege
					}
					if = {
						limit = { $TARGET$ = scope:mw_crown.primary_title }
						subtract = mw_has_alliance_with_side
					}
				}
			}
			else_if = { # scope:mw_ruler_scope has sworn loyality to one of the sides
				limit = {
					OR = {
						AND = {
							var:mw_is_loyal_to ?= scope:mw_crown
							agot_mw_has_alliance_with_rebel_leader_trigger = no # conflict of conscience --> favour neutral stance
							NOT = { agot_mw_has_personal_relation_to_TARGET_trigger = { TARGET = scope:mw_rebel_leader } }
						}
						AND = {
							var:mw_is_loyal_to ?= scope:mw_rebel_leader
							agot_mw_has_alliance_with_crown_trigger = no # conflict of conscience --> favour neutral stance
							NOT = { agot_mw_has_personal_relation_to_TARGET_trigger = { TARGET = scope:mw_crown } }
						}
					}
				}

				if = { # Crown side
					limit = { var:mw_is_loyal_to ?= scope:mw_crown }
					if = {
						limit = { $TARGET$ = scope:mw_crown.primary_title }
						add = mw_oath_to_crown_value # alliances with TARGET/liege/liege.liege
					}
					if = {
						limit = { $TARGET$ = scope:mw_rebel_leader.primary_title }
						subtract = mw_oath_to_crown_value
					}
				}
				if = { # Rebel side
					limit = { var:mw_is_loyal_to ?= scope:mw_rebel_leader }
					if = {
						limit = { $TARGET$ = scope:mw_rebel_leader.primary_title }
						add = mw_oath_to_rebel_leader_value # alliances with TARGET/liege/liege.liege
					}
					if = {
						limit = { $TARGET$ = scope:mw_crown.primary_title }
						subtract = mw_oath_to_rebel_leader_value
