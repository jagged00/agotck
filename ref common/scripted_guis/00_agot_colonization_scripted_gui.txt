agot_county_holder_is_wilderness = {
	scope = province

	is_shown = {
		trigger_if = {
			limit = {
				exists = county
				NOR = {
					county.tier = tier_duchy
					county.tier = tier_kingdom
					county.tier = tier_empire
				}
			}
		}
		county.holder ?= { has_trait = wilderness }
	}
}

agot_can_colonize = {
	scope = character

	# Triggered at yearly_on_action.txt -> and then agot_colonization_events.0006 at agot_colonization_events.txt
	ai_is_valid = { always = no }

	is_valid = {
		exists = root

		gold >= colonize_cost_val

		NOT = { government_has_flag = government_is_nw }
		NOT = { government_has_flag = government_is_uninteractable }

		trigger_if = {
			limit = {
				scope:wilderness ?= { geographical_region = world_westeros_beyond_the_wall }
				culture = { agot_is_wildling_culture = no }
				NOT = {
					any_sub_realm_county = {
						any_neighboring_county = {
							this = scope:wilderness.county
						}
					}
				}
			}
			custom_tooltip = {
				text = colonize_btw
				always = no
			}
		}
		trigger_else_if = {
			limit = {
				primary_title = {
					agot_is_pirate_domicile_title_trigger = yes
				}
			}
			custom_tooltip = {
				text = pirates_must_use_decision_to_colonize
				always = no
			}
		}
		trigger_else = {
			custom_tooltip = {
				text = colonize_border_county_trigger_tooltip
				# Only surrounding lands
				any_sub_realm_county = {
					any_title_to_title_neighboring_and_across_water_county = {
						this = scope:wilderness.county
					}
				}
			}
		}
		any_sub_realm_county = {
			any_title_to_title_neighboring_and_across_water_county = {
				NOT = { has_county_modifier = block_settlement_ability }
			}
		}
		custom_tooltip = {
			text = colonization_cooldown
			NOT = { has_variable = colonization_cooldown }
		}
	}

	## Check agot_colonization_effects.txt
	effect = {
		custom_tooltip = colonize_land_cost_desc
		if = {
			limit = {
				has_government = pirate_government
			}
			custom_tooltip = colonize_land_holding_pirates_desc
		}
		else = {
			custom_tooltip = colonize_land_holding_desc
		}
		hidden_effect = { ai_colonization_effect = { WILDERNESS = scope:wilderness ROOT_SCOPE = root } }
	}
}

agot_settlement_count = {
	scope = character

	is_valid = {
		trigger_if = {
			limit = { exists = root }
			trigger_if = {
				limit = { is_alive = yes }
				has_variable = num_settled_wilderness
				var:num_settled_wilderness > settlement_title_tier_limit
			}
		}
	}

	is_shown = {
		trigger_if = {
			limit = { exists = root }
			trigger_if = {
				limit = { is_alive = yes }
				has_variable = num_settled_wilderness
				var:num_settled_wilderness > 0
			}
		}
	}
}
