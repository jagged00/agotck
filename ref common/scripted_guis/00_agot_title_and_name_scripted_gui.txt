agot_character_title = {
	scope = character
	is_valid = { exists = this }
	effect = {
		save_temporary_scope_as = character

		#1. Assign Character Parameters
		assign_character_parameter_effect = { SCOPE = scope:character }

		#2. Localize Character Title
		if = {
			limit = { scope:ruler_status ?= flag:royal }
			apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:character.primary_title }
		}
		else_if = {
			limit = { scope:spouse_status ?= flag:royal }
			apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:spouse_scope.primary_title }
		}
		else_if = {
			limit = {
				scope:scion_status ?= flag:royal
				NOR = {
					exists = scope:maester
					exists = scope:silent_sister
					exists = scope:septon
					agot_localization_government_trigger = { GOV = nights_watch SPOUSE_SCOPE = scope:spouse_scope }
					agot_localization_government_trigger = { GOV = first_ranger SPOUSE_SCOPE = scope:spouse_scope }
				}
			}
			if = {
				limit = { exists = scope:scion_1_title }
				apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:scion_1_title }
			}
			else_if = {
				limit = { exists = scope:scion_2_title }
				apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:scion_2_title }
			}
			else_if = {
				limit = { exists = scope:scion_3_title }
				apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:scion_3_title }
			}
		}
		else = {
			if = {
				limit = { scope:ruler_status ?= flag:active }
				apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:character.primary_title }
			}
			else_if = {
				limit = { scope:spouse_status ?= flag:active }
				apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:spouse_scope.primary_title }
			}
			else = {
				apply_character_title_data = { SCOPE = scope:character SCOPED_TITLE = scope:dummy_gender }
			}
		}
	}
}

agot_character_title_suffix = {
	scope = character
	is_valid = { exists = this }
	effect = {
		save_temporary_scope_as = character

		#1. Assign Character Parameters
		assign_character_parameter_effect = { SCOPE = scope:character }

		#2. Localize Character Title Suffix
		if = {
			limit = { scope:ruler_status ?= flag:royal }
			apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:character.primary_title }
		}
		else_if = {
			limit = { scope:spouse_status ?= flag:royal }
			apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:spouse_scope.primary_title }
		}
		else_if = {
			limit = {
				scope:scion_status ?= flag:royal
				NOR = {
					exists = scope:maester
					exists = scope:silent_sister
					exists = scope:septon
					agot_localization_government_trigger = { GOV = nights_watch SPOUSE_SCOPE = scope:spouse_scope }
					agot_localization_government_trigger = { GOV = first_ranger SPOUSE_SCOPE = scope:spouse_scope }
				}
			}
			if = {
				limit = { exists = scope:scion_1_title }
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:scion_1_title }
			}
			else_if = {
				limit = { exists = scope:scion_2_title }
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:scion_2_title }
			}
			else_if = {
				limit = { exists = scope:scion_3_title }
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:scion_3_title }
			}
		}
		else = {
			if = {
				limit = {
					OR = {
						AND = {
							scope:ruler_status ?= flag:active
							OR = {
								agot_is_historical_government_trigger = { GOV = lp_feudal }
								agot_is_historical_government_trigger = { GOV = feudal }
							}
						}
						AND = {
							scope:spouse_status ?= flag:active
							scope:spouse_scope ?= {
								OR = {
									agot_is_historical_government_trigger = { GOV = lp_feudal }
									agot_is_historical_government_trigger = { GOV = feudal }
								}
							}
						}
						scope:scion_status ?= flag:active
						AND = {
							exists = scope:councillor
							NOR = {
								exists = scope:knighthood
								exists = scope:maester
								exists = scope:silent_sister
								exists = scope:septon
							}
						}
					}
					exists = scope:regent_status
				}
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:character.liege.primary_title }
			}
			else_if = {
				limit = { scope:ruler_status ?= flag:active }
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:character.primary_title }
			}
			else_if = {
				limit = { scope:spouse_status ?= flag:active }
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:spouse_scope.primary_title }
			}
			else_if = {
				limit = { scope:knighthood ?= yes }
				if = {
					limit = { scope:character ?= { has_trait = kingsguard } }
					apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = title:d_kingsguard }
				}
			}
			else = {
				apply_character_title_suffix_data = { SCOPE = scope:character SCOPED_TITLE = scope:dummy_gender }
			}
		}
	}
}
