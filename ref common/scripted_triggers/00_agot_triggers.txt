agot_death_can_start_feud_trigger = {
	$VICTIM$ = { save_temporary_scope_as = victim }
	trigger_if = {
		limit = {
			exists = scope:victim
			exists = scope:victim.killer
			OR = {
				scope:victim.killer ?= { has_dead_character_flag = should_not_start_feud }
				scope:victim ?= { has_dead_character_flag = should_not_start_feud }
			}
		}
		always = no
	}
	trigger_else = {
		always = yes
	}
}

### Used to set Title and holder names flavour
agot_flavour_is_in_westeros_trigger = {
	trigger_if = {
		limit = { is_alive = no }
		$CHAR$.primary_title.title_province ?= { geographical_region = world_westeros_seven_kingdoms_without_wall }
	}
	trigger_else = {
		agot_flavour_is_living_in_westerosi_realm_trigger = { CHAR = $CHAR$ }
	}
}

agot_flavour_capital_is_in_westerosi_trigger = {
	is_landed = yes
	capital_county.title_province ?= { geographical_region = world_westeros_seven_kingdoms_without_wall }
}

agot_flavour_is_living_in_westerosi_realm_trigger = {
	$CHAR$ = {
		is_landed = yes
		# Baron > Count Liege
		trigger_if = {
			limit = {
				top_liege = { has_variable = pre_war_liege }
			}
			top_liege.var:pre_war_liege = {
				# Count > Duke Liege
				trigger_if = {
					limit = {
						top_liege = { has_variable = pre_war_liege }
					}
					top_liege.var:pre_war_liege = {
						# Duke > King Liege
						trigger_if = {
							limit = {
								top_liege = { has_variable = pre_war_liege }
							}
							top_liege.var:pre_war_liege = {
								# King > Emperor Liege
								trigger_if = {
									limit = {
										top_liege = { has_variable = pre_war_liege }
									}
									top_liege.var:pre_war_liege = {
										# Emperor > Heggy Liege
										trigger_if = {
											limit = {
												top_liege = { has_variable = pre_war_liege }
											}
										}
										trigger_else = {
											top_liege = { agot_flavour_capital_is_in_westerosi_trigger = yes }
										}
									}
								}
								trigger_else = {
									top_liege = { agot_flavour_capital_is_in_westerosi_trigger = yes }
								}
							}
						}
						trigger_else = {
							top_liege = { agot_flavour_capital_is_in_westerosi_trigger = yes }
						}
					}
				}
				trigger_else = {
					top_liege = { agot_flavour_capital_is_in_westerosi_trigger = yes }
				}
			}
		}
		trigger_else = {
			top_liege = { agot_flavour_capital_is_in_westerosi_trigger = yes }
		}
	}
}

agot_flavour_iron_throne_exists_trigger = {
	title:h_the_iron_throne = { is_title_created = yes }
}

agot_flavour_hegemony_is_kingdom_trigger = {
	$CHAR$ ?= {
		trigger_if = {
			limit = { primary_title = title:h_the_iron_throne }
			always = yes
		}
		trigger_else = { # else check dynamic requirements
			agot_flavour_is_in_westeros_trigger = { CHAR = this }
			primary_title.tier = tier_hegemony
		}
	}
}

agot_flavour_palatinate_is_kingdom_trigger = {
	$CHAR$ = {
		highest_held_title_tier >= tier_kingdom
		OR = {
			AND = {
				vassal_contract_has_flag = has_palatinate_contract
				liege = top_liege
			}
			AND = {
				is_alive = yes
				has_variable = mw_obl_special_contract_palatinate
				var:pre_war_liege ?= { agot_is_independent_ruler = yes }
			}
			has_inactive_trait = had_palatinate_contract
		}
	}
}

agot_flavour_is_lord_in_westeros_trigger = {
	$CHAR$ = {
		trigger_if = { # if char is dead check inactive trait
			limit = { is_alive = no }
			has_inactive_trait = agot_was_westerosi_lord
		}
		trigger_else = { # else check dynamic requirements
			agot_flavour_is_living_in_westerosi_realm_trigger = { CHAR = this }
			agot_is_independent_ruler = no
		}
	}
}

agot_flavour_char_special_title_trigger = {
	trigger_if = {
		limit = {
			trigger_if = {
				limit = { is_landed = yes }
				agot_flavour_is_in_westeros_trigger = { CHAR = $CHAR$ }
			}
			trigger_else = {
				OR = {
					NOT = { agot_is_historical_government_trigger = { GOV = lp_feudal } }
					NOT = { agot_is_historical_government_trigger = { GOV = feudal } }
				}
			}
			agot_flavour_iron_throne_exists_trigger = yes
		}
		trigger_if = {
			limit = { is_alive = no }
			historical_independent_ruler = yes
		}
		trigger_else = {
			is_independent_ruler = yes
			NOT = { has_variable = pre_war_liege }
		}
	}
	trigger_else = {
		always = yes
	}
	$CHAR$.primary_title ?= $TITLE$
}

agot_flavour_is_prince_dragonstone_trigger = {
	$CHAR$ ?= {
		# You are Targ: your or your parents hold Dragonstone
		OR = {
			AND = {
				primary_title ?= title:k_dragonstone
				dynasty ?= dynasty:dynn_Targaryen
			}
			mother ?= {
				dynasty ?= dynasty:dynn_Targaryen
				primary_title ?= title:k_dragonstone
			}
			father ?= {
				dynasty ?= dynasty:dynn_Targaryen
				primary_title ?= title:k_dragonstone
			}
		}
	}
}
