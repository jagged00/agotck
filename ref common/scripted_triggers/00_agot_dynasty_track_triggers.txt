agot_fp1_adventure_legacy_track_trigger = {
	dynasty = {
		OR = {
			dynast = {
				culture = {
					OR = {
						has_cultural_pillar = heritage_ironman
						this = culture:westerosi_valyrian
						any_parent_culture_or_above = {
							this = culture:westerosi_valyrian
						}
						this = culture:vineman
						any_parent_culture_or_above = {
							this = culture:vineman
						}
					}
				}
			}
			this = dynasty:dynn_Farman
			this = dynasty:dynn_Estermont
		}
	}
}

agot_fp1_pillage_legacy_track_trigger = {
	dynasty = {
		dynast = {
			OR = {
				culture = {
					OR = {
						#No, you cannot use any scripted triggers here, it will crash the game
						has_variable = wildling_culture
						has_cultural_pillar = heritage_ironman
						has_cultural_tradition = tradition_agot_wildling
						has_cultural_tradition = tradition_practiced_pirates
					}
				}
				government_has_flag = government_is_pirate
			}
		}
	}
}
