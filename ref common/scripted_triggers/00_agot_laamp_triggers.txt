### AGOT Laamp Triggers ###
###########################

#Checks if character is custom landless
agot_is_landless_character = {
	OR = {
		agot_is_landless_nw_character = yes
		agot_is_landless_pirate_character = yes
	}
}

agot_domicile_is_non_nomad_county_tier_domicile_trigger = {
	is_domicile_type = agot_pirate_ship
}

agot_is_landless_pirate_character = {
	government_has_flag = government_is_pirate
	is_landed = no
	is_ruler = yes
	domicile ?= { is_domicile_type = agot_pirate_ship }
}

agot_character_is_valid_to_become_landless_pirate = {
	is_human = yes
	agot_can_be_ruler = yes
	NOT = { has_character_flag = agot_is_at_unknown_place }
	is_imprisoned = no
	health >= fine_health
	OR = {
		ai_boldness >= low_negative_ai_value
		ai_energy >= low_negative_ai_value
	}
	NOR = {
		has_trait = content # Too chill to adventure
		has_trait = craven # Too scared to adventure
		has_trait = lazy # Too lazy to adventure
	}
	trigger_if = {
		limit = {
			exists = primary_spouse
			is_female = yes
		}
		OR = {
			matrilinear_marriage = yes
			has_character_modifier = agot_tomboy_modifier
		}
	}
	trigger_if = {
		limit = {
			exists = primary_spouse
			is_male = yes
		}
		patrilinear_marriage = yes
	}
	OR = {
		save_temporary_scope_as = self
		can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = scope:self }
		AND = {
			is_male = yes
			faith_dominant_gender_male_or_equal = yes
		}
		AND = {
			is_female = yes
			faith_dominant_gender_female_or_equal = yes
		}
	}
	custom_tooltip = {
		text = adventurer_not_nf_head_trigger_tt
		NOT = {
			any_held_title = { is_noble_family_title = yes }
		}
	}
}
